55 /**
56  * A utility class for the unification process required to eliminate aliases.
57  * A substitution maps variables to terms and can be applied as a transformation
58  * to Arguments.
59  */
60 class Substitution {
61  // map type used for internally storing var->term mappings
62  // - note: variables are identified by their names
63  using map_t = std::map<std::string, Own<Argument>>;
64 
65  // the mapping of variables to terms
66  map_t varToTerm;
67 
68 public:
69  // -- Constructors/Destructors --
70 
71  Substitution() = default;
72 
73  Substitution(const std::string& var, const Argument* arg) {
74  varToTerm.insert(std::make_pair(var, souffle::clone(arg)));
75  }
76 
77  ~Substitution() = default;
78 
79  /**
80  * Applies this substitution to the given argument and returns a pointer
81  * to the modified argument.
82  *
83  * @param node the node to be transformed
84  * @return a pointer to the modified or replaced node
85  */
86  Own<Node> operator()(Own<Node> node) const {
87  // create a substitution mapper
88  struct M : public NodeMapper {
89  const map_t& map;
90 
91  M(const map_t& map) : map(map) {}
92 
93  using NodeMapper::operator();
94 
95  Own<Node> operator()(Own<Node> node) const override {
96  // see whether it is a variable to be substituted
97  if (auto var = dynamic_cast<ast::Variable*>(node.get())) {
98  auto pos = map.find(var->getName());
99  if (pos != map.end()) {
100  return souffle::clone(pos->second);
101  }
102  }
103 
104  // otherwise, apply the mapper recursively
105  node->apply(*this);
106  return node;
107  }
108  };
109 
110  // apply the mapper
111  return M(varToTerm)(std::move(node));
112  }
113 
114  /**
115  * A generic, type consistent wrapper of the transformation operation above.
116  */
117  template <typename T>
118  Own<T> operator()(Own<T> node) const {
119  Own<Node> resPtr = (*this)(Own<Node>(node.release()));
120  assert(isA<T>(resPtr.get()) && "Invalid node type mapping.");
121  return Own<T>(dynamic_cast<T*>(resPtr.release()));
122  }
123 
124  /**
125  * Appends the given substitution s to this substitution t such that the
126  * result t' is s composed with t (s o t).
127  * i.e.,
128  * - if t(x) = y, then t'(x) = s(y)
129  * - if s(x) = y, and x is not mapped by t, then t'(x) = y
130  */
131  void append(const Substitution& sub) {
132  // apply substitution on the rhs of all current mappings
133  for (auto& pair : varToTerm) {
134  pair.second = sub(std::move(pair.second));
135  }
136 
137  // append unseen variables to the end
138  for (const auto& pair : sub.varToTerm) {
139  if (varToTerm.find(pair.first) == varToTerm.end()) {
140  // not seen yet, add it in
141  varToTerm.insert(std::make_pair(pair.first, souffle::clone(pair.second)));
142  }
143  }
144  }
145 
146  /** A print function (for debugging) */
147  void print(std::ostream& out) const {
148  out << "{"
149  << join(varToTerm, ",",
150  [](std::ostream& out, const std::pair<const std::string, Own<Argument>>& cur) {
151  out << cur.first << " -> " << *cur.second;
152  })
153  << "}";
154  }
155 
156  friend std::ostream& operator<<(std::ostream& out, const Substitution& s) __attribute__((unused)) {
157  s.print(out);
158  return out;
159  }
160 };

166 class Equation {
167 public:
168  // the two terms to be equivalent
169  Own<Argument> lhs;
170  Own<Argument> rhs;
171 
172  Equation(const Argument& lhs, const Argument& rhs)
173  : lhs(souffle::clone(&lhs)), rhs(souffle::clone(&rhs)) {}
174 
175  Equation(const Argument* lhs, const Argument* rhs) : lhs(souffle::clone(lhs)), rhs(souffle::clone(rhs)) {}
176 
177  Equation(const Equation& other) : lhs(souffle::clone(other.lhs)), rhs(souffle::clone(other.rhs)) {}
178

180

182 
183  /**
184  * Applies the given substitution to both sides of the equation.
185  */
186  void apply(const Substitution& sub) {
187  lhs = sub(std::move(lhs));
188  rhs = sub(std::move(rhs));
189  }
190

201  }

206 Own<Clause> ResolveAliasesTransformer::resolveAliases(const Clause& clause) {
207  // -- utilities --
208 
209  // tests whether something is a variable
210  auto isVar = [&](const Argument& arg) { return isA<ast::Variable>(&arg); };
211 
212  // tests whether something is a record
213  auto isRec = [&](const Argument& arg) { return isA<RecordInit>(&arg); };
214 
215  // tests whether a value `a` occurs in a term `b`
216  auto occurs = [](const Argument& a, const Argument& b) {
217  bool res = false;
218  visitDepthFirst(b, [&](const Argument& arg) { res = (res || (arg == a)); });
219  return res;
220  };
221 
222  // variables appearing as functorless arguments in atoms or records should not
223  // be resolved
224  std::set<std::string> baseGroundedVariables;
225  for (const auto* atom : getBodyLiterals<Atom>(clause)) {
226  for (const Argument* arg : atom->getArguments()) {
227  if (const auto* var = dynamic_cast<const ast::Variable*>(arg)) {
228  baseGroundedVariables.insert(var->getName());
229  }
230  }
231  visitDepthFirst(*atom, [&](const RecordInit& rec) {
232  for (const Argument* arg : rec.getArguments()) {
233  if (const auto* var = dynamic_cast<const ast::Variable*>(arg)) {
234  baseGroundedVariables.insert(var->getName());
235  }
236  }
237  });
238  }
239 
240  // I) extract equations
241  std::vector<Equation> equations;
242  visitDepthFirst(clause, [&](const BinaryConstraint& constraint) {
243  if (isEqConstraint(constraint.getOperator())) {
244  equations.push_back(Equation(constraint.getLHS(), constraint.getRHS()));
245  }
246  });
247 
248  // II) compute unifying substitution
249  Substitution substitution;
250 
251  // a utility for processing newly identified mappings
252  auto newMapping = [&](const std::string& var, const Argument* term) {
253  // found a new substitution
254  Substitution newMapping(var, term);
255 
256  // apply substitution to all remaining equations
257  for (auto& equation : equations) {
258  equation.apply(newMapping);
259  }
260 
261  // add mapping v -> t to substitution
262  substitution.append(newMapping);
263  };
264 
265  while (!equations.empty()) {
266  // get next equation to compute
267  Equation equation = equations.back();
268  equations.pop_back();
269 
270  // shortcuts for left/right
271  const Argument& lhs = *equation.lhs;
272  const Argument& rhs = *equation.rhs;
273 
274  // #1: t = t => skip
275  if (lhs == rhs) {
276  continue;
277  }
278 
279  // #2: [..] = [..] => decompose
280  if (isRec(lhs) && isRec(rhs)) {
281  // get arguments
282  const auto& lhs_args = static_cast<const RecordInit&>(lhs).getArguments();
283  const auto& rhs_args = static_cast<const RecordInit&>(rhs).getArguments();
284 
285  // make sure sizes are identical
286  assert(lhs_args.size() == rhs_args.size() && "Record lengths not equal");
287 
288  // create new equalities
289  for (size_t i = 0; i < lhs_args.size(); i++) {
290  equations.push_back(Equation(lhs_args[i], rhs_args[i]));
291  }
292 
293  continue;
294  }
295 
296  // #3: neither is a variable => skip
297  if (!isVar(lhs) && !isVar(rhs)) {
298  continue;
299  }
300 
301  // #4: v = w => add mapping
302  if (isVar(lhs) && isVar(rhs)) {
303  auto& var = static_cast<const ast::Variable&>(lhs);
304  newMapping(var.getName(), &rhs);
305  continue;
306  }
307 
308  // #5: t = v => swap
309  if (!isVar(lhs)) {
310  equations.push_back(Equation(rhs, lhs));
311  continue;
312  }
313 
314  // now we know lhs is a variable
315  assert(isVar(lhs));
316 
317  // therefore, we have v = t
318  const auto& v = static_cast<const ast::Variable&>(lhs);
319  const Argument& t = rhs;
320 
321  // #6: v occurs in t => skip
322  if (occurs(v, t)) {
323  continue;
324  }
325 
326  assert(!occurs(v, t));
327 
328  // #7: t is a record => add mapping
329  if (isRec(t)) {
330  newMapping(v.getName(), &t);
331  continue;
332  }
333 
334  // #8: v is already grounded => skip
335  auto pos = baseGroundedVariables.find(v.getName());
336  if (pos != baseGroundedVariables.end()) {
337  continue;
338  }
339 
340  // add new mapping
341  newMapping(v.getName(), &t);
342  }
343 
344  // III) compute resulting clause
345  return substitution(souffle::clone(&clause));
346 }

348 Own<Clause> ResolveAliasesTransformer::removeTrivialEquality(const Clause& clause) {
349  Own<Clause> res(cloneHead(&clause));
350 
351  // add all literals, except filtering out t = t constraints
352  for (Literal* literal : clause.getBodyLiterals()) {
353  if (auto* constraint = dynamic_cast<BinaryConstraint*>(literal)) {
354  // TODO: don't filter out `FEQ` constraints, since `x = x` can fail when `x` is a NaN
355  if (isEqConstraint(constraint->getOperator())) {
356  if (*constraint->getLHS() == *constraint->getRHS()) {
357  continue; // skip this one
358  }
359  }
360  }
361 
362  res->addToBody(souffle::clone(literal));
363  }
364 
365  // done
366  return res;
367 }

369 Own<Clause> ResolveAliasesTransformer::removeComplexTermsInAtoms(const Clause& clause) {
370  Own<Clause> res(clause.clone());
371 
372  // get list of atoms
373  std::vector<Atom*> atoms = getBodyLiterals<Atom>(*res);
374 
375  // find all functors in atoms
376  std::vector<const Argument*> terms;
377  for (const Atom* atom : atoms) {
378  for (const Argument* arg : atom->getArguments()) {
379  // ignore if not a functor
380  if (!isA<Functor>(arg)) {
381  continue;
382  }
383 
384  // add this functor if not seen yet
385  if (!any_of(terms, [&](const Argument* cur) { return *cur == *arg; })) {
386  terms.push_back(arg);
387  }
388  }
389  }
390 
391  // find all functors in records too
392  visitDepthFirst(atoms, [&](const RecordInit& rec) {
393  for (const Argument* arg : rec.getArguments()) {
394  // ignore if not a functor
395  if (!isA<Functor>(arg)) {
396  continue;
397  }
398 
399  // add this functor if not seen yet
400  if (!any_of(terms, [&](const Argument* cur) { return *cur == *arg; })) {
401  terms.push_back(arg);
402  }
403  }
404  });
405 
406  // substitute them with new variables (a real map would compare pointers)
407  using substitution_map = std::vector<std::pair<Own<Argument>, Own<ast::Variable>>>;
408  substitution_map termToVar;
409 
410  static int varCounter = 0;
411  for (const Argument* arg : terms) {
412  // create a new mapping for this term
413  auto term = souffle::clone(arg);
414  auto newVariable = mk<ast::Variable>(" _tmp_" + toString(varCounter++));
415  termToVar.push_back(std::make_pair(std::move(term), std::move(newVariable)));
416  }
417 
418  // apply mapping to replace the terms with the variables
419  struct Update : public NodeMapper {
420  const substitution_map& map;
421 
422  Update(const substitution_map& map) : map(map) {}
423 
424  Own<Node> operator()(Own<Node> node) const override {
425  // check whether node needs to be replaced
426  for (const auto& pair : map) {
427  auto& term = pair.first;
428  auto& variable = pair.second;
429 
430  if (*term == *node) {
431  return souffle::clone(variable);
432  }
433  }
434 
435  // continue recursively
436  node->apply(*this);
437  return node;
438  }
439  };
440 
441  // update atoms
442  Update update(termToVar);
443  for (Atom* atom : atoms) {
444  atom->apply(update);
445  }
446 
447  // add the necessary variable constraints to the clause
448  for (const auto& pair : termToVar) {
449  auto& term = pair.first;
450  auto& variable = pair.second;
451 
452  res->addToBody(
453  mk<BinaryConstraint>(BinaryConstraintOp::EQ, souffle::clone(variable), souffle::clone(term)));
454  }
455 
456  return res;
457 }

459 bool ResolveAliasesTransformer::transform(TranslationUnit& translationUnit) {
460  bool changed = false;
461  Program& program = translationUnit.getProgram();
462 
463  // get all clauses
464  std::vector<const Clause*> clauses;
465  visitDepthFirst(program, [&](const Relation& rel) {
466  for (const auto& clause : getClauses(program, rel)) {
467  clauses.push_back(clause);
468  }
469  });
470 
471  // clean all clauses
472  for (const Clause* clause : clauses) {
473  // -- Step 1 --
474  // get rid of aliases
475  Own<Clause> noAlias = resolveAliases(*clause);
476 
477  // clean up equalities
478  Own<Clause> cleaned = removeTrivialEquality(*noAlias);
479 
480  // -- Step 2 --
481  // restore simple terms in atoms
482  Own<Clause> normalised = removeComplexTermsInAtoms(*cleaned);
483 
484  // swap if changed
485  if (*normalised != *clause) {
486  changed = true;
487  program.removeClause(clause);
488  program.addClause(std::move(normalised));
489  }
490  }
491 
492  return changed;
493 }
