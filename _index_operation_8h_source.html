45 public:
46  IndexOperation(Own<RelationReference> r, int ident, RamPattern queryPattern, Own<Operation> nested,
47  std::string profileText = "")
48  : RelationOperation(std::move(r), ident, std::move(nested), std::move(profileText)),
49  queryPattern(std::move(queryPattern)) {
50  assert(getRangePattern().first.size() == getRelation().getArity() &&
51  getRangePattern().second.size() == getRelation().getArity());
52  for (const auto& pattern : queryPattern.first) {
53  assert(pattern != nullptr && "pattern is a null-pointer");
54  }
55  for (const auto& pattern : queryPattern.second) {
56  assert(pattern != nullptr && "pattern is a null-pointer");
57  }

60  /**
61  * @brief Get range pattern
62  * @return A pair of std::vector of pointers to Expression objects
63  * These vectors represent the lower and upper bounds for each attribute in the tuple
64  * <expr1> <= Tuple[level, element] <= <expr2>
65  * We have that at an index for an attribute, its bounds are <expr1> and <expr2> respectively
66  * */
67  std::pair<std::vector<Expression*>, std::vector<Expression*>> getRangePattern() const {
68  return std::make_pair(toPtrVector(queryPattern.first), toPtrVector(queryPattern.second));
69  }

92  IndexOperation* clone() const override {
93  RamPattern resQueryPattern;

95  resQueryPattern.first.emplace_back(i->clone());

101  souffle::clone(&getOperation()), getProfileText());

104  /** @brief Helper method for printing */
105  void printIndex(std::ostream& os) const {
106  // const auto& attrib = getRelation().getAttributeNames();
107  bool first = true;
108  for (unsigned int i = 0; i < getRelation().getArity(); ++i) {
109  // TODO: print proper upper lower/bound
110 
111  // early exit if no upper/lower bounds are defined
112  if (isUndefValue(queryPattern.first[i].get()) && isUndefValue(queryPattern.second[i].get())) {
113  continue;
114  }
115 
116  if (first) {
117  os << " ON INDEX ";
118  first = false;
119  } else {
120  os << " AND ";
121  }
122 
123  // both bounds defined and equal => equality
124  if (!isUndefValue(queryPattern.first[i].get()) && !isUndefValue(queryPattern.second[i].get())) {
125  // print equality when lower bound = upper bound
126  if (*(queryPattern.first[i]) == *(queryPattern.second[i])) {
127  os << "t" << getTupleId() << ".";
128  os << i << " = ";
129  os << *(queryPattern.first[i]);
130  continue;
131  }
132  }
133  // at least one bound defined => inequality
134  if (!isUndefValue(queryPattern.first[i].get()) || !isUndefValue(queryPattern.second[i].get())) {
135  if (!isUndefValue(queryPattern.first[i].get())) {
136  os << *(queryPattern.first[i]) << " <= ";
137  }
138 
139  os << "t" << getTupleId() << ".";
140  os << i;
141 
142  if (!isUndefValue(queryPattern.second[i].get())) {
143  os << " <= " << *(queryPattern.second[i]);
144  }
145 
146  continue;
147  }
148  }
149  }
