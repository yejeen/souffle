184  * An abstract perspective on a data range. A stream is a pair of iterators,
185  * referencing the begin and end of a traversible sequence. A stream can only
186  * be traversed once.
187  */
188 class Stream {
189 public:
190  // the size of the internally maintained buffer, corresponding
191  // to the maximum chunk size retrieved from the source
192  constexpr static int BUFFER_SIZE = 128;
193 
194  // the 'interface' for data sources
195  class Source {
196  public:
197  virtual ~Source() = default;
198 
199  /**
200  * Requests the source to retrieve the next set of elements,
201  * to be stored in an array addressed by the first parameter.
202  * The second parameter states an upper limit for the number
203  * of elements to be retrieved.
204  *
205  * @return the number of elements retrieved, 0 if end has reached.
206  */
207  virtual int load(TupleRef* trg, int max) = 0;
208 
209  /**
210  * Fill the target array in the first parameter with references to the current set of elements.
211  * This is useful for cloning references to this Source. The second parameter limits the range
212  * copied.
213  *
214  * @return the number of elements retrieved, 0 if end has reached.
215  */
216  virtual int reload(TupleRef* trg, int max) = 0;
217 
218  /**
219  * Clone a source with the exact same state
220  */
221  virtual Own<Source> clone() = 0;
222  };
223 
224 private:
225  // the source to read data from
226  Own<Source> source = nullptr;
227 
228  // an internal buffer for decoded elements
229  std::array<TupleRef, BUFFER_SIZE> buffer{};
230 
231  // the current position in the buffer
232  int cur = 0;
233 
234  // the end of valid elements in the buffer
235  int limit = 0;
236 
237 public:
238  Stream(Own<Source>&& src) : source(std::move(src)) {
239  loadNext();
240  }
241 
242  Stream() = default;
243 
244  Stream(Stream& other) = delete;
245 
246  Stream& operator=(Stream& other) = delete;
247 
248  Stream(Stream&& other)
249  : source(std::move(other.source)), buffer(other.buffer), cur(other.cur), limit(other.limit) {}
250 
251  Stream& operator=(Stream&& other) {
252  source = std::move(other.source);
253  // only copy important data
254  std::memcpy(
255  &buffer[other.cur], &other.buffer[other.cur], sizeof(TupleRef) * (other.limit - other.cur));
256  cur = other.cur;
257  limit = other.limit;
258  return *this;
259  }
260 
261  template <typename S>
262  Stream(Own<S>&& src) : Stream(Own<Source>(std::move(src))) {}
263 
264  Own<Stream> clone() const {
265  if (source == nullptr) {
266  return mk<Stream>();
267  }
268  auto newStream = mk<Stream>(source->clone());
269  newStream->source->reload(&newStream->buffer[0], limit);
270  newStream->cur = cur;
271  newStream->limit = limit;
272  return newStream;
273  }

275  /**
276  * The iterator exposed by this stream to iterate through
277  * its elements using a range-based for.
278  */
279  class Iterator : public std::iterator<std::forward_iterator_tag, TupleRef> {
280  Stream* stream = nullptr;
281 
282  public:
283  Iterator() = default;
284 
285  Iterator(Stream& stream) : stream(&stream) {
286  if (stream.cur >= stream.limit) {
287  this->stream = nullptr;
288  }
289  }
290 
291  Iterator& operator++() {
292  ++stream->cur;
293  if (stream->cur < stream->limit) {
294  return *this;
295  }
296  stream->loadNext();
297  if (stream->cur >= stream->limit) {
298  stream = nullptr;
299  }
300  return *this;
301  }
302 
303  const TupleRef& operator*() const {
304  return stream->buffer[stream->cur];
305  }
306 
307  bool operator!=(const Iterator& other) const {
308  return stream != other.stream;
309  }
310 
311  bool operator==(const Iterator& other) const {
312  return stream == other.stream;
313  }
314  };
315 
316  // support for ranged based for loops
317  Iterator begin() {
318  return Iterator(*this);
319  }
320  Iterator end() const {
321  return Iterator();
322  }
323 
324 private:
325  /**
326  * Retrieves the next chunk of elements from the source.
327  */
328  void loadNext() {

482 // -------- generic tuple comparator ----------
483 
484 template <unsigned... Columns>
485 struct comparator;
486 
487 template <unsigned First, unsigned... Rest>
488 struct comparator<First, Rest...> {
489  template <typename T>
490  int operator()(const T& a, const T& b) const {
491  return (a[First] < b[First]) ? -1 : ((a[First] > b[First]) ? 1 : comparator<Rest...>()(a, b));
492  }
493  template <typename T>
494  bool less(const T& a, const T& b) const {
495  return a[First] < b[First] || (a[First] == b[First] && comparator<Rest...>().less(a, b));
496  }
497  template <typename T>
498  bool equal(const T& a, const T& b) const {
499  return a[First] == b[First] && comparator<Rest...>().equal(a, b);
500  }
501 };
502

504 struct comparator<> {
505  template <typename T>
506  int operator()(const T&, const T&) const {
507  return 0;

511  return false;

695 protected:
696  using Entry = typename Structure::element_type;
697  using Hints = typename Structure::operation_hints;
698  static constexpr int Arity = Entry::arity;
699

708  // a source adapter for streaming through data
709  class Source : public Stream::Source {
710  const Order& order;
711 
712  // the begin and end of the stream
713  iter cur;
714  iter end;
715 
716  // an internal buffer for re-ordered elements
717  std::array<Entry, Stream::BUFFER_SIZE> buffer;
718 
719  public:

721  : order(order), cur(std::move(begin)), end(std::move(end)) {}

723  int load(TupleRef* out, int max) override {
724  int c = 0;
725  while (cur != end && c < max) {
726  buffer[c] = order.decode(*cur);
727  out[c] = buffer[c];
728  ++cur;
729  ++c;
730  }
731  return c;
732  }

735  int c = 0;
736  max = std::min(max, Stream::BUFFER_SIZE);
737  while (c < max) {
738  out[c] = buffer[c];
739  ++c;
740  }
741  return c;
742  }

745  auto source = mk<Source>(order, cur, end);
746  source->buffer = this->buffer;
747  return source;
748  }
