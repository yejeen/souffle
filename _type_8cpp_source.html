77 /**
78  * An implementation of a meet operation between sets of types computing
79  * the set of pair-wise greatest common subtypes.
80  */
81 struct sub_type {
82  bool operator()(TypeSet& a, const TypeSet& b) const {
83  // compute result set
84  TypeSet greatestCommonSubtypes = getGreatestCommonSubtypes(a, b);
85 
86  // check whether a should change
87  if (greatestCommonSubtypes == a) {
88  return false;
89  }
90 
91  // update a
92  a = greatestCommonSubtypes;
93  return true;
94  }
95 };

129  * A constraint factory ensuring that all the types associated to the variable
130  * a are subtypes of type b.
131  */
132 TypeConstraint isSubtypeOf(const TypeVar& variable, const Type& type) {
133  struct C : public Constraint<TypeVar> {
134  TypeVar variable;
135  const Type& type;
136 
137  C(TypeVar variable, const Type& type) : variable(std::move(variable)), type(type) {}
138 
139  bool update(Assignment<TypeVar>& assignments) const override {
140  TypeSet& assignment = assignments[variable];
141 
142  if (assignment.isAll()) {
143  assignment = TypeSet(type);
144  return true;
145  }
146 
147  TypeSet newAssignment;
148  for (const Type& t : assignment) {
149  newAssignment.insert(getGreatestCommonSubtypes(t, type));
150  }
151 
152  // check whether there was a change
153  if (assignment == newAssignment) {
154  return false;
155  }
156  assignment = newAssignment;
157  return true;
158  }
159 
160  void print(std::ostream& out) const override {
161  out << variable << " <: " << type.getName();
162  }
163  };
164 
165  return std::make_shared<C>(variable, type);
166 }

174 TypeConstraint hasSuperTypeInSet(const TypeVar& var, TypeSet values) {
175  struct C : public Constraint<TypeVar> {
176  TypeVar var;
177  TypeSet values;
178 
179  C(TypeVar var, TypeSet values) : var(std::move(var)), values(std::move(values)) {}
180 
181  bool update(Assignment<TypeVar>& assigment) const override {
182  // get current value of variable a
183  TypeSet& assigments = assigment[var];
184 
185  // remove all types that are not sub-types of b
186  if (assigments.isAll()) {
187  assigments = values;
188  return true;
189  }
190 
191  TypeSet newAssigments;
192  for (const Type& type : assigments) {
193  bool existsSuperTypeInValues =
194  any_of(values, [&type](const Type& value) { return isSubtypeOf(type, value); });
195  if (existsSuperTypeInValues) {
196  newAssigments.insert(type);

198  }
199  // check whether there was a change
200  if (newAssigments == assigments) {
201  return false;
202  }
203  assigments = newAssigments;
204  return true;
205  }
206

208  out << "∃ t ∈ " << values << ": " << var << " <: t";

210  };

224 /**
225  * Ensure that types of left and right have the same base types.
226  */
227 TypeConstraint subtypesOfTheSameBaseType(const TypeVar& left, const TypeVar& right) {
228  struct C : public Constraint<TypeVar> {
229  TypeVar left;
230  TypeVar right;
231 
232  C(TypeVar left, TypeVar right) : left(std::move(left)), right(std::move(right)) {}
233 
234  bool update(Assignment<TypeVar>& assigment) const override {
235  // get current value of variable a
236  TypeSet& assigmentsLeft = assigment[left];
237  TypeSet& assigmentsRight = assigment[right];
238 
239  // Base types common to left and right variables.
240  TypeSet baseTypes;
241 
242  // Base types present in left/right variable.
243  TypeSet baseTypesLeft;
244  TypeSet baseTypesRight;
245 
246  // Iterate over possible types extracting base types.
247  // Left
248  if (!assigmentsLeft.isAll()) {
249  for (const Type& type : assigmentsLeft) {
250  if (isA<SubsetType>(type) || isA<ConstantType>(type)) {
251  baseTypesLeft.insert(getBaseType(&type));
252  }
253  }
254  }
255  // Right
256  if (!assigmentsRight.isAll()) {
257  for (const Type& type : assigmentsRight) {
258  if (isA<SubsetType>(type) || isA<ConstantType>(type)) {
259  baseTypesRight.insert(getBaseType(&type));
260  }
261  }
262  }
263 
264  TypeSet resultLeft;
265  TypeSet resultRight;
266 
267  // Handle all
268  if (assigmentsLeft.isAll() && assigmentsRight.isAll()) {
269  return false;
270  }
271 
272  // If left xor right is all, assign base types of the other side as possible values.
273  if (assigmentsLeft.isAll()) {
274  assigmentsLeft = baseTypesRight;
275  return true;
276  }
277  if (assigmentsRight.isAll()) {
278  assigmentsRight = baseTypesLeft;
279  return true;
280  }
281 
282  baseTypes = TypeSet::intersection(baseTypesLeft, baseTypesRight);
283 
284  // Allow types if they are subtypes of any of the common base types.
285  for (const Type& type : assigmentsLeft) {
286  bool isSubtypeOfCommonBaseType = any_of(baseTypes.begin(), baseTypes.end(),
287  [&type](const Type& baseType) { return isSubtypeOf(type, baseType); });
288  if (isSubtypeOfCommonBaseType) {
289  resultLeft.insert(type);
290  }
291  }
292 
293  for (const Type& type : assigmentsRight) {
294  bool isSubtypeOfCommonBaseType = any_of(baseTypes.begin(), baseTypes.end(),
295  [&type](const Type& baseType) { return isSubtypeOf(type, baseType); });
296  if (isSubtypeOfCommonBaseType) {
297  resultRight.insert(type);
298  }
299  }
300 
301  // check whether there was a change
302  if (resultLeft == assigmentsLeft && resultRight == assigmentsRight) {
303  return false;
304  }
305  assigmentsLeft = resultLeft;
306  assigmentsRight = resultRight;
307  return true;
308  }
309  //
310  void print(std::ostream& out) const override {
311  out << "∃ t : (" << left << " <: t)"
312  << " ∧ "
313  << "(" << right << " <: t)"
314  << " where t is a base type";
315  }
316  };
317 
318  return std::make_shared<C>(left, right);
319 }

321 /**
322  * Given a set of overloads, wait the list of candidates to reduce to one and then apply its constraints.
323  * NOTE: `subtypeResult` implies that `func <: overload-return-type`, rather than
324  * `func = overload-return-type`. This is required for old type semantics.
325  * See #1296 and tests/semantic/type_system4
326  */
327 TypeConstraint satisfiesOverload(const TypeEnvironment& typeEnv, IntrinsicFunctors overloads, TypeVar result,
328  std::vector<TypeVar> args, bool subtypeResult) {
329  struct C : public Constraint<TypeVar> {
330  // Check if there already was a non-monotonic update
331  mutable bool nonMonotonicUpdate = false;
332 
333  const TypeEnvironment& typeEnv;
334  mutable IntrinsicFunctors overloads;
335  TypeVar result;
336  std::vector<TypeVar> args;
337  bool subtypeResult;
338 
339  C(const TypeEnvironment& typeEnv, IntrinsicFunctors overloads, TypeVar result,
340  std::vector<TypeVar> args, bool subtypeResult)
341  : typeEnv(typeEnv), overloads(std::move(overloads)), result(std::move(result)),
342  args(std::move(args)), subtypeResult(subtypeResult) {}
343 
344  bool update(Assignment<TypeVar>& assigment) const override {
345  auto subtypesOf = [&](const TypeSet& src, TypeAttribute tyAttr) {
346  auto& ty = typeEnv.getConstantType(tyAttr);
347  return src.filter(TypeSet(true), [&](auto&& x) { return isSubtypeOf(x, ty); });
348  };
349 
350  auto possible = [&](TypeAttribute ty, const TypeVar& var) {
351  auto& curr = assigment[var];
352  return curr.isAll() || any_of(curr, [&](auto&& t) { return getTypeAttribute(t) == ty; });
353  };
354 
355  overloads = filterNot(std::move(overloads), [&](const IntrinsicFunctorInfo& x) -> bool {
356  if (!x.variadic && args.size() != x.params.size()) return true; // arity mismatch?
357 
358  for (size_t i = 0; i < args.size(); ++i)
359  if (!possible(x.params[x.variadic ? 0 : i], args[i])) return true;
360 
361  return !possible(x.result, result);
362  });
363 
364  bool changed = false;
365  auto newResult = [&]() -> std::optional<TypeSet> {
366  if (0 == overloads.size()) return TypeSet();
367  if (1 < overloads.size()) return {};
368 
369  auto& overload = overloads.front().get();
370  // `ord` is freakin' magical: it has the signature `a -> Int`.
371  // As a consequence, we might be given non-primitive arguments (i.e. types for which
372  // `TypeEnv::getConstantType` is undefined).
373  // Handle this by not imposing constraints on the arguments.
374  if (overload.op != FunctorOp::ORD) {
375  for (size_t i = 0; i < args.size(); ++i) {
376  auto argTy = overload.params[overload.variadic ? 0 : i];
377  auto& currArg = assigment[args[i]];
378  auto newArg = subtypesOf(currArg, argTy);
379  changed |= currArg != newArg;
380  // 2020-05-09: CI linter says to remove `std::move`, but clang-tidy-10 is happy.
381  currArg = std::move(newArg); // NOLINT
382  }
383  }
384 
385  if (nonMonotonicUpdate || subtypeResult) {
386  return subtypesOf(assigment[result], overload.result);
387  } else {
388  nonMonotonicUpdate = true;
389  return TypeSet{typeEnv.getConstantType(overload.result)};
390  }
391  }();
392 
393  if (newResult) {
394  auto& curr = assigment[result];
395  changed |= curr != *newResult;
396  // 2020-05-09: CI linter says to remove `std::move`, but clang-tidy-10 is happy.
397  curr = std::move(*newResult); // NOLINT
398  }
399 
400  return changed;
401  }
402 
403  void print(std::ostream& out) const override {
404  // TODO (darth_tytus): is this description correct?
405  out << "∃ t : " << result << " <: t where t is a base type";
406  }
407  };
408 
409  return std::make_shared<C>(
410  typeEnv, std::move(overloads), std::move(result), std::move(args), subtypeResult);
411 }

413 /**
414  * Constraint on record type and its elements.
415  */
416 TypeConstraint isSubtypeOfComponent(
417  const TypeVar& elementVariable, const TypeVar& recordVariable, size_t index) {
418  struct C : public Constraint<TypeVar> {
419  TypeVar elementVariable;
420  TypeVar recordVariable;
421  unsigned index;
422 
423  C(TypeVar elementVariable, TypeVar recordVariable, int index)
424  : elementVariable(std::move(elementVariable)), recordVariable(std::move(recordVariable)),
425  index(index) {}
426 
427  bool update(Assignment<TypeVar>& assignment) const override {
428  // get list of types for b
429  const TypeSet& recordTypes = assignment[recordVariable];
430 
431  // if it is (not yet) constrainted => skip
432  if (recordTypes.isAll()) {
433  return false;
434  }
435 
436  // compute new types for element and record
437  TypeSet newElementTypes;
438  TypeSet newRecordTypes;
439 
440  for (const Type& type : recordTypes) {
441  // A type must be either a record type or a subset of a record type
442  if (!isOfKind(type, TypeAttribute::Record)) {
443  continue;
444  }
445 
446  const auto& typeAsRecord = *as<RecordType>(type);
447 
448  // Wrong size => skip.
449  if (typeAsRecord.getFields().size() <= index) {
450  continue;
451  }
452 
453  // Valid type for record.
454  newRecordTypes.insert(type);
455 
456  // and its corresponding field.
457  newElementTypes.insert(*typeAsRecord.getFields()[index]);
458  }
459 
460  // combine with current types assigned to element
461  newElementTypes = getGreatestCommonSubtypes(assignment[elementVariable], newElementTypes);
462 
463  // update values
464  bool changed = false;
465  if (newRecordTypes != recordTypes) {
466  assignment[recordVariable] = newRecordTypes;
467  changed = true;
468  }
469 
470  if (assignment[elementVariable] != newElementTypes) {
471  assignment[elementVariable] = newElementTypes;
472  changed = true;
473  }
474 
475  return changed;
476  }
477 
478  void print(std::ostream& out) const override {
479  out << elementVariable << " <: " << recordVariable << "::" << index;
480  }
481  };
482 
483  return std::make_shared<C>(elementVariable, recordVariable, index);
484 }

487 /* Return a new clause with type-annotated variables */
488 Own<Clause> createAnnotatedClause(
489  const Clause* clause, const std::map<const Argument*, TypeSet> argumentTypes) {
490  // Annotates each variable with its type based on a given type analysis result
491  struct TypeAnnotator : public NodeMapper {
492  const std::map<const Argument*, TypeSet>& types;
493 
494  TypeAnnotator(const std::map<const Argument*, TypeSet>& types) : types(types) {}
495 
496  Own<Node> operator()(Own<Node> node) const override {
497  if (auto* var = dynamic_cast<ast::Variable*>(node.get())) {
498  std::stringstream newVarName;
499  newVarName << var->getName() << "&isin;" << types.find(var)->second;
500  return mk<ast::Variable>(newVarName.str());
501  } else if (auto* var = dynamic_cast<UnnamedVariable*>(node.get())) {
502  std::stringstream newVarName;
503  newVarName << "_"
504  << "&isin;" << types.find(var)->second;
505  return mk<ast::Variable>(newVarName.str());
506  }
507  node->apply(*this);
508  return node;
509  }

512  /* Note:
513  * Because the type of each argument is stored in the form [address -> type-set],
514  * the type-analysis result does not immediately apply to the clone due to differing
515  * addresses.
516  * Two ways around this:
517  * (1) Perform the type-analysis again for the cloned clause
518  * (2) Keep track of the addresses of equivalent arguments in the cloned clause
519  * Method (2) was chosen to avoid having to recompute the analysis each time.
520  */
521  auto annotatedClause = souffle::clone(clause);
522 
523  // Maps x -> y, where x is the address of an argument in the original clause, and y
524  // is the address of the equivalent argument in the clone.
525  std::map<const Argument*, const Argument*> memoryMap;
526 
527  std::vector<const Argument*> originalAddresses;
528  visitDepthFirst(*clause, [&](const Argument& arg) { originalAddresses.push_back(&arg); });
529 
530  std::vector<const Argument*> cloneAddresses;
531  visitDepthFirst(*annotatedClause, [&](const Argument& arg) { cloneAddresses.push_back(&arg); });
532 
533  assert(cloneAddresses.size() == originalAddresses.size());
534 
535  for (size_t i = 0; i < originalAddresses.size(); i++) {
536  memoryMap[originalAddresses[i]] = cloneAddresses[i];
537  }
538 
539  // Map the types to the clause clone
540  std::map<const Argument*, TypeSet> cloneArgumentTypes;
541  for (auto& pair : argumentTypes) {
542  cloneArgumentTypes[memoryMap[pair.first]] = pair.second;
543  }
544 
545  // Create the type-annotated clause
546  TypeAnnotator annotator(cloneArgumentTypes);
547  annotatedClause->apply(annotator);
548  return annotatedClause;
549 }

551 /**
552  * Constraint analysis framework for types.
553  *
554  * The analysis operates on the concept of sinks and sources.
555  * If the atom is negated or is a head then it's a sink,
556  * and we can only extract the kind constraint from it
557  * Otherwise it is a source, and the type of the element must
558  * be a subtype of source attribute.
559  */
560 class TypeConstraintsAnalysis : public ConstraintAnalysis<TypeVar> {
561 public:
562  TypeConstraintsAnalysis(const TranslationUnit& tu) : tu(tu) {}
563 
564 private:
565  const TranslationUnit& tu;
566  const TypeEnvironment& typeEnv = tu.getAnalysis<TypeEnvironmentAnalysis>()->getTypeEnvironment();
567  const Program& program = tu.getProgram();
568  const SumTypeBranchesAnalysis& sumTypesBranches = *tu.getAnalysis<SumTypeBranchesAnalysis>();
569 
570  // Sinks = {head} ∪ {negated atoms}
571  std::set<const Atom*> sinks;
572 
573  void collectConstraints(const Clause& clause) override {
574  sinks.insert(clause.getHead());
575  visitDepthFirstPreOrder(clause, *this);
576  }
577 
578  void visitSink(const Atom& atom) {
579  iterateOverAtom(atom, [&](const Argument& argument, const Type& attributeType) {
580  if (isA<RecordType>(attributeType)) {
581  addConstraint(isSubtypeOf(getVar(argument), getBaseType(&attributeType)));
582  return;
583  }
584  for (auto& constantType : typeEnv.getConstantTypes()) {
585  if (isSubtypeOf(attributeType, constantType)) {
586  addConstraint(isSubtypeOf(getVar(argument), constantType));
587  }
588  }
589  });
590  }
591 
592  void visitAtom(const Atom& atom) override {
593  if (contains(sinks, &atom)) {
594  visitSink(atom);
595  return;
596  }
597 
598  iterateOverAtom(atom, [&](const Argument& argument, const Type& attributeType) {
599  addConstraint(isSubtypeOf(getVar(argument), attributeType));
600  });
601  }
602 
603  void visitNegation(const Negation& cur) override {
604  sinks.insert(cur.getAtom());
605  }
606 
607  void visitStringConstant(const StringConstant& cnst) override {
608  addConstraint(isSubtypeOf(getVar(cnst), typeEnv.getConstantType(TypeAttribute::Symbol)));
609  }

611  void visitNumericConstant(const NumericConstant& constant) override {
612  TypeSet possibleTypes;
613 
614  // Check if the type is given.
615  if (constant.getType().has_value()) {
616  switch (*constant.getType()) {
617  // Insert a type, but only after checking that parsing is possible.
618  case NumericConstant::Type::Int:
619  if (canBeParsedAsRamSigned(constant.getConstant())) {
620  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Signed));
621  }
622  break;
623  case NumericConstant::Type::Uint:
624  if (canBeParsedAsRamUnsigned(constant.getConstant())) {
625  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Unsigned));
626  }
627  break;
628  case NumericConstant::Type::Float:
629  if (canBeParsedAsRamFloat(constant.getConstant())) {
630  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Float));
631  }
632  break;
633  }
634  // Else: all numeric types that can be parsed are valid.
635  } else {
636  if (canBeParsedAsRamSigned(constant.getConstant())) {
637  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Signed));
638  }
639 
640  if (canBeParsedAsRamUnsigned(constant.getConstant())) {
641  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Unsigned));
642  }
643 
644  if (canBeParsedAsRamFloat(constant.getConstant())) {
645  possibleTypes.insert(typeEnv.getConstantType(TypeAttribute::Float));
646  }
647  }
648 
649  addConstraint(hasSuperTypeInSet(getVar(constant), possibleTypes));
650  }

659  void visitFunctor(const Functor& fun) override {
660  auto functorVar = getVar(fun);
661 
662  auto intrFun = as<IntrinsicFunctor>(fun);
663  if (intrFun) {
664  auto argVars = map(intrFun->getArguments(), [&](auto&& x) { return getVar(x); });
665  // The type of the user-defined function might not be set at this stage.
666  // If so then add overloads as alternatives
667  if (!intrFun->getFunctionInfo())
668  addConstraint(satisfiesOverload(typeEnv, functorBuiltIn(intrFun->getFunction()), functorVar,
669  argVars, isInfixFunctorOp(intrFun->getFunction())));
670 
671  // In polymorphic case
672  // We only require arguments to share a base type with a return type.
673  // (instead of, for example, requiring them to be of the same type)
674  // This approach is related to old type semantics
675  // See #1296 and tests/semantic/type_system4
676  if (isInfixFunctorOp(intrFun->getFunction())) {
677  for (auto&& var : argVars)
678  addConstraint(subtypesOfTheSameBaseType(var, functorVar));
679 
680  return;
681  }
682 
683  if (!intrFun->getFunctionInfo()) return;
684  }
685 
686  // add a constraint for the return type of the functor
687  addConstraint(isSubtypeOf(functorVar, typeEnv.getConstantType(fun.getReturnType())));
688 
689  // Special case. Ord returns the ram representation of any object.
690  if (intrFun && intrFun->getFunctionInfo()->op == FunctorOp::ORD) return;
691 
692  auto arguments = fun.getArguments();
693  for (size_t i = 0; i < arguments.size(); ++i) {
694  addConstraint(isSubtypeOf(getVar(arguments[i]), typeEnv.getConstantType(fun.getArgType(i))));
695  }
696  }

699  addConstraint(isSubtypeOf(getVar(counter), typeEnv.getConstantType(TypeAttribute::Signed)));
700  }

702  void visitTypeCast(const ast::TypeCast& typeCast) override {
703  auto& typeName = typeCast.getType();
704  if (!typeEnv.isType(typeName)) {
705  return;
706  }
707 
708  addConstraint(isSubtypeOf(getVar(typeCast), typeEnv.getType(typeName)));
709 
710  // If we are dealing with a constant then its type must be deduced from the cast
711  // Otherwise, expression like: to_string(as(2, float)) couldn't be typed.
712  auto* value = typeCast.getValue();
713 
714  if (isA<Constant>(value)) {
715  addConstraint(isSubtypeOf(getVar(*value), typeEnv.getType(typeName)));
716  }
717  }

726  void visitBranchInit(const BranchInit& adt) override {
727  auto* correspondingType = sumTypesBranches.getType(adt.getConstructor());
728 
729  if (correspondingType == nullptr) {
730  return; // malformed program.
731  }
732 
733  // Sanity check
734  assert(isA<AlgebraicDataType>(correspondingType));
735 
736  // Constraint on the whole branch. $Branch(...) <: ADTtype
737  addConstraint(isSubtypeOf(getVar(adt), *correspondingType));
738 
739  // Constraints on arguments
740  try {
741  auto branchTypes = as<AlgebraicDataType>(correspondingType)->getBranchTypes(adt.getConstructor());
742  auto branchArgs = adt.getArguments();
743 
744  if (branchTypes.size() != branchArgs.size()) {
745  // handled by semantic checker later.
746  throw std::invalid_argument("Invalid arity");
747  }
748 
749  // Add constraints for each of the branch arguments.
750  for (size_t i = 0; i < branchArgs.size(); ++i) {
751  auto argVar = getVar(branchArgs[i]);
752  addConstraint(isSubtypeOf(argVar, *branchTypes[i]));
753  }
754  } catch (...) {
755  return; // Invalid program.
756  }
757  }
758 
759  void visitAggregator(const Aggregator& agg) override {
760  if (agg.getOperator() == AggregateOp::COUNT) {
761  addConstraint(isSubtypeOf(getVar(agg), typeEnv.getConstantType(TypeAttribute::Signed)));
762  } else if (agg.getOperator() == AggregateOp::MEAN) {
763  addConstraint(isSubtypeOf(getVar(agg), typeEnv.getConstantType(TypeAttribute::Float)));
764  } else {
765  addConstraint(hasSuperTypeInSet(getVar(agg), typeEnv.getConstantNumericTypes()));
766  }
767 
768  // If there is a target expression - it should be of the same type as the aggregator.
769  if (auto expr = agg.getTargetExpression()) {
770  addConstraint(isSubtypeOf(getVar(expr), getVar(agg)));
771  addConstraint(isSubtypeOf(getVar(agg), getVar(expr)));
772  }
773  }

775  /**
776  * Utility function.
777  * Iterate over atoms valid pairs of (argument, type-attribute) and apply procedure `map` for its
778  * side-effects.
779  */
780  void iterateOverAtom(const Atom& atom, std::function<void(const Argument&, const Type&)> map) {
781  // get relation
782  auto rel = getAtomRelation(&atom, &program);
783  if (rel == nullptr) {
784  return; // error in input program
785  }
786 
787  auto atts = rel->getAttributes();
788  auto args = atom.getArguments();
789  if (atts.size() != args.size()) {
790  return; // error in input program
791  }
792 
793  for (size_t i = 0; i < atts.size(); i++) {
794  const auto& typeName = atts[i]->getTypeName();
795  if (typeEnv.isType(typeName)) {
796  map(*args[i], typeEnv.getType(typeName));
797  }
798  }
799  }
