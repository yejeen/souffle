53  /** @brief templated method to compute/retrieve an analysis for a translation unit */
54  template <class Analysis>
55  Analysis* getAnalysis() const {
56  static const bool debug = Global::config().has("debug-report");
57  std::string name = Analysis::name;
58  auto it = analyses.find(name);
59  if (it == analyses.end()) {
60  // analysis does not exist yet, create instance and run it.
61  auto analysis = mk<Analysis>(Analysis::name);
62  analysis->run(*this);
63  // output analysis in debug report
64  if (debug) {
65  std::stringstream ramAnalysisStr;
66  ramAnalysisStr << *analysis;
67  if (!ramAnalysisStr.str().empty()) {
68  debugReport.addSection(
69  analysis->getName(), "RAM Analysis " + analysis->getName(), ramAnalysisStr.str());
70  }
71  }
72  // check it hasn't been created by someone else, and insert if not
73  it = analyses.find(name);
74  if (it == analyses.end()) {
75  analyses[name] = std::move(analysis);
76  }
77  }
78  return dynamic_cast<Analysis*>(analyses[name].get());
79  }
