35 /**

37  * @brief Emptiness check for a relation
38  *
39  * Evaluates to true if the given relation is the empty set
40  *
41  * For example:
42  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
43  * (B = ∅)
44  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
45  */

47 public:
48  EmptinessCheck(Own<RelationReference> relRef) : relationRef(std::move(relRef)) {
49  assert(relationRef != nullptr && "Relation reference is a nullptr");
50  }
51 
52  /** @brief Get relation */
53  const Relation& getRelation() const {
54  return *relationRef->get();
55  }
56 
57  std::vector<const Node*> getChildNodes() const override {
58  return {relationRef.get()};
59  }
60 
61  EmptinessCheck* clone() const override {
62  return new EmptinessCheck(souffle::clone(relationRef));
63  }
64 
65  void apply(const NodeMapper& map) override {
66  relationRef = map(std::move(relationRef));
67  }
68 
69 protected:
70  void print(std::ostream& os) const override {
71  os << "(" << getRelation().getName() << " = ∅)";
72  }
73 
74  bool equal(const Node& node) const override {
75  const auto& other = static_cast<const EmptinessCheck&>(node);
