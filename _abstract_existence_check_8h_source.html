37  * @class AbstractExistenceCheck
38  * @brief Abstract existence check for a tuple in a relation
39  */
40 class AbstractExistenceCheck : public Condition {
41 public:
42  AbstractExistenceCheck(Own<RelationReference> relRef, VecOwn<Expression> vals)
43  : relationRef(std::move(relRef)), values(std::move(vals)) {
44  assert(relationRef != nullptr && "Relation reference is a nullptr");
45  for (const auto& v : values) {
46  assert(v != nullptr && "value is a nullptr");
47  }
48  }
49 
50  /** @brief Get relation */
51  const Relation& getRelation() const {
52  return *relationRef->get();
53  }
54 
55  /**
56  * @brief Get arguments of the tuple/pattern
57  * A null pointer element in the vector denotes an unspecified
58  * pattern for a tuple element.
59  */
60  const std::vector<Expression*> getValues() const {
61  return toPtrVector(values);
62  }
63 
64  std::vector<const Node*> getChildNodes() const override {
65  std::vector<const Node*> res = {relationRef.get()};
66  for (const auto& cur : values) {
67  res.push_back(cur.get());
68  }
69  return res;
70  }
71 
72  void apply(const NodeMapper& map) override {
73  relationRef = map(std::move(relationRef));
74  for (auto& val : values) {
75  val = map(std::move(val));
76  }
77  }
78 
79 protected:
80  void print(std::ostream& os) const override {
81  os << "("
82  << join(values, ",",
83  [](std::ostream& out, const Own<Expression>& value) {
84  if (!value) {
85  out << "_";
86  } else {
87  out << *value;
88  }
89  })
90  << ") ∈ " << getRelation().getName();
91  }
92 
93  bool equal(const Node& node) const override {
94  const auto& other = static_cast<const AbstractExistenceCheck&>(node);
95  return equal_ptr(relationRef, other.relationRef) && equal_targets(values, other.values);
96  }
97 
98  /** Relation */
99  Own<RelationReference> relationRef;
100 
101  /** Pattern -- nullptr if undefined */
102  VecOwn<Expression> values;
103 };
