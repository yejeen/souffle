77  class Hasher {
78  public:
79  size_t operator()(const SearchSignature& searchSignature) const {
80  std::size_t seed = searchSignature.arity();
81  for (auto& constraint : searchSignature.constraints) {
82  seed ^= static_cast<size_t>(constraint) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
83  }
84  return seed;
85  }

97  *
98  * This class is a helper class for RamIndex.
99  *
100  * This implements a standard maximum matching algorithm for a bi-partite graph
101  * also known as a marriage problem. Given a set of edges in a bi-partite graph
102  * select a subset of edges that each node in the bi-partite graph has at most
103  * one adjacent edge associated with.
104  *
105  * The nodes of the bi-partite graph represent index-signatures stemming from
106  * RAM operations and RAM existence checks for a relation. A relation between
107  * two nodes represent whether an index operation subsumes another operation.
108  *

110  */

119  * Matching represent a solution of the matching, i.e., which node in the bi-partite
120  * graph maps to another node. If no map exist for a node, there is no adjacent edge
121  * exists for that node.
122  */

283  /** @Brief insert a total order index
284  * @param size of the index
285  */
286  void insertDefaultTotalIndex(size_t arity) {
287  Chain chain = std::vector<SearchSignature>();
288  SearchSignature fullIndexKey = SearchSignature::getFullSearchSignature(arity);
289  chain.push_back(fullIndexKey);
290  chainToOrder.push_back(std::move(chain));
291  LexOrder totalOrder;
292  for (size_t i = 0; i < arity; ++i) {
293  totalOrder.push_back(i);
294  }

311  /** @Brief count the number of constraints in key */
312  static size_t card(SearchSignature cols) {
313  size_t sz = 0;
314  for (size_t i = 0; i < cols.arity(); i++) {

316  sz++;
317  }
318  }

322  /** @Brief maps search columns to an lexicographical order (labeled by a number) */
323  int map(SearchSignature cols) const {
324  assert(orders.size() == chainToOrder.size() && "Order and Chain Sizes do not match!!");
325  int i = 0;
326  for (auto it = chainToOrder.begin(); it != chainToOrder.end(); ++it, ++i) {
327  if (std::find(it->begin(), it->end(), cols) != it->end()) {
328  assert((size_t)i < orders.size());
329  return i;
330  }
331  }
332  fatal("cannot find matching lexicographical order");
333  }

335  /** @Brief insert an index based on the delta */
336  void insertIndex(LexOrder& ids, SearchSignature delta) {
337  LexOrder backlog; // add inequalities at the end
338  for (size_t pos = 0; pos < delta.arity(); pos++) {
339  if (delta[pos] == AttributeConstraint::Equal) {
340  ids.push_back(pos);
341  } else if (delta[pos] == AttributeConstraint::Inequal) {
342  backlog.push_back(pos);
343  }
344  }
345  ids.insert(ids.end(), backlog.begin(), backlog.end());
346  }
