32 /* B-Tree Indirect indexes */
33 class IndirectIndex : public InterpreterIndex {
34 public:
35  using AttributeIndex = uint32_t;
36  using AttributeOrder = std::vector<AttributeIndex>;
37  /* lexicographical comparison operation on two tuple pointers */
38  struct comparator {
39  const AttributeOrder order;
40 
41  /* constructor to initialize state */
42  comparator(AttributeOrder order) : order(std::move(order)) {}
43 
44  /* comparison function */
45  int operator()(const TupleRef& x, const TupleRef& y) const {
46  for (auto& i : order) {
47  if (x[i] < y[i]) {
48  return -1;
49  }
50  if (x[i] > y[i]) {
51  return 1;
52  }
53  }
54  return 0;
55  }
56 
57  /* less comparison */
58  bool less(const TupleRef& x, const TupleRef& y) const {
59  return operator()(x, y) < 0;
60  }
61 
62  /* equal comparison */
63  bool equal(const TupleRef& x, const TupleRef& y) const {
64  for (auto& i : order) {
65  if (x[i] != y[i]) {
66  return false;
67  }
68  }
69  return true;
70  }
71  };
72 
73  /* btree for storing tuple pointers with a given lexicographical order */
74  using index_set = btree_multiset<TupleRef, comparator, std::allocator<TupleRef>, 512>;
75  using Hints = typename index_set::operation_hints;
76 
77  class Source : public Stream::Source {
78  // the begin and end of the stream
79  using iter = btree_multiset<TupleRef, comparator, std::allocator<TupleRef>, 512>::iterator;
80  iter cur;
81  iter end;
82 
83  // an internal buffer for re-ordered elements
84  std::array<TupleRef, Stream::BUFFER_SIZE> buffer;
85 
86  public:
87  Source(iter begin, iter end) : cur(begin), end(end) {}
88 
89  int load(TupleRef* out, int max) override {
90  int c = 0;
91  while (cur != end && c < max) {
92  buffer[c] = *cur;
93  out[c] = buffer[c];
94  ++cur;
95  ++c;
96  }
97  return c;
98  }
99 
100  int reload(TupleRef* out, int max) override {
101  int c = 0;
102  max = std::min(max, Stream::BUFFER_SIZE);
103  while (c < max) {
104  out[c] = buffer[c];
105  ++c;
106  }
107  return c;
108  }
109 
110  Own<Stream::Source> clone() override {
111  auto* source = new Source(cur, end);
112  source->buffer = this->buffer;
113  return Own<Stream::Source>(source);
114  }
115  };
