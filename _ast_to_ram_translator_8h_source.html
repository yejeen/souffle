102  struct Location {
103  int identifier{};
104  int element{};
105  Own<ram::RelationReference> relation{nullptr};
106 
107  Location() = default;
108 
109  Location(int ident, int elem, Own<ram::RelationReference> rel = nullptr)
110  : identifier(ident), element(elem), relation(std::move(rel)) {}
111 
112  Location(const Location& l) : identifier(l.identifier), element(l.element) {
113  if (l.relation != nullptr) {
114  relation = souffle::clone(l.relation);
115  }
116  }
117 
118  Location& operator=(Location other) {
119  identifier = other.identifier;
120  element = other.element;
121  relation = std::move(other.relation);
122  return *this;
123  }
124 
125  bool operator==(const Location& loc) const {
126  return identifier == loc.identifier && element == loc.element;
127  }
128 
129  bool operator!=(const Location& loc) const {
130  return !(*this == loc);
131  }
132 
133  bool operator<(const Location& loc) const {
134  return identifier < loc.identifier || (identifier == loc.identifier && element < loc.element);
135  }
136 
137  void print(std::ostream& out) const {
138  out << "(" << identifier << "," << element << ")";
139  }
140 
141  friend std::ostream& operator<<(std::ostream& out, const Location& loc) {
142  loc.print(out);
143  return out;
144  }
145  };

152  class ValueIndex {
153  /**
154  * The type mapping variables (referenced by their names) to the
155  * locations where they are used.
156  */
157  using variable_reference_map = std::map<std::string, std::set<Location>>;
158 
159  /**
160  * The type mapping record init expressions to their definition points,
161  * hence the point where they get grounded/bound.
162  */
163  using record_definition_map = std::map<const ast::RecordInit*, Location>;
164 
165  /**
166  * A map from generative `ast::Argument`s to storage locations. Note,
167  * since in this case ast::Argument are indexed by their values (not their
168  * address) no standard map can be utilized.
169  * (By-value indexing induces an ad-hoc form of CSE.)
170  */
171  using generator_location_map = std::vector<std::pair<const ast::Argument*, Location>>;
172 
173  /** The index of variable accesses */
174  variable_reference_map var_references;
175 
176  /** The index of record definition points */
177  record_definition_map record_definitions;
178 
179  /** The level of a nested ram operation that is handling a generator operation */
180  generator_location_map arg_generator_locations;
181 
182  public:
183  // -- variables --
184 
185  void addVarReference(const ast::Variable& var, const Location& l) {
186  std::set<Location>& locs = var_references[var.getName()];
187  locs.insert(l);
188  }
189 
190  void addVarReference(

192  addVarReference(var, Location({ident, pos, std::move(rel)}));
193  }
194 
195  bool isDefined(const ast::Variable& var) const {
196  return var_references.find(var.getName()) != var_references.end();
197  }
198 
199  const Location& getDefinitionPoint(const ast::Variable& var) const {
200  auto pos = var_references.find(var.getName());
201  assert(pos != var_references.end() && "Undefined variable referenced!");
202  return *pos->second.begin();
203  }
204 
205  const variable_reference_map& getVariableReferences() const {
206  return var_references;
207  }
208 
209  // -- records --

214  record_definitions[&init] = l;
215  }
216 
217  void setRecordDefinition(

219  setRecordDefinition(init, Location({ident, pos, std::move(rel)}));
220  }
221 
222  const Location& getDefinitionPoint(const ast::RecordInit& init) const {
223  auto pos = record_definitions.find(&init);
224  if (pos != record_definitions.end()) {
225  return pos->second;
226  }

231  // -- generators (aggregates & some functors) --
232 
233  void setGeneratorLoc(const ast::Argument& agg, const Location& loc) {
234  arg_generator_locations.push_back(std::make_pair(&agg, loc));
235  }
236 
237  const Location& getGeneratorLoc(const ast::Argument& arg) const {
238  // search list
239  for (const auto& cur : arg_generator_locations) {
240  if (*cur.first == arg) {
241  return cur.second;
242  }
243  }
244 
245  fatal("arg `%s` has no generator location", arg);
246  }
247 
248  // -- others --
249 
250  bool isGenerator(const int level) const {
251  // check for aggregator definitions
252  return any_of(arg_generator_locations,
253  [&level](const auto& location) { return location.second.identifier == level; });
254  }
255 
256  bool isSomethingDefinedOn(int level) const {
257  // check for variable definitions
258  for (const auto& cur : var_references) {
259  if (cur.second.begin()->identifier == level) {
260  return true;
261  }
262  }
263  // check for record definitions
264  for (const auto& cur : record_definitions) {
265  if (cur.second.identifier == level) {
266  return true;
267  }
268  }
269  // nothing defined on this level
270  return false;
271  }

393  /**
394  * Get ram representation of constant.
395  */
396  RamDomain getConstantRamRepresentation(const ast::Constant& constant) {
397  if (auto strConstant = dynamic_cast<const ast::StringConstant*>(&constant)) {
398  return getSymbolTable().lookup(strConstant->getConstant());
399  } else if (isA<ast::NilConstant>(&constant)) {
400  return 0;
401  } else if (auto* numConstant = dynamic_cast<const ast::NumericConstant*>(&constant)) {
402  assert(numConstant->getType().has_value());
403  switch (*numConstant->getType()) {
404  case ast::NumericConstant::Type::Int:
405  return RamSignedFromString(numConstant->getConstant(), nullptr, 0);
406  case ast::NumericConstant::Type::Uint:
407  return RamUnsignedFromString(numConstant->getConstant(), nullptr, 0);
408  case ast::NumericConstant::Type::Float: return RamFloatFromString(numConstant->getConstant());
409  }
410  }
411 
412  fatal("unaccounted-for constant");
413  }
