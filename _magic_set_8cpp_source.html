62 std::set<QualifiedName> MagicSetTransformer::getIgnoredRelations(const TranslationUnit& tu) {
63  Program& program = tu.getProgram();
64  const auto& ioTypes = *tu.getAnalysis<analysis::IOTypeAnalysis>();
65 
66  std::set<QualifiedName> relationsToIgnore;
67 
68  // - Any relations not specified to magic-set
69  std::vector<QualifiedName> specifiedRelations;
70 
71  // Pick up specified relations from config
72  std::vector<std::string> configRels = splitString(Global::config().get("magic-transform"), ',');
73  for (const auto& relStr : configRels) {
74  std::vector<std::string> qualifiers = splitString(relStr, '.');
75  specifiedRelations.push_back(QualifiedName(qualifiers));
76  }
77 
78  // Pick up specified relations from relation tags
79  for (const auto* rel : program.getRelations()) {
80  if (rel->hasQualifier(RelationQualifier::MAGIC)) {
81  specifiedRelations.push_back(rel->getQualifiedName());
82  }
83  }
84 
85  // Get the complement if not everything is magic'd
86  if (!contains(configRels, "*")) {
87  for (const Relation* rel : program.getRelations()) {
88  if (!contains(specifiedRelations, rel->getQualifiedName())) {
89  relationsToIgnore.insert(rel->getQualifiedName());
90  }
91  }
92  }
93 
94  // - Any relations known in constant time (IDB relations)
95  for (auto* rel : program.getRelations()) {
96  // Input relations
97  if (ioTypes.isInput(rel)) {
98  relationsToIgnore.insert(rel->getQualifiedName());
99  continue;
100  }
101 
102  // Any relations not dependent on any atoms
103  bool hasRules = false;
104  for (const auto* clause : getClauses(program, rel->getQualifiedName())) {
105  visitDepthFirst(clause->getBodyLiterals(), [&](const Atom& /* atom */) { hasRules = true; });
106  }
107  if (!hasRules) {
108  relationsToIgnore.insert(rel->getQualifiedName());
109  }
110  }
111 
112  // - Any relation with a neglabel
113  visitDepthFirst(program, [&](const Atom& atom) {
114  const auto& qualifiers = atom.getQualifiedName().getQualifiers();
115  if (!qualifiers.empty() && qualifiers[0] == "@neglabel") {
116  relationsToIgnore.insert(atom.getQualifiedName());
117  }
118  });
119 
120  // - Any relation with a clause containing float-related binary constraints
121  const std::set<BinaryConstraintOp> floatOps(
122  {BinaryConstraintOp::FEQ, BinaryConstraintOp::FNE, BinaryConstraintOp::FLE,
123  BinaryConstraintOp::FGE, BinaryConstraintOp::FLT, BinaryConstraintOp::FGT});
124  for (const auto* clause : program.getClauses()) {
125  visitDepthFirst(*clause, [&](const BinaryConstraint& bc) {
126  if (contains(floatOps, bc.getOperator())) {
127  relationsToIgnore.insert(clause->getHead()->getQualifiedName());
128  }
129  });
130  }
131 
132  // - Any relation with a clause containing order-dependent functors
133  const std::set<FunctorOp> orderDepFuncOps(
134  {FunctorOp::MOD, FunctorOp::FDIV, FunctorOp::DIV, FunctorOp::UMOD});
135  for (const auto* clause : program.getClauses()) {
136  visitDepthFirst(*clause, [&](const IntrinsicFunctor& functor) {
137  if (contains(orderDepFuncOps, functor.getFunctionInfo()->op)) {
138  relationsToIgnore.insert(clause->getHead()->getQualifiedName());
139  }
140  });
141  }
142 
143  // - Any eqrel relation
144  for (auto* rel : program.getRelations()) {
145  if (rel->getRepresentation() == RelationRepresentation::EQREL) {
146  relationsToIgnore.insert(rel->getQualifiedName());
147  }
148  }
149 
150  // - Any relation with execution plans
151  for (auto* clause : program.getClauses()) {
152  if (clause->getExecutionPlan() != nullptr) {
153  relationsToIgnore.insert(clause->getHead()->getQualifiedName());
154  }
155  }
156 
157  // - Any atom appearing in a clause containing a counter
158  for (auto* clause : program.getClauses()) {
159  bool containsCounter = false;
160  visitDepthFirst(*clause, [&](const Counter& /* counter */) { containsCounter = true; });
161  if (containsCounter) {
162  visitDepthFirst(
163  *clause, [&](const Atom& atom) { relationsToIgnore.insert(atom.getQualifiedName()); });
164  }
165  }
166 
167  return relationsToIgnore;
168 }

179 bool NormaliseDatabaseTransformer::transform(TranslationUnit& translationUnit) {
180  bool changed = false;
181 
182  /** (1) Partition input and output relations */
183  changed |= partitionIO(translationUnit);
184  if (changed) translationUnit.invalidateAnalyses();
185 
186  /** (2) Separate the IDB from the EDB */
187  changed |= extractIDB(translationUnit);
188  if (changed) translationUnit.invalidateAnalyses();
189 
190  /** (3) Normalise arguments within each clause */
191  changed |= normaliseArguments(translationUnit);
192  if (changed) translationUnit.invalidateAnalyses();
193 
194  /** (4) Querify output relations */
195  changed |= querifyOutputRelations(translationUnit);
196  if (changed) translationUnit.invalidateAnalyses();
197 
198  return changed;
199 }

201 bool NormaliseDatabaseTransformer::partitionIO(TranslationUnit& translationUnit) {
202  Program& program = translationUnit.getProgram();
203  const auto& ioTypes = *translationUnit.getAnalysis<analysis::IOTypeAnalysis>();
204 
205  // Get all relations that are both input and output
206  std::set<QualifiedName> relationsToSplit;
207  for (auto* rel : program.getRelations()) {
208  if (ioTypes.isInput(rel) && (ioTypes.isOutput(rel) || ioTypes.isPrintSize(rel))) {
209  relationsToSplit.insert(rel->getQualifiedName());
210  }
211  }
212 
213  // For each of these relations I, add a new relation I' that's input instead.
214  // The old relation I is no longer input, but copies over the data from I'.
215  for (auto relName : relationsToSplit) {
216  const auto* rel = getRelation(program, relName);
217  assert(rel != nullptr && "relation does not exist");
218  auto newRelName = QualifiedName(relName);
219  newRelName.prepend("@split_in");
220 
221  // Create a new intermediate input relation, I'
222  auto newRelation = mk<Relation>(newRelName);
223  for (const auto* attr : rel->getAttributes()) {
224  newRelation->addAttribute(souffle::clone(attr));
225  }
226 
227  // Add the rule I <- I'
228  auto newClause = mk<Clause>();
229  auto newHeadAtom = mk<Atom>(relName);
230  auto newBodyAtom = mk<Atom>(newRelName);
231  for (size_t i = 0; i < rel->getArity(); i++) {
232  std::stringstream varName;
233  varName << "@var" << i;
234  newHeadAtom->addArgument(mk<ast::Variable>(varName.str()));
235  newBodyAtom->addArgument(mk<ast::Variable>(varName.str()));
236  }
237  newClause->setHead(std::move(newHeadAtom));
238  newClause->addToBody(std::move(newBodyAtom));
239 
240  // New relation I' should be input, original should not
241  std::set<const Directive*> iosToDelete;
242  std::set<Own<Directive>> iosToAdd;
243  for (const auto* io : program.getDirectives()) {
244  if (io->getQualifiedName() == relName && io->getType() == ast::DirectiveType::input) {
245  // New relation inherits the old input rules
246  auto newIO = souffle::clone(io);
247  newIO->setQualifiedName(newRelName);
248  iosToAdd.insert(std::move(newIO));
249 
250  // Original no longer has them
251  iosToDelete.insert(io);
252  }
253  }
254  for (const auto* io : iosToDelete) {
255  program.removeDirective(io);
256  }
257  for (auto& io : iosToAdd) {
258  program.addDirective(souffle::clone(io));
259  }
260 
261  // Add in the new relation and the copy clause
262  program.addRelation(std::move(newRelation));
263  program.addClause(std::move(newClause));
264  }
265 
266  return !relationsToSplit.empty();
267 }

269 bool NormaliseDatabaseTransformer::extractIDB(TranslationUnit& translationUnit) {
270  Program& program = translationUnit.getProgram();
271  const auto& ioTypes = *translationUnit.getAnalysis<analysis::IOTypeAnalysis>();
272 
273  // Helper method to check if an input relation has no associated rules
274  auto isStrictlyEDB = [&](const Relation* rel) {
275  bool hasRules = false;
276  for (const auto* clause : getClauses(program, rel->getQualifiedName())) {
277  visitDepthFirst(clause->getBodyLiterals(), [&](const Atom& /* atom */) { hasRules = true; });
278  }
279  return !hasRules;
280  };
281 
282  // Get all input relations that also have IDB rules attached
283  std::set<QualifiedName> inputRelationNames;
284  for (auto* rel : program.getRelations()) {
285  if (ioTypes.isInput(rel) && !isStrictlyEDB(rel)) {
286  assert(!ioTypes.isOutput(rel) && !ioTypes.isPrintSize(rel) &&
287  "input relations should not be output at this stage");
288  inputRelationNames.insert(rel->getQualifiedName());
289  }
290  }
291 
292  // Add a new intermediate non-input relation for each
293  // These will cover relation appearances in IDB rules
294  std::map<QualifiedName, QualifiedName> inputToIntermediate;
295  for (const auto& inputRelationName : inputRelationNames) {
296  // Give it a unique name
297  QualifiedName intermediateName(inputRelationName);
298  intermediateName.prepend("@interm_in");
299  inputToIntermediate[inputRelationName] = intermediateName;
300 
301  // Add the relation
302  auto intermediateRelation = souffle::clone(getRelation(program, inputRelationName));
303  intermediateRelation->setQualifiedName(intermediateName);
304  program.addRelation(std::move(intermediateRelation));
305  }
306 
307  // Rename them systematically
308  renameAtoms(program, inputToIntermediate);
309 
310  // Add the rule I' <- I
311  for (const auto& inputRelationName : inputRelationNames) {
312  auto queryHead = mk<Atom>(inputToIntermediate.at(inputRelationName));
313  auto queryLiteral = mk<Atom>(inputRelationName);
314 
315  // Give them identical arguments
316  const auto* inputRelation = getRelation(program, inputRelationName);
317  for (size_t i = 0; i < inputRelation->getArity(); i++) {
318  std::stringstream var;
319  var << "@query_x" << i;
320  queryHead->addArgument(mk<ast::Variable>(var.str()));
321  queryLiteral->addArgument(mk<ast::Variable>(var.str()));
322  }
323 
324  auto query = mk<Clause>(std::move(queryHead));
325  query->addToBody(std::move(queryLiteral));
326  program.addClause(std::move(query));
327  }
328 
329  return !inputRelationNames.empty();
330 }

332 bool NormaliseDatabaseTransformer::querifyOutputRelations(TranslationUnit& translationUnit) {
333  Program& program = translationUnit.getProgram();
334 
335  // Helper method to check if a relation is a single-rule output query
336  auto isStrictlyOutput = [&](const Relation* rel) {
337  bool strictlyOutput = true;
338  size_t ruleCount = 0;
339 
340  for (const auto* clause : program.getClauses()) {
341  // Check if the relation is used in the body of any rules
342  visitDepthFirst(clause->getBodyLiterals(), [&](const Atom& atom) {
343  if (atom.getQualifiedName() == rel->getQualifiedName()) {
344  strictlyOutput = false;
345  }
346  });
347 
348  // Keep track of number of rules defining the relation
349  if (clause->getHead()->getQualifiedName() == rel->getQualifiedName()) {
350  ruleCount++;
351  }
352  }
353 
354  return strictlyOutput && ruleCount <= 1;
355  };
356 
357  // Get all output relations that need to be normalised
358  const auto& ioTypes = *translationUnit.getAnalysis<analysis::IOTypeAnalysis>();
359  std::set<QualifiedName> outputRelationNames;
360  for (auto* rel : program.getRelations()) {
361  if ((ioTypes.isOutput(rel) || ioTypes.isPrintSize(rel)) && !isStrictlyOutput(rel)) {
362  assert(!ioTypes.isInput(rel) && "output relations should not be input at this stage");
363  outputRelationNames.insert(rel->getQualifiedName());
364  }
365  }
366 
367  // Add a new intermediate non-output relation for each
368  // These will cover relation appearances in intermediate rules
369  std::map<QualifiedName, QualifiedName> outputToIntermediate;
370  for (const auto& outputRelationName : outputRelationNames) {
371  // Give it a unique name
372  QualifiedName intermediateName(outputRelationName);
373  intermediateName.prepend("@interm_out");
374  outputToIntermediate[outputRelationName] = intermediateName;
375 
376  // Add the relation
377  auto intermediateRelation = souffle::clone(getRelation(program, outputRelationName));
378  intermediateRelation->setQualifiedName(intermediateName);
379  program.addRelation(std::move(intermediateRelation));
380  }
381 
382  // Rename them systematically
383  renameAtoms(program, outputToIntermediate);
384 
385  // Add the rule I <- I'
386  for (const auto& outputRelationName : outputRelationNames) {
387  auto queryHead = mk<Atom>(outputRelationName);
388  auto queryLiteral = mk<Atom>(outputToIntermediate.at(outputRelationName));
389 
390  // Give them identical arguments
391  const auto* outputRelation = getRelation(program, outputRelationName);
392  for (size_t i = 0; i < outputRelation->getArity(); i++) {
393  std::stringstream var;
394  var << "@query_x" << i;
395  queryHead->addArgument(mk<ast::Variable>(var.str()));
396  queryLiteral->addArgument(mk<ast::Variable>(var.str()));
397  }
398  auto query = mk<Clause>(std::move(queryHead));
399  query->addToBody(std::move(queryLiteral));
400  program.addClause(std::move(query));
401  }
402 
403  return !outputRelationNames.empty();
404 }

406 bool NormaliseDatabaseTransformer::normaliseArguments(TranslationUnit& translationUnit) {
407  Program& program = translationUnit.getProgram();
408 
409  // Replace all non-variable-arguments nested inside the node with named variables
410  // Also, keeps track of constraints to add to keep the clause semantically equivalent
411  struct argument_normaliser : public NodeMapper {
412  std::set<Own<BinaryConstraint>>& constraints;
413  int& changeCount;
414 
415  argument_normaliser(std::set<Own<BinaryConstraint>>& constraints, int& changeCount)
416  : constraints(constraints), changeCount(changeCount) {}
417 
418  Own<Node> operator()(Own<Node> node) const override {
419  if (auto* aggr = dynamic_cast<Aggregator*>(node.get())) {
420  // Aggregator variable scopes should be maintained, so changes shouldn't propagate
421  // above this level.
422  std::set<Own<BinaryConstraint>> subConstraints;
423  argument_normaliser aggrUpdate(subConstraints, changeCount);
424  aggr->apply(aggrUpdate);
425 
426  // Add the constraints to this level
427  std::vector<Own<Literal>> newBodyLiterals;
428  for (const auto* lit : aggr->getBodyLiterals()) {
429  newBodyLiterals.push_back(souffle::clone(lit));
430  }
431  for (auto& constr : subConstraints) {
432  newBodyLiterals.push_back(souffle::clone(constr));
433  }
434 
435  // Update the node to reflect normalised aggregator
436  node = aggr->getTargetExpression() != nullptr
437  ? mk<Aggregator>(aggr->getOperator(),
438  souffle::clone(aggr->getTargetExpression()),
439  std::move(newBodyLiterals))
440  : mk<Aggregator>(aggr->getOperator(), nullptr, std::move(newBodyLiterals));
441  } else {
442  // Otherwise, just normalise children as usual.
443  node->apply(*this);
444  }
445 
446  // All non-variables should be normalised
447  if (auto* arg = dynamic_cast<Argument*>(node.get())) {
448  if (!isA<ast::Variable>(arg)) {
449  std::stringstream name;
450  name << "@abdul" << changeCount++;
451 
452  // Unnamed variables don't need a new constraint, just give them a name
453  if (isA<UnnamedVariable>(arg)) {
454  return mk<ast::Variable>(name.str());
455  }
456 
457  // Link other variables back to their original value with a `<var> = <arg>` constraint
458  constraints.insert(mk<BinaryConstraint>(
459  BinaryConstraintOp::EQ, mk<ast::Variable>(name.str()), souffle::clone(arg)));
460  return mk<ast::Variable>(name.str());
461  }
462  }
463  return node;
464  }
465  };
466 
467  // Transform each clause so that all arguments are:
468  // 1) a variable, or
469  // 2) the RHS of a `<var> = <arg>` constraint
470  bool changed = false;
471  for (auto* clause : program.getClauses()) {
472  int changeCount = 0;
473  std::set<Own<BinaryConstraint>> constraintsToAdd;
474  argument_normaliser update(constraintsToAdd, changeCount);
475 
476  // Apply to each clause head
477  clause->getHead()->apply(update);
478 
479  // Apply to each body literal that isn't already a `<var> = <arg>` constraint
480  for (Literal* lit : clause->getBodyLiterals()) {
481  if (auto* bc = dynamic_cast<BinaryConstraint*>(lit)) {
482  if (bc->getOperator() == BinaryConstraintOp::EQ && isA<ast::Variable>(bc->getLHS())) {
483  continue;
484  }
485  }
486  lit->apply(update);
487  }
488 
489  // Also apply to each record
490  visitDepthFirst(*clause, [&](const RecordInit& rec) {
491  for (Argument* arg : rec.getArguments()) {
492  arg->apply(update);
493  }
494  });
495 
496  // Add each necessary new constraint to the clause
497  for (auto& constraint : constraintsToAdd) {
498  clause->addToBody(souffle::clone(constraint));
499  }
500 
501  changed |= changeCount != 0;
502  }
503 
504  return changed;
505 }

517 Own<Clause> AdornDatabaseTransformer::adornClause(const Clause* clause, const std::string& adornmentMarker) {
518  const auto& relName = clause->getHead()->getQualifiedName();
519  const auto& headArgs = clause->getHead()->getArguments();
520  BindingStore variableBindings(clause);
521 
522  /* Note that variables can be bound through:
523  * (1) an appearance in a body atom (strong)
524  * (2) an appearance in a bound field of the head atom (weak)
525  * (3) equality with a fully bound functor (via dependency analysis)
526  *
527  * When computing (3), appearances (1) and (2) must be separated to maintain the termination semantics of
528  * the original program. Functor variables are not considered bound if they are only bound via the head.
529  *
530  * Justification: Suppose a new variable Y is marked as bound because of its appearance in a functor
531  * Y=X+1, and X was already found to be bound:
532  * (1) If X was bound through a body atom, then the behaviour of typical magic-set is exhibited, where
533  * the magic-set of Y is bounded by the values that X can take, which is bounded by induction.
534  * (2) If X was bound only through the head atom, then Y is only fixed to an appearance in a magic-atom.
535  * In the presence of recursion, this can potentially lead to an infinitely-sized magic-set for an atom.
536  *
537  * Therefore, bound head atom vars should be marked as weakly bound.
538  */
539  for (size_t i = 0; i < adornmentMarker.length(); i++) {
540  const auto* var = dynamic_cast<ast::Variable*>(headArgs[i]);
541  assert(var != nullptr && "expected only variables in head");
542  if (adornmentMarker[i] == 'b') {
543  variableBindings.bindVariableWeakly(var->getName());
544  }
545  }
546 
547  // Create the adorned clause with an empty body
548  auto adornedClause = mk<Clause>();
549 
550  // Copy over plans if needed
551  if (clause->getExecutionPlan() != nullptr) {
552  assert(contains(relationsToIgnore, clause->getHead()->getQualifiedName()) &&
553  "clauses with plans should be ignored");
554  adornedClause->setExecutionPlan(souffle::clone(clause->getExecutionPlan()));
555  }
556 
557  // Create the head atom
558  auto adornedHeadAtom = mk<Atom>(getAdornmentID(relName, adornmentMarker));
559  assert((adornmentMarker == "" || headArgs.size() == adornmentMarker.length()) &&
560  "adornment marker should correspond to head atom variables");
561  for (const auto* arg : headArgs) {
562  const auto* var = dynamic_cast<const ast::Variable*>(arg);
563  assert(var != nullptr && "expected only variables in head");
564  adornedHeadAtom->addArgument(souffle::clone(var));
565  }
566  adornedClause->setHead(std::move(adornedHeadAtom));
567 
568  // Add in adorned body literals
569  std::vector<Own<Literal>> adornedBodyLiterals;
570  for (const auto* lit : clause->getBodyLiterals()) {
571  if (const auto* negation = dynamic_cast<const Negation*>(lit)) {
572  // Negated atoms should not be adorned, but their clauses should be anyway
573  const auto negatedAtomName = negation->getAtom()->getQualifiedName();
574  assert(contains(relationsToIgnore, negatedAtomName) && "negated atoms should not be adorned");
575  queueAdornment(negatedAtomName, "");
576  }
577 
578  if (!isA<Atom>(lit)) {
579  // Non-atoms are added directly
580  adornedBodyLiterals.push_back(souffle::clone(lit));
581  continue;
582  }
583 
584  const auto* atom = dynamic_cast<const Atom*>(lit);
585  assert(atom != nullptr && "expected atom");
586 
587  // Form the appropriate adornment marker
588  std::stringstream atomAdornment;
589  if (!contains(relationsToIgnore, atom->getQualifiedName())) {
590  for (const auto* arg : atom->getArguments()) {
591  const auto* var = dynamic_cast<const ast::Variable*>(arg);
592  assert(var != nullptr && "expected only variables in atom");
593  atomAdornment << (variableBindings.isBound(var->getName()) ? "b" : "f");
594  }
595  }
596  auto currAtomAdornmentID = getAdornmentID(atom->getQualifiedName(), atomAdornment.str());
597 
598  // Add to the ToDo queue if needed
599  queueAdornment(atom->getQualifiedName(), atomAdornment.str());
600 
601  // Add the adorned version to the clause
602  auto adornedBodyAtom = souffle::clone(atom);
603  adornedBodyAtom->setQualifiedName(currAtomAdornmentID);
604  adornedBodyLiterals.push_back(std::move(adornedBodyAtom));
605 
606  // All arguments are now bound

713  relationsToLabel.insert(relName);

737  }

742  assert(originalRel != nullptr && "unlabelled relation does not exist");

744  labelledRelation->setQualifiedName(getNegativeLabel(relName));
745  program.addRelation(std::move(labelledRelation));
746  }

756 bool PositiveLabellingTransformer::transform(TranslationUnit& translationUnit) {
757  Program& program = translationUnit.getProgram();
758  const auto& sccGraph = *translationUnit.getAnalysis<analysis::SCCGraphAnalysis>();
759  const auto& precedenceGraph = translationUnit.getAnalysis<analysis::PrecedenceGraphAnalysis>()->graph();
760  auto ignoredRelations = getIgnoredRelations(translationUnit);
761 
762  // Partition the strata into neglabelled and regular
763  std::set<size_t> neglabelledStrata;
764  std::map<size_t, size_t> originalStrataCopyCount;
765  for (size_t stratum = 0; stratum < sccGraph.getNumberOfSCCs(); stratum++) {
766  size_t numNeggedRelations = 0;
767  const auto& stratumRels = sccGraph.getInternalRelations(stratum);
768 
769  // Count how many relations in this node are neglabelled
770  for (const auto* rel : stratumRels) {
771  if (isNegativelyLabelled(rel->getQualifiedName())) {
772  numNeggedRelations++;
773  }
774  }
775  assert((numNeggedRelations == 0 || numNeggedRelations == stratumRels.size()) &&
776  "stratum cannot contain a mix of neglabelled and unlabelled relations");
777 
778  if (numNeggedRelations > 0) {
779  // This is a neglabelled stratum that will not be copied
780  neglabelledStrata.insert(stratum);
781  } else {
782  // This is a regular stratum that may be copied
783  originalStrataCopyCount[stratum] = 0;
784  }
785  }
786 
787  // Keep track of strata that depend on each stratum
788  // e.g. T in dependentStrata[S] iff a relation in T depends on a relation in S
789  std::map<size_t, std::set<size_t>> dependentStrata;
790  for (size_t stratum = 0; stratum < sccGraph.getNumberOfSCCs(); stratum++) {
791  dependentStrata[stratum] = std::set<size_t>();
792  }
793  for (const auto* rel : program.getRelations()) {
794  size_t stratum = sccGraph.getSCC(rel);
795  precedenceGraph.visitDepthFirst(rel, [&](const auto* dependentRel) {
796  dependentStrata[stratum].insert(sccGraph.getSCC(dependentRel));
797  });
798  }
799 
800  // Label the positive derived literals in the clauses of neglabelled relations
801  // Need a new copy of those relations up to that point for each
802  for (size_t stratum = 0; stratum < sccGraph.getNumberOfSCCs(); stratum++) {
803  if (!contains(neglabelledStrata, stratum)) continue;
804 
805  // Rename in the current stratum
806  for (const auto* rel : sccGraph.getInternalRelations(stratum)) {
807  assert(isNegativelyLabelled(rel->getQualifiedName()) &&
808  "should only be looking at neglabelled strata");
809  const auto& clauses = getClauses(program, *rel);
810  std::set<QualifiedName> relsToCopy;
811 
812  // Get the unignored unlabelled relations appearing in the rules
813  for (const auto* clause : clauses) {
814  visitDepthFirst(*clause, [&](const Atom& atom) {
815  const auto& name = atom.getQualifiedName();
816  if (!contains(ignoredRelations, name) && !isNegativelyLabelled(name)) {
817  relsToCopy.insert(name);
818  }
819  });
820  }
821 
822  // Positively label them
823  for (auto* clause : clauses) {
824  std::map<QualifiedName, QualifiedName> labelledNames;
825  for (const auto& relName : relsToCopy) {
826  size_t relStratum = sccGraph.getSCC(getRelation(program, relName));
827  size_t copyCount = originalStrataCopyCount.at(relStratum) + 1;
828  labelledNames[relName] = getPositiveLabel(relName, copyCount);
829  }
830  renameAtoms(*clause, labelledNames);
831  }
832  }
833 
834  // Create the rules (from all previous strata) for the newly positive labelled literals
835  for (int preStratum = stratum - 1; preStratum >= 0; preStratum--) {
836  if (contains(neglabelledStrata, preStratum)) continue;
837  if (!contains(dependentStrata[preStratum], stratum)) continue;
838 
839  for (const auto* rel : sccGraph.getInternalRelations(preStratum)) {
840  if (contains(ignoredRelations, rel->getQualifiedName())) continue;
841 
842  for (const auto* clause : getClauses(program, rel->getQualifiedName())) {
843  // Grab the new names for all unignored unlabelled positive atoms
844  std::map<QualifiedName, QualifiedName> labelledNames;
845  visitDepthFirst(*clause, [&](const Atom& atom) {
846  const auto& relName = atom.getQualifiedName();
847  if (contains(ignoredRelations, relName) || isNegativelyLabelled(relName)) return;
848  size_t relStratum = sccGraph.getSCC(getRelation(program, relName));
849  size_t copyCount = originalStrataCopyCount.at(relStratum) + 1;
850  labelledNames[relName] = getPositiveLabel(relName, copyCount);
851  });
852 
853  // Rename atoms accordingly
854  auto labelledClause = souffle::clone(clause);
855  renameAtoms(*labelledClause, labelledNames);
856  program.addClause(std::move(labelledClause));
857  }
858  }
859 
860  originalStrataCopyCount[preStratum]++;
861  }
862  }
863 
864  // Add each copy for each relation in
865  bool changed = false;
866  for (const auto& [stratum, numCopies] : originalStrataCopyCount) {
867  const auto& stratumRels = sccGraph.getInternalRelations(stratum);
868  for (size_t copy = 0; copy < numCopies; copy++) {
869  for (auto* rel : stratumRels) {
870  auto newRelation = souffle::clone(rel);
871  newRelation->setQualifiedName(getPositiveLabel(newRelation->getQualifiedName(), copy + 1));
872  program.addRelation(std::move(newRelation));
873  changed = true;
874  }
875  }
876  }
877  return changed;
878 }

934 void MagicSetCoreTransformer::addRelevantVariables(
935  std::set<std::string>& variables, const std::vector<const BinaryConstraint*> eqConstraints) {
936  // Helper method to check if all variables in an argument are bound
937  auto isFullyBound = [&](const Argument* arg) {
938  bool fullyBound = true;
939  visitDepthFirst(
940  *arg, [&](const ast::Variable& var) { fullyBound &= contains(variables, var.getName()); });
941  return fullyBound;
942  };
943 
944  // Helper method to add all newly relevant variables given a lhs = rhs constraint
945  auto addLocallyRelevantVariables = [&](const Argument* lhs, const Argument* rhs) {
946  const auto* lhsVar = dynamic_cast<const ast::Variable*>(lhs);
947  if (lhsVar == nullptr) return true;
948 
949  // if the rhs is fully bound, lhs is now bound
950  if (!contains(variables, lhsVar->getName())) {
951  if (isFullyBound(rhs)) {
952  variables.insert(lhsVar->getName());
953  return false;
954  } else {
955  return true;
956  }
957  }
958 
959  // if the rhs is a record, and lhs is a bound var, then all rhs vars are bound
960  bool fixpointReached = true;
961  if (const auto* rhsRec = dynamic_cast<const RecordInit*>(rhs)) {
962  for (const auto* arg : rhsRec->getArguments()) {
963  const auto* subVar = dynamic_cast<const ast::Variable*>(arg);
964  assert(subVar != nullptr && "expected only variable arguments");
965  if (!contains(variables, subVar->getName())) {
966  fixpointReached = false;
967  variables.insert(subVar->getName());
968  }
969  }
970  }
971 
972  return fixpointReached;
973  };
974 
975  // Keep adding in relevant variables until we reach a fixpoint
976  bool fixpointReached = false;
977  while (!fixpointReached) {
978  fixpointReached = true;
979  for (const auto* eqConstraint : eqConstraints) {
980  assert(eqConstraint->getOperator() == BinaryConstraintOp::EQ && "expected only eq constraints");
981  fixpointReached &= addLocallyRelevantVariables(eqConstraint->getLHS(), eqConstraint->getRHS());
982  fixpointReached &= addLocallyRelevantVariables(eqConstraint->getRHS(), eqConstraint->getLHS());
983  }
984  }
985 }

1038 bool MagicSetCoreTransformer::transform(TranslationUnit& translationUnit) {
1039  Program& program = translationUnit.getProgram();
1040  std::set<Own<Clause>> clausesToRemove;
1041  std::set<Own<Clause>> clausesToAdd;
1042 
1043  /** Perform the Magic Set Transformation */
1044  for (const auto* clause : program.getClauses()) {
1045  clausesToRemove.insert(souffle::clone(clause));
1046 
1047  const auto* head = clause->getHead();
1048  auto relName = head->getQualifiedName();
1049 
1050  // (1) Add the refined clause
1051  if (!isAdorned(relName)) {
1052  // Unadorned relations need not be refined, as every possible tuple is relevant
1053  clausesToAdd.insert(souffle::clone(clause));
1054  } else {
1055  // Refine the clause with a prepended magic atom
1056  auto magicAtom = createMagicAtom(head);
1057  auto refinedClause = mk<Clause>();
1058  refinedClause->setHead(souffle::clone(head));
1059  refinedClause->addToBody(souffle::clone(magicAtom));
1060  for (auto* literal : clause->getBodyLiterals()) {
1061  refinedClause->addToBody(souffle::clone(literal));
1062  }
1063  clausesToAdd.insert(std::move(refinedClause));
1064  }
1065 
1066  // (2) Add the associated magic rules
1067  std::vector<const BinaryConstraint*> eqConstraints = getBindingEqualityConstraints(clause);
1068  std::vector<Own<Atom>> atomsToTheLeft;
1069  if (isAdorned(relName)) {
1070  // Add the specialising head atom
1071  // Output relations are not specialised, and so the head will not contribute to specialisation
1072  atomsToTheLeft.push_back(createMagicAtom(clause->getHead()));
1073  }
1074  for (const auto* lit : clause->getBodyLiterals()) {
1075  const auto* atom = dynamic_cast<const Atom*>(lit);
1076  if (atom == nullptr) continue;
1077  if (!isAdorned(atom->getQualifiedName())) {
1078  atomsToTheLeft.push_back(souffle::clone(atom));
1079  continue;
1080  }
1081 
1082  // Need to create a magic rule
1083  auto magicClause = createMagicClause(atom, atomsToTheLeft, eqConstraints);
1084  atomsToTheLeft.push_back(souffle::clone(atom));
1085  clausesToAdd.insert(std::move(magicClause));
1086  }
1087  }
1088 
1089  for (auto& clause : clausesToAdd) {
1090  program.addClause(souffle::clone(clause));
1091  }
1092  for (const auto& clause : clausesToRemove) {
1093  program.removeClause(clause.get());
1094  }
1095 
1096  // Add in the magic relations
1097  bool changed = false;
1098  for (const auto* rel : program.getRelations()) {
1099  const auto& origName = rel->getQualifiedName();
1100  if (!isAdorned(origName)) continue;
1101  auto magicRelation = mk<Relation>(getMagicName(origName));
1102  auto attributes = getRelation(program, origName)->getAttributes();
1103  auto adornmentMarker = getAdornment(origName);
1104  for (size_t i = 0; i < attributes.size(); i++) {
1105  if (adornmentMarker[i] == 'b') {
1106  magicRelation->addAttribute(souffle::clone(attributes[i]));
1107  }
1108  }
1109  changed = true;
1110  program.addRelation(std::move(magicRelation));
1111  }
1112  return changed;
1113 }
