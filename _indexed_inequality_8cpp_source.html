37 bool IndexedInequalityTransformer::transformIndexToFilter(Program& program) {
38  bool changed = false;
39 
40  // helper for collecting conditions from filter operations
41  auto addCondition = [](Own<Condition> condition, Own<Condition> c) -> Own<Condition> {
42  if (condition == nullptr) {
43  return c;
44  } else {
45  return mk<Conjunction>(std::move(condition), std::move(c));
46  }
47  };

49  visitDepthFirst(program, [&](const Query& query) {
50  std::function<Own<Node>(Own<Node>)> indexToFilterRewriter = [&](Own<Node> node) -> Own<Node> {
51  // find a IndexOperation
52  if (const IndexOperation* indexOperation = dynamic_cast<IndexOperation*>(node.get())) {
53  auto indexSelection = idxAnalysis->getIndexes(indexOperation->getRelation());
54  auto attributesToDischarge = indexSelection.getAttributesToDischarge(
55  idxAnalysis->getSearchSignature(indexOperation), indexOperation->getRelation());
56  auto pattern = indexOperation->getRangePattern();
57  Own<Condition> condition;
58  RamPattern updatedPattern;
59 
60  for (Expression* p : indexOperation->getRangePattern().first) {
61  updatedPattern.first.emplace_back(p->clone());
62  }
63  for (Expression* p : indexOperation->getRangePattern().second) {
64  updatedPattern.second.emplace_back(p->clone());
65  }
66  for (auto i : attributesToDischarge) {
67  // move constraints out of the indexed inequality and into a conjuction
68  Own<Constraint> lowerBound;
69  Own<Constraint> upperBound;
70  changed = true;
71 
72  if (!isUndefValue(pattern.first[i])) {
73  lowerBound =
74  mk<Constraint>(getGreaterEqualConstraint(
75  indexOperation->getRelation().getAttributeTypes()[i]),
76  mk<TupleElement>(indexOperation->getTupleId(), i),
77  souffle::clone(pattern.first[i]));
78  condition = addCondition(std::move(condition), souffle::clone(lowerBound));
79  }
80 
81  if (!isUndefValue(pattern.second[i])) {
82  upperBound = mk<Constraint>(
83  getLessEqualConstraint(indexOperation->getRelation().getAttributeTypes()[i]),
84  mk<TupleElement>(indexOperation->getTupleId(), i),
85  souffle::clone(pattern.second[i]));
86  condition = addCondition(std::move(condition), souffle::clone(upperBound));
87  }
88 
89  // reset the bounds
90  updatedPattern.first[i] = mk<UndefValue>();
91  updatedPattern.second[i] = mk<UndefValue>();
92  }
93 
94  if (condition) {
95  auto nestedOp = souffle::clone(&indexOperation->getOperation());
96  auto filter = mk<Filter>(souffle::clone(condition), souffle::clone(nestedOp));
97 
98  // need to rewrite the node with the same index operation
99  if (const IndexScan* iscan = dynamic_cast<IndexScan*>(node.get())) {
100  node = mk<IndexScan>(mk<RelationReference>(&iscan->getRelation()),
101  iscan->getTupleId(), std::move(updatedPattern), std::move(filter),
102  iscan->getProfileText());
103  } else if (const ParallelIndexScan* pscan =
104  dynamic_cast<ParallelIndexScan*>(node.get())) {
105  node = mk<ParallelIndexScan>(mk<RelationReference>(&pscan->getRelation()),
106  pscan->getTupleId(), std::move(updatedPattern), std::move(filter),
107  pscan->getProfileText());
108  } else if (const IndexChoice* ichoice = dynamic_cast<IndexChoice*>(node.get())) {
109  node = mk<IndexChoice>(mk<RelationReference>(&ichoice->getRelation()),
110  ichoice->getTupleId(), souffle::clone(&ichoice->getCondition()),
111  std::move(updatedPattern), std::move(filter), ichoice->getProfileText());
112  } else if (const IndexAggregate* iagg = dynamic_cast<IndexAggregate*>(node.get())) {
113  // in the case of an aggregate we must strengthen the condition of the aggregate
114  // it doesn't make sense to nest a filter operation because the aggregate needs the
115  // condition in its scope
116  auto strengthenedCondition = addCondition(
117  Own<Condition>(souffle::clone(&iagg->getCondition())), std::move(condition));
118 
119  node = mk<IndexAggregate>(std::move(nestedOp), iagg->getFunction(),
120  mk<RelationReference>(&iagg->getRelation()),
121  souffle::clone(&iagg->getExpression()), std::move(strengthenedCondition),
122  std::move(updatedPattern), iagg->getTupleId());
123  } else {
124  fatal("New IndexOperation subclass found but not supported while making index.");
125  }
126  }
127  }
128  node->apply(makeLambdaRamMapper(indexToFilterRewriter));
129  return node;
130  };
131  const_cast<Query*>(&query)->apply(makeLambdaRamMapper(indexToFilterRewriter));
132  });
133 
134  visitDepthFirst(program, [&](const Query& query) {
135  std::function<Own<Node>(Own<Node>)> removeEmptyIndexRewriter = [&](Own<Node> node) -> Own<Node> {
136  // find an IndexOperation
137  if (const IndexOperation* indexOperation = dynamic_cast<IndexOperation*>(node.get())) {
138  auto pattern = indexOperation->getRangePattern();
139  size_t length = pattern.first.size();
140  bool foundRealIndexableOperation = false;
141 
142  for (size_t i = 0; i < length; ++i) {
143  // if both bounds are undefined we don't have a box query
144  if (isUndefValue(pattern.first[i]) && isUndefValue(pattern.second[i])) {
145  continue;
146  }
147  // if lower and upper bounds are equal its also not a box query
148  foundRealIndexableOperation = true;
149  break;
150  }
151  if (!foundRealIndexableOperation) {
152  // need to rewrite the node with a semantically equivalent operation to get rid of the
153  // index operation i.e. IndexScan with no indexable attributes -> Scan
154  if (const IndexScan* iscan = dynamic_cast<IndexScan*>(node.get())) {
155  node = mk<Scan>(mk<RelationReference>(&iscan->getRelation()), iscan->getTupleId(),
156  souffle::clone(&iscan->getOperation()), iscan->getProfileText());
157  } else if (const ParallelIndexScan* pscan =
158  dynamic_cast<ParallelIndexScan*>(node.get())) {
159  node = mk<ParallelScan>(mk<RelationReference>(&pscan->getRelation()),
160  pscan->getTupleId(), souffle::clone(&pscan->getOperation()),
161  pscan->getProfileText());
162  } else if (const IndexChoice* ichoice = dynamic_cast<IndexChoice*>(node.get())) {
163  node = mk<Choice>(mk<RelationReference>(&ichoice->getRelation()),
164  ichoice->getTupleId(), souffle::clone(&ichoice->getCondition()),
165  souffle::clone(&ichoice->getOperation()), ichoice->getProfileText());
166  } else if (const IndexAggregate* iagg = dynamic_cast<IndexAggregate*>(node.get())) {
167  node = mk<Aggregate>(souffle::clone(&iagg->getOperation()), iagg->getFunction(),
168  mk<RelationReference>(&iagg->getRelation()),
169  souffle::clone(&iagg->getExpression()), souffle::clone(&iagg->getCondition()),
170  iagg->getTupleId());
171  } else {
172  fatal("New IndexOperation subclass found but not supported while transforming "
173  "index.");
174  }
175  }
176  }
177  node->apply(makeLambdaRamMapper(removeEmptyIndexRewriter));
178  return node;
179  };
180  const_cast<Query*>(&query)->apply(makeLambdaRamMapper(removeEmptyIndexRewriter));
181  });
182  return changed;
183 }
