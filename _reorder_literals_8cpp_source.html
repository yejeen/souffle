39 std::unique_ptr<SipsMetric> ReorderLiteralsTransformer::getSipsFunction(
40  const std::string& sipsChosen, const TranslationUnit& tu) {
41  if (sipsChosen == "strict")
42  return std::make_unique<StrictSips>();
43  else if (sipsChosen == "all-bound")
44  return std::make_unique<AllBoundSips>();
45  else if (sipsChosen == "naive")
46  return std::make_unique<NaiveSips>();
47  else if (sipsChosen == "max-bound")
48  return std::make_unique<MaxBoundSips>();
49  else if (sipsChosen == "max-ratio")
50  return std::make_unique<MaxRatioSips>();
51  else if (sipsChosen == "least-free")
52  return std::make_unique<LeastFreeSips>();
53  else if (sipsChosen == "least-free-vars")
54  return std::make_unique<LeastFreeVarsSips>();
55  else if (sipsChosen == "profile-use")
56  return std::make_unique<ProfileUseSips>(*tu.getAnalysis<analysis::ProfileUseAnalysis>());
57 
58  // default is all-bound
59  return getSipsFunction("all-bound", tu);
60 }

62 Clause* ReorderLiteralsTransformer::reorderClauseWithSips(const SipsMetric& sips, const Clause* clause) {
63  // ignore clauses with fixed execution plans
64  if (clause->getExecutionPlan() != nullptr) {
65  return nullptr;
66  }
67 
68  // get the ordering corresponding to the SIPS
69  std::vector<unsigned int> newOrdering = sips.getReordering(clause);
70 
71  // check if we need a change
72  bool changeNeeded = false;
73  for (unsigned int i = 0; i < newOrdering.size(); i++) {
74  if (newOrdering[i] != i) {
75  changeNeeded = true;
76  }
77  }
78 
79  // reorder if needed
80  return changeNeeded ? reorderAtoms(clause, newOrdering) : nullptr;
81 }

83 bool ReorderLiteralsTransformer::transform(TranslationUnit& translationUnit) {
84  bool changed = false;
85  Program& program = translationUnit.getProgram();
86 
87  // --- SIPS-based static reordering ---
88  // ordering is based on the given SIPS
89 
90  // default SIPS to choose is 'all-bound'
91  std::string sipsChosen = "all-bound";
92  if (Global::config().has("SIPS")) {
93  sipsChosen = Global::config().get("SIPS");
94  }
95  auto sipsFunction = getSipsFunction(sipsChosen, translationUnit);
96 
97  // literal reordering is a rule-local transformation
98  std::vector<Clause*> clausesToRemove;
99 
100  for (Clause* clause : program.getClauses()) {
101  Clause* newClause = reorderClauseWithSips(*sipsFunction, clause);
102  if (newClause != nullptr) {
103  // reordering needed - swap around
104  clausesToRemove.push_back(clause);
105  program.addClause(Own<Clause>(newClause));
106  }
107  }
108 
109  changed |= !clausesToRemove.empty();
110  for (auto* clause : clausesToRemove) {
111  program.removeClause(clause);
112  }
113 
114  // --- profile-guided reordering ---
115  if (Global::config().has("profile-use")) {
116  // parse supplied profile information
117  auto profilerSips = getSipsFunction("profiler", translationUnit);
118 
119  // change the ordering of literals within clauses
120  std::vector<Clause*> clausesToRemove;
121  for (Clause* clause : program.getClauses()) {
122  Clause* newClause = reorderClauseWithSips(*profilerSips, clause);
123  if (newClause != nullptr) {
124  // reordering needed - swap around
125  clausesToRemove.push_back(clause);
126  program.addClause(Own<Clause>(newClause));
127  }
128  }
129 
130  changed |= !clausesToRemove.empty();
131  for (auto* clause : clausesToRemove) {
132  program.removeClause(clause);
133  }
134  }
135 
136  return changed;
137 }
