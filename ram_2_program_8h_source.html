57  Program(VecOwn<Relation> rels, Own<Statement> main, std::map<std::string, Own<Statement>> subs)
58  : relations(std::move(rels)), main(std::move(main)), subroutines(std::move(subs)) {
59  assert(this->main != nullptr && "Main program is a null-pointer");
60  for (const auto& rel : relations) {

104  Program* clone() const override {
105  auto* res = new Program();
106  res->main = souffle::clone(main);
107  for (auto& rel : relations) {
108  res->relations.push_back(souffle::clone(rel));
109  }
110  for (auto& sub : subroutines) {
111  res->subroutines[sub.first] = souffle::clone(sub.second);
112  }
113  std::map<const Relation*, const Relation*> refMap;
114  res->apply(makeLambdaRamMapper([&](Own<Node> node) -> Own<Node> {
115  // rewire relation references to newly cloned relations
116  if (const RelationReference* relRef = dynamic_cast<RelationReference*>(node.get())) {
117  const Relation* rel = refMap[relRef->get()];
118  assert(rel != nullptr && "dangling RAM relation reference");
119  return mk<RelationReference>(rel);
120  } else {
121  return node;
122  }
123  }));
124  return res;
125  }

127  void apply(const NodeMapper& map) override {
128  main = map(std::move(main));
129  for (auto& rel : relations) {
130  rel = map(std::move(rel));
131  }
132  for (auto& sub : subroutines) {
133  sub.second = map(std::move(sub.second));
134  }

142  out << " " << *rel << std::endl;

151  main->print(out, 2);
152  out << " END MAIN" << std::endl;
153  out << "END PROGRAM" << std::endl;
154  }
