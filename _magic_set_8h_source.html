39 /**
40  * Magic Set Transformation.
41  * Involves four stages:
42  * (1) NormaliseDatabaseTransformer, for assumptions to hold
43  * (2) LabelDatabaseTransformer, to support negation
44  * (3) AdornDatabaseTransformer, to annotate information flow
45  * (4) MagicSetCoreTransformer, to perform the core magifying transformation
46  */

229  /** Get the unique identifier corresponding to an adorned predicate. */
230  static QualifiedName getAdornmentID(const QualifiedName& relName, const std::string& adornmentMarker);
231 
232  /** Add an adornment to the ToDo queue if it hasn't been processed before. */
233  void queueAdornment(const QualifiedName& relName, const std::string& adornmentMarker) {
234  auto adornmentID = getAdornmentID(relName, adornmentMarker);
235  if (!contains(headAdornmentsSeen, adornmentID)) {
236  headAdornmentsToDo.insert(std::make_pair(relName, adornmentMarker));
237  headAdornmentsSeen.insert(adornmentID);
238  }
239  }
240 
241  /** Check if any more relations need to be adorned. */
242  bool hasAdornmentToProcess() const {
243  return !headAdornmentsToDo.empty();
244  }
245 
246  /** Pop off the next predicate adornment to process. **/
247  adorned_predicate nextAdornmentToProcess() {
248  assert(hasAdornmentToProcess() && "no adornment to pop");
249  auto headAdornment = *(headAdornmentsToDo.begin());
250  headAdornmentsToDo.erase(headAdornmentsToDo.begin());
251  return headAdornment;
252  }
253 
254  /** Returns the adorned version of a clause. */
255  Own<Clause> adornClause(const Clause* clause, const std::string& adornmentMarker);
