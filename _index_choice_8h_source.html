38 /**
39  * @class IndexChoice
40  * @brief Use an index to find a tuple in a relation such that a given condition holds.
41  *
42  * Only one tuple is returned (if one exists), even
43  * if multiple tuples satisfying the condition exist.
44  *
45  * For example:
46  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
47  * QUERY
48  * ...
49  * CHOICE A AS t1 ON INDEX t1.x=10 AND t1.y = 20
50  * WHERE (t1.x, t1.y) NOT IN A
51  * ...
52  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
53  */
54 class IndexChoice : public IndexOperation, public AbstractChoice {
55 public:
56  IndexChoice(Own<RelationReference> r, int ident, Own<Condition> cond, RamPattern queryPattern,
57  Own<Operation> nested, std::string profileText = "")
58  : IndexOperation(std::move(r), ident, std::move(queryPattern), std::move(nested),
59  std::move(profileText)),
60  AbstractChoice(std::move(cond)) {
61  assert(getRangePattern().first.size() == getRelation().getArity());
62  assert(getRangePattern().second.size() == getRelation().getArity());
63  }
64 
65  void apply(const NodeMapper& map) override {
66  RelationOperation::apply(map);
67  for (auto& pattern : queryPattern.first) {
68  pattern = map(std::move(pattern));
69  }
70  for (auto& pattern : queryPattern.second) {
71  pattern = map(std::move(pattern));
72  }
73  AbstractChoice::apply(map);
74  }

76  std::vector<const Node*> getChildNodes() const override {
77  auto res = IndexOperation::getChildNodes();
78  res.push_back(AbstractChoice::getChildNodes().at(0));
79  return res;
80  }

82  IndexChoice* clone() const override {
83  RamPattern resQueryPattern;
84  for (const auto& i : queryPattern.first) {
85  resQueryPattern.first.emplace_back(i->clone());
86  }
87  for (const auto& i : queryPattern.second) {
88  resQueryPattern.second.emplace_back(i->clone());
89  }
90  auto* res = new IndexChoice(souffle::clone(relationRef), getTupleId(), souffle::clone(condition),
91  std::move(resQueryPattern), souffle::clone(&getOperation()), getProfileText());
92  return res;
93  }
