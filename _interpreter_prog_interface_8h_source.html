134  /** Get current tuple */
135  tuple& operator*() override {
136  // set tuple stream to first element
137  tup.rewind();
138 
139  // construct the tuple to return
140  for (size_t i = 0; i < ramRelationInterface->getArity(); i++) {
141  switch (*(ramRelationInterface->getAttrType(i))) {
142  case 's': {
143  std::string s = ramRelationInterface->getSymbolTable().resolve((*it)[i]);
144  tup << s;
145  break;
146  }
147  case 'f': {
148  tup << ramBitCast<RamFloat>((*it)[i]);
149  break;
150  }
151  case 'u': {
152  tup << ramBitCast<RamUnsigned>((*it)[i]);
153  break;
154  }
155  default: {
156  tup << (*it)[i];
157  break;
158  }
159  }
160  }
161  tup.rewind();
162  return tup;
163  }

207 /**
208  * Implementation of SouffleProgram interface for an interpreter instance
209  */
210 class InterpreterProgInterface : public SouffleProgram {
211 public:
212  explicit InterpreterProgInterface(InterpreterEngine& interp)
213  : prog(interp.getTranslationUnit().getProgram()), exec(interp),
214  symTable(interp.getTranslationUnit().getSymbolTable()) {
215  uint32_t id = 0;
216 
217  // Retrieve AST Relations and store them in a map
218  std::map<std::string, const ram::Relation*> map;
219  visitDepthFirst(prog, [&](const ram::Relation& rel) { map[rel.getName()] = &rel; });
220 
221  // Build wrapper relations for Souffle's interface
222  for (auto& relHandler : exec.getRelationMap()) {
223  if (relHandler == nullptr) {
224  // Skip droped relation.
225  continue;
226  }
227  auto& interpreterRel = **relHandler;
228  auto& name = interpreterRel.getName();
229  assert(map[name]);
230  const ram::Relation& rel = *map[name];
231 
232  // construct types and names vectors
233  std::vector<std::string> types = rel.getAttributeTypes();
234  std::vector<std::string> attrNames = rel.getAttributeNames();
235 
236  auto* interface = new InterpreterRelInterface(
237  interpreterRel, symTable, rel.getName(), types, attrNames, id);
238  interfaces.push_back(interface);
239  bool input = false;
240  bool output = false;
241  visitDepthFirst(prog, [&](const ram::IO& io) {
242  if (io.getRelation() == rel) {
243  const std::string& op = io.get("operation");
244  if (op == "input") {
245  input = true;
246  } else if (op == "output" || op == "printsize") {
247  output = true;
248  } else {
249  assert("wrong i/o operation");
250  }
251  }
252  });
253  addRelation(rel.getName(), interface, input, output);
254  id++;
255  }
256  }
257  ~InterpreterProgInterface() override {
258  for (auto* interface : interfaces) {
259  delete interface;
260  }
261  }
262 
263  /** Run program instance: not implemented */
264  void run() override {}
265 
266  /** Load data, run program instance, store data: not implemented */
267  void runAll(std::string, std::string) override {}
268 
269  /** Load input data: not implemented */
270  void loadAll(std::string) override {}
271 
272  /** Print output data: not implemented */
273  void printAll(std::string) override {}
274 
275  /** Dump inputs: not implemented */
276  void dumpInputs() override {}
277 
278  /** Dump outputs: not implemented */
279  void dumpOutputs() override {}
280 
281  /** Run subroutine */
282  void executeSubroutine(
283  std::string name, const std::vector<RamDomain>& args, std::vector<RamDomain>& ret) override {
284  exec.executeSubroutine(name, args, ret);
285  }
286 
287  /** Get symbol table */
288  SymbolTable& getSymbolTable() override {
289  return symTable;
290  }
291 
292 private:

295  SymbolTable& symTable;
296  std::vector<InterpreterRelInterface*> interfaces;
297 };
