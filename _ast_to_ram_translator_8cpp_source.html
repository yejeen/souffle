354  /** for provenance negation */
355  Own<ram::Condition> visitProvenanceNegation(const ast::ProvenanceNegation& neg) override {
356  const auto* atom = neg.getAtom();
357  size_t auxiliaryArity = translator.getEvaluationArity(atom);
358  assert(auxiliaryArity <= atom->getArity() && "auxiliary arity out of bounds");
359  size_t arity = atom->getArity() - auxiliaryArity;
360  VecOwn<ram::Expression> values;
361 
362  auto args = atom->getArguments();
363  for (size_t i = 0; i < arity; i++) {
364  values.push_back(translator.translateValue(args[i], index));
365  }
366  // we don't care about the provenance columns when doing the existence check
367  if (Global::config().has("provenance")) {
368  // undefined value for rule number
369  values.push_back(mk<ram::UndefValue>());
370  // add the height annotation for provenanceNotExists
371  for (size_t height = 1; height < auxiliaryArity; height++) {
372  values.push_back(translator.translateValue(args[arity + height], index));
373  }
374  }
375  return mk<ram::Negation>(
376  mk<ram::ProvenanceExistenceCheck>(translator.translateRelation(atom), std::move(values)));
377  }
378 
379  /** for negations */
380  Own<ram::Condition> visitNegation(const ast::Negation& neg) override {
381  const auto* atom = neg.getAtom();
382  size_t auxiliaryArity = translator.getEvaluationArity(atom);
383  assert(auxiliaryArity <= atom->getArity() && "auxiliary arity out of bounds");
384  size_t arity = atom->getArity() - auxiliaryArity;
385 
386  if (arity == 0) {
387  // for a nullary, negation is a simple emptiness check
388  return mk<ram::EmptinessCheck>(translator.translateRelation(atom));
389  }
390 
391  // else, we construct the atom and create a negation
392  VecOwn<ram::Expression> values;
393  auto args = atom->getArguments();
394  for (size_t i = 0; i < arity; i++) {
395  values.push_back(translator.translateValue(args[i], index));
396  }
397  for (size_t i = 0; i < auxiliaryArity; i++) {
398  values.push_back(mk<ram::UndefValue>());
399  }
400  return mk<ram::Negation>(
401  mk<ram::ExistenceCheck>(translator.translateRelation(atom), std::move(values)));
402  }
403  };
404  return ConstraintTranslator(*this, index)(*lit);
405 }

407 Own<ast::Clause> AstToRamTranslator::ClauseTranslator::getReorderedClause(
408  const ast::Clause& clause, const int version) const {
409  const auto plan = clause.getExecutionPlan();
410 
411  // check whether there is an imposed order constraint
412  if (plan == nullptr) {
413  // no plan, so reorder it according to the internal heuristic
414  auto sips = std::make_unique<ast::AllBoundSips>();
415  if (auto* reorderedClause =
416  ast::transform::ReorderLiteralsTransformer::reorderClauseWithSips(*sips, &clause)) {
417  return Own<ast::Clause>(reorderedClause);
418  }
419  return nullptr;
420  }
421  auto orders = plan->getOrders();
422  if (orders.find(version) == orders.end()) {
423  return nullptr;
424  }
425 
426  // get the imposed order
427  const auto& order = orders[version];
428 
429  // create a copy and fix order
430  Own<ast::Clause> reorderedClause(clause.clone());
431 
432  // Change order to start at zero
433  std::vector<unsigned int> newOrder(order->getOrder().size());
434  std::transform(order->getOrder().begin(), order->getOrder().end(), newOrder.begin(),
435  [](unsigned int i) -> unsigned int { return i - 1; });
436 
437  // re-order atoms
438  reorderedClause.reset(reorderAtoms(reorderedClause.get(), newOrder));
439 
440  // clear other order and fix plan
441  reorderedClause->clearExecutionPlan();
442 
443  return reorderedClause;
444 }

465  // get argument

493 void AstToRamTranslator::ClauseTranslator::createValueIndex(const ast::Clause& clause) {
494  for (const auto* atom : ast::getBodyLiterals<ast::Atom>(clause)) {
495  // std::map<const arg_list*, int> arg_level;
496  std::map<const ast::Node*, Own<arg_list>> nodeArgs;
497 
498  std::map<const arg_list*, int> arg_level;
499  nodeArgs[atom] = mk<arg_list>(atom->getArguments());
500  // the atom is obtained at the current level
501  // increment nesting level for the atom
502  arg_level[nodeArgs[atom].get()] = level++;
503  op_nesting.push_back(atom);
504 
505  indexValues(atom, nodeArgs, arg_level, translator.translateRelation(atom).get());
506  }
507 
508  // add aggregation functions
509  visitDepthFirstPostOrder(clause, [&](const ast::Argument& arg) {
510  // returns the write-location for this generator (or none if an equiv arg was already seen)
511  auto addGenerator = [&]() -> std::optional<int> {
512  // The by-value compare means that we're effectively doing CSE for any
513  // generator args during code-gen. This is a weird place to do this.
514  if (any_of(generators, [&](auto* x) { return *x == arg; })) return {};
515  generators.push_back(&arg);
516 
517  int aggLoc = level++;
518  valueIndex.setGeneratorLoc(arg, Location({aggLoc, 0}));
519  return aggLoc;
520  };
521 
522  if (auto agg = dynamic_cast<const ast::Aggregator*>(&arg)) {
523  if (auto aggLoc = addGenerator()) {
524  // bind aggregator variables to locations
525  const ast::Atom* atom = nullptr;
526  for (auto lit : agg->getBodyLiterals()) {
527  if (atom == nullptr) {
528  atom = dynamic_cast<const ast::Atom*>(lit);
529  } else {
530  break;
531  }
532  }
533  if (atom != nullptr) {
534  size_t pos = 0;
535  for (auto* arg : atom->getArguments()) {
536  if (const auto* var = dynamic_cast<const ast::Variable*>(arg)) {
537  valueIndex.addVarReference(
538  *var, *aggLoc, (int)pos, translator.translateRelation(atom));
539  }
540  ++pos;
541  }
542  }
543  }
544  }
545 
546  auto func = dynamic_cast<const ast::IntrinsicFunctor*>(&arg);
547  if (func && func->getFunctionInfo()->multipleResults) {
548  addGenerator();
549  }
550  });
551 }

572 Own<ram::Operation> AstToRamTranslator::ProvenanceClauseTranslator::createOperation(
573  const ast::Clause& clause) {
574  VecOwn<ram::Expression> values;
575 
576  // get all values in the body
577  for (ast::Literal* lit : clause.getBodyLiterals()) {
578  if (auto atom = dynamic_cast<ast::Atom*>(lit)) {
579  for (ast::Argument* arg : atom->getArguments()) {
580  values.push_back(translator.translateValue(arg, valueIndex));
581  }
582  } else if (auto neg = dynamic_cast<ast::ProvenanceNegation*>(lit)) {
583  size_t auxiliaryArity = translator.getEvaluationArity(neg->getAtom());
584  for (size_t i = 0; i < neg->getAtom()->getArguments().size() - auxiliaryArity; ++i) {
585  auto arg = neg->getAtom()->getArguments()[i];
586  values.push_back(translator.translateValue(arg, valueIndex));
587  }
588  for (size_t i = 0; i < auxiliaryArity; ++i) {
589  values.push_back(mk<ram::SignedConstant>(-1));
590  }
591  } else if (auto neg = dynamic_cast<ast::Negation*>(lit)) {
592  for (ast::Argument* arg : neg->getAtom()->getArguments()) {
593  values.push_back(translator.translateValue(arg, valueIndex));
594  }

638  if (constrainByFunctors) {
639  op = mkFilter(func->getReturnType() == TypeAttribute::Float,

644  ++pos;

650 /** generate RAM code for a clause */
651 Own<ram::Statement> AstToRamTranslator::ClauseTranslator::translateClause(
652  const ast::Clause& clause, const ast::Clause& originalClause, const int version) {
653  if (auto reorderedClause = getReorderedClause(clause, version)) {
654  // translate reordered clause
655  return translateClause(*reorderedClause, originalClause, version);
656  }
657 
658  // get extract some details
659  const ast::Atom* head = clause.getHead();
660 
661  // handle facts
662  if (isFact(clause)) {
663  // translate arguments
664  VecOwn<ram::Expression> values;
665  for (auto& arg : head->getArguments()) {
666  values.push_back(translator.translateValue(arg, ValueIndex()));
667  }
668 
669  // create a fact statement
670  return mk<ram::Query>(mk<ram::Project>(translator.translateRelation(head), std::move(values)));
671  }
672 
673  // the rest should be rules
674  assert(isRule(clause));
675 
676  createValueIndex(clause);
677 
678  // -- create RAM statement --
679 
680  Own<ram::Operation> op = createOperation(clause);
681 
682  /* add equivalence constraints imposed by variable binding */
683  for (const auto& cur : valueIndex.getVariableReferences()) {
684  // the first appearance
685  const Location& first = *cur.second.begin();
686  // all other appearances
687  for (const Location& loc : cur.second) {
688  if (first != loc && !valueIndex.isGenerator(loc.identifier)) {
689  // FIXME: equiv' for float types (`FEQ`)
690  op = mk<ram::Filter>(mk<ram::Constraint>(BinaryConstraintOp::EQ, makeRamTupleElement(first),
691  makeRamTupleElement(loc)),
692  std::move(op));
693  }
694  }
695  }
696 
697  /* add conditions caused by atoms, negations, and binary relations */
698  for (const auto& lit : clause.getBodyLiterals()) {
699  if (auto condition = translator.translateConstraint(lit, valueIndex)) {
700  op = mk<ram::Filter>(std::move(condition), std::move(op));
701  }
702  }
703 
704  // add aggregator conditions
705  size_t curLevel = op_nesting.size() - 1;
706  for (auto it = op_nesting.rbegin(); it != op_nesting.rend(); ++it, --curLevel) {
707  const ast::Node* cur = *it;
708 
709  if (const auto* atom = dynamic_cast<const ast::Atom*>(cur)) {
710  // add constraints
711  size_t pos = 0;
712  for (auto arg : atom->getArguments()) {
713  if (auto* agg = dynamic_cast<ast::Aggregator*>(arg)) {
714  auto loc = valueIndex.getGeneratorLoc(*agg);
715  // FIXME: equiv' for float types (`FEQ`)
716  op = mk<ram::Filter>(
717  mk<ram::Constraint>(BinaryConstraintOp::EQ, mk<ram::TupleElement>(curLevel, pos),
718  makeRamTupleElement(loc)),
719  std::move(op));
720  }
721  ++pos;
722  }
723  }
724  }
725 
726  // add generator levels
727  --level;
728  for (auto* cur : reverse(generators)) {
729  if (auto agg = dynamic_cast<const ast::Aggregator*>(cur)) {
730  // condition for aggregate and helper function to add terms
731  Own<ram::Condition> aggCond;
732  auto addAggCondition = [&](Own<ram::Condition> arg) {
733  aggCond = aggCond ? mk<ram::Conjunction>(std::move(aggCond), std::move(arg)) : std::move(arg);
734  };
735 
736  // translate constraints of sub-clause
737  for (auto&& lit : agg->getBodyLiterals()) {
738  if (auto newCondition = translator.translateConstraint(lit, valueIndex)) {
739  addAggCondition(std::move(newCondition));
740  }
741  }
742 
743  // get the first predicate of the sub-clause
744  // NB: at most one atom is permitted in a sub-clause
745  const ast::Atom* atom = nullptr;
746  for (auto&& lit : agg->getBodyLiterals()) {
747  if (atom == nullptr) {
748  atom = dynamic_cast<const ast::Atom*>(lit);
749  } else {
750  assert(!isA<ast::Atom>(lit) && "Unsupported complex aggregation body encountered!");
751  }
752  }
753 
754  // translate arguments's of atom (if exists) to conditions
755  if (atom != nullptr) {
756  size_t pos = 0;
757  auto addAggEqCondition = [&](Own<ram::Expression> value) {
758  if (isUndefValue(value.get())) return;
759 
760  // FIXME: equiv' for float types (`FEQ`)
761  addAggCondition(mk<ram::Constraint>(
762  BinaryConstraintOp::EQ, mk<ram::TupleElement>(level, pos), std::move(value)));
763  };
764  for (auto* arg : atom->getArguments()) {
765  // variable bindings are issued differently since we don't want self
766  // referential variable bindings
767  if (auto* var = dynamic_cast<const ast::Variable*>(arg)) {
768  for (auto&& loc : valueIndex.getVariableReferences().find(var->getName())->second) {
769  if (level != loc.identifier || (int)pos != loc.element) {
770  addAggEqCondition(makeRamTupleElement(loc));
771  break;
772  }
773  }
774  } else if (auto value = translator.translateValue(arg, valueIndex)) {
775  addAggEqCondition(std::move(value));
776  }
777  ++pos;
778  }
779  }
780 
781  // translate aggregate expression
782  auto expr = translator.translateValue(agg->getTargetExpression(), valueIndex);
783 
784  // add Ram-Aggregation layer
785  op = mk<ram::Aggregate>(std::move(op), agg->getOperator(), translator.translateRelation(atom),
786  expr ? std::move(expr) : mk<ram::UndefValue>(),
787  aggCond ? std::move(aggCond) : mk<ram::True>(), level);
788  } else if (const auto* func = dynamic_cast<const ast::IntrinsicFunctor*>(cur)) {
789  VecOwn<ram::Expression> args;
790  for (auto&& x : func->getArguments()) {
791  args.push_back(translator.translateValue(x, valueIndex));
792  }
793 
794  auto func_op = [&]() -> ram::NestedIntrinsicOp {
795  switch (func->getFunctionInfo()->op) {
796  case FunctorOp::RANGE: return ram::NestedIntrinsicOp::RANGE;
797  case FunctorOp::URANGE: return ram::NestedIntrinsicOp::URANGE;
798  case FunctorOp::FRANGE: return ram::NestedIntrinsicOp::FRANGE;
799 
800  default:
801  assert(func->getFunctionInfo()->multipleResults);
802  fatal("missing case handler or bad code-gen");
803  }
804  };
805 
806  op = mk<ram::NestedIntrinsicOperator>(func_op(), std::move(args), std::move(op), level);
807  }
808 
809  --level;
810  }
811 
812  // build operation bottom-up
813  while (!op_nesting.empty()) {
814  // get next operator
815  const ast::Node* cur = op_nesting.back();
816  op_nesting.pop_back();
817 
818  // get current nesting level
819  auto level = op_nesting.size();
820 
821  if (const auto* atom = dynamic_cast<const ast::Atom*>(cur)) {
822  // add constraints
823  // TODO: do we wish to enable constraints by header functor? record inits do so...
824  op = filterByConstraints(level, atom->getArguments(), std::move(op), false);
825 
826  // check whether all arguments are unnamed variables
827  bool isAllArgsUnnamed = true;
828  for (auto* argument : atom->getArguments()) {
829  if (!isA<ast::UnnamedVariable>(argument)) {
830  isAllArgsUnnamed = false;
831  }
832  }
833 
834  // add check for emptiness for an atom
835  op = mk<ram::Filter>(
836  mk<ram::Negation>(mk<ram::EmptinessCheck>(translator.translateRelation(atom))),
837  std::move(op));
838 
839  // add a scan level
840  if (atom->getArity() != 0 && !isAllArgsUnnamed) {
841  if (head->getArity() == 0) {
842  op = mk<ram::Break>(
843  mk<ram::Negation>(mk<ram::EmptinessCheck>(translator.translateRelation(head))),
844  std::move(op));
845  }
846  if (Global::config().has("profile")) {
847  std::stringstream ss;
848  ss << head->getQualifiedName();
849  ss.str("");
850  ss << "@frequency-atom" << ';';
851  ss << originalClause.getHead()->getQualifiedName() << ';';
852  ss << version << ';';
853  ss << stringify(toString(clause)) << ';';
854  ss << stringify(toString(*atom)) << ';';
855  ss << stringify(toString(originalClause)) << ';';
856  ss << level << ';';
857  op = mk<ram::Scan>(translator.translateRelation(atom), level, std::move(op), ss.str());
858  } else {
859  op = mk<ram::Scan>(translator.translateRelation(atom), level, std::move(op));
860  }
861  }
862 
863  // TODO: support constants in nested records!
864  } else if (const auto* rec = dynamic_cast<const ast::RecordInit*>(cur)) {
865  // add constant constraints
866  op = filterByConstraints(level, rec->getArguments(), std::move(op));
867 
868  // add an unpack level
869  const Location& loc = valueIndex.getDefinitionPoint(*rec);
870  op = mk<ram::UnpackRecord>(
871  std::move(op), level, makeRamTupleElement(loc), rec->getArguments().size());
872  } else {
873  fatal("Unsupported AST node for creation of scan-level!");
874  }
875  }
876 
877  /* generate the final RAM Insert statement */
878  Own<ram::Condition> cond = createCondition(originalClause);
879  if (cond != nullptr) {
880  return mk<ram::Query>(mk<ram::Filter>(std::move(cond), std::move(op)));
881  } else {
882  return mk<ram::Query>(std::move(op));
883  }
884 }

998 /** generate RAM code for recursive relations in a strongly-connected component */
999 Own<ram::Statement> AstToRamTranslator::translateRecursiveRelation(const std::set<const ast::Relation*>& scc,
1000  const ast::analysis::RecursiveClausesAnalysis* recursiveClauses) {
1001  // initialize sections
1002  VecOwn<ram::Statement> preamble;
1003  VecOwn<ram::Statement> updateTable;
1004  VecOwn<ram::Statement> postamble;
1005 
1006  auto genMerge = [](const ram::RelationReference* dest,
1007  const ram::RelationReference* src) -> Own<ram::Statement> {
1008  VecOwn<ram::Expression> values;
1009  if (src->get()->getArity() == 0) {
1010  return mk<ram::Query>(
1011  mk<ram::Filter>(mk<ram::Negation>(mk<ram::EmptinessCheck>(souffle::clone(src))),
1012  mk<ram::Project>(souffle::clone(dest), std::move(values))));
1013  }
1014  for (std::size_t i = 0; i < dest->get()->getArity(); i++) {
1015  values.push_back(mk<ram::TupleElement>(0, i));
1016  }
1017  auto stmt = mk<ram::Query>(mk<ram::Scan>(
1018  souffle::clone(src), 0, mk<ram::Project>(souffle::clone(dest), std::move(values))));
1019  if (dest->get()->getRepresentation() == RelationRepresentation::EQREL) {
1020  return mk<ram::Sequence>(
1021  mk<ram::Extend>(souffle::clone(dest), souffle::clone(src)), std::move(stmt));
1022  }
1023  return stmt;
1024  };
1025 
1026  // --- create preamble ---
1027 
1028  /* Compute non-recursive clauses for relations in scc and push
1029  the results in their delta tables. */
1030  for (const ast::Relation* rel : scc) {
1031  /* create update statements for fixpoint (even iteration) */
1032  Own<ram::Statement> updateRelTable =
1033  mk<ram::Sequence>(genMerge(translateRelation(rel).get(), translateNewRelation(rel).get()),
1034  mk<ram::Swap>(translateDeltaRelation(rel), translateNewRelation(rel)),
1035  mk<ram::Clear>(translateNewRelation(rel)));
1036 
1037  /* measure update time for each relation */
1038  if (Global::config().has("profile")) {
1039  updateRelTable = mk<ram::LogRelationTimer>(std::move(updateRelTable),
1040  LogStatement::cRecursiveRelation(toString(rel->getQualifiedName()), rel->getSrcLoc()),
1041  translateNewRelation(rel));
1042  }
1043 
1044  /* drop temporary tables after recursion */
1045  appendStmt(postamble, mk<ram::Clear>(translateDeltaRelation(rel)));
1046  appendStmt(postamble, mk<ram::Clear>(translateNewRelation(rel)));
1047 
1048  /* Generate code for non-recursive part of relation */
1049  /* Generate merge operation for temp tables */
1050  appendStmt(preamble, translateNonRecursiveRelation(*rel, recursiveClauses));
1051  appendStmt(preamble, genMerge(translateDeltaRelation(rel).get(), translateRelation(rel).get()));
1052 
1053  /* Add update operations of relations to parallel statements */
1054  appendStmt(updateTable, std::move(updateRelTable));
1055  }
1056 
1057  // --- build main loop ---
1058 
1059  VecOwn<ram::Statement> loopSeq;
1060 
1061  // create a utility to check SCC membership
1062  auto isInSameSCC = [&](const ast::Relation* rel) {
1063  return std::find(scc.begin(), scc.end(), rel) != scc.end();
1064  };
1065 
1066  /* Compute temp for the current tables */
1067  for (const ast::Relation* rel : scc) {
1068  VecOwn<ram::Statement> loopRelSeq;
1069 
1070  /* Find clauses for relation rel */
1071  for (const auto& cl : getClauses(*program, *rel)) {
1072  // skip non-recursive clauses
1073  if (!recursiveClauses->recursive(cl)) {
1074  continue;
1075  }
1076 
1077  // each recursive rule results in several operations
1078  int version = 0;
1079  const auto& atoms = ast::getBodyLiterals<ast::Atom>(*cl);
1080  for (size_t j = 0; j < atoms.size(); ++j) {
1081  const ast::Atom* atom = atoms[j];
1082  const ast::Relation* atomRelation = getAtomRelation(atom, program);
1083 
1084  // only interested in atoms within the same SCC
1085  if (!isInSameSCC(atomRelation)) {
1086  continue;
1087  }
1088 
1089  // modify the processed rule to use delta relation and write to new relation
1090  Own<ast::Clause> r1(cl->clone());
1091  r1->getHead()->setQualifiedName(translateNewRelation(rel)->get()->getName());
1092  ast::getBodyLiterals<ast::Atom>(*r1)[j]->setQualifiedName(
1093  translateDeltaRelation(atomRelation)->get()->getName());
1094  if (Global::config().has("provenance")) {
1095  r1->addToBody(mk<ast::ProvenanceNegation>(souffle::clone(cl->getHead())));
1096  } else {
1097  if (r1->getHead()->getArity() > 0) {
1098  r1->addToBody(mk<ast::Negation>(souffle::clone(cl->getHead())));
1099  }
1100  }
1101 
1102  // replace wildcards with variables (reduces indices when wildcards are used in recursive
1103  // atoms)
1104  nameUnnamedVariables(r1.get());
1105 
1106  // reduce R to P ...
1107  for (size_t k = j + 1; k < atoms.size(); k++) {
1108  if (isInSameSCC(getAtomRelation(atoms[k], program))) {
1109  auto cur = souffle::clone(ast::getBodyLiterals<ast::Atom>(*r1)[k]);
1110  cur->setQualifiedName(
1111  translateDeltaRelation(getAtomRelation(atoms[k], program))->get()->getName());
1112  r1->addToBody(mk<ast::Negation>(std::move(cur)));
1113  }
1114  }
1115 
1116  Own<ram::Statement> rule = ClauseTranslator(*this).translateClause(*r1, *cl, version);
1117 
1118  /* add logging */
1119  if (Global::config().has("profile")) {
1120  const std::string& relationName = toString(rel->getQualifiedName());
1121  const SrcLocation& srcLocation = cl->getSrcLoc();
1122  const std::string clauseText = stringify(toString(*cl));
1123  const std::string logTimerStatement =
1124  LogStatement::tRecursiveRule(relationName, version, srcLocation, clauseText);
1125  const std::string logSizeStatement =
1126  LogStatement::nRecursiveRule(relationName, version, srcLocation, clauseText);
1127  rule = mk<ram::LogRelationTimer>(
1128  std::move(rule), logTimerStatement, translateNewRelation(rel));
1129  }
1130 
1131  // add debug info
1132  std::ostringstream ds;
1133  ds << toString(*cl) << "\nin file ";
1134  ds << cl->getSrcLoc();
1135  rule = mk<ram::DebugInfo>(std::move(rule), ds.str());
1136 
1137  // add to loop body
1138  appendStmt(loopRelSeq, std::move(rule));
1139 
1140  // increment version counter
1141  version++;
1142  }
1143 
1144  if (cl->getExecutionPlan() != nullptr) {
1145  // ensure that all required versions have been created, as expected
1146  int maxVersion = -1;
1147  for (auto const& cur : cl->getExecutionPlan()->getOrders()) {
1148  maxVersion = std::max(cur.first, maxVersion);
1149  }
1150  assert(version > maxVersion && "missing clause versions");
1151  }
1152  }
1153 
1154  // if there was no rule, continue
1155  if (loopRelSeq.size() == 0) {
1156  continue;
1157  }
1158 
1159  // label all versions
1160  if (Global::config().has("profile")) {
1161  const std::string& relationName = toString(rel->getQualifiedName());
1162  const SrcLocation& srcLocation = rel->getSrcLoc();
1163  const std::string logTimerStatement = LogStatement::tRecursiveRelation(relationName, srcLocation);
1164  const std::string logSizeStatement = LogStatement::nRecursiveRelation(relationName, srcLocation);
1165  auto newStmt = mk<ram::LogRelationTimer>(
1166  mk<ram::Sequence>(std::move(loopRelSeq)), logTimerStatement, translateNewRelation(rel));
1167  loopRelSeq.clear();
1168  appendStmt(loopRelSeq, std::move(newStmt));
1169  }
1170 
1171  /* add rule computations of a relation to parallel statement */
1172  appendStmt(loopSeq, mk<ram::Sequence>(std::move(loopRelSeq)));
1173  }
1174  auto loop = mk<ram::Parallel>(std::move(loopSeq));
1175 
1176  /* construct exit conditions for odd and even iteration */
1177  auto addCondition = [](Own<ram::Condition>& cond, Own<ram::Condition> clause) {
1178  cond = ((cond) ? mk<ram::Conjunction>(std::move(cond), std::move(clause)) : std::move(clause));
1179  };
1180 
1181  Own<ram::Condition> exitCond;
1182  VecOwn<ram::Statement> exitStmts;
1183  for (const ast::Relation* rel : scc) {
