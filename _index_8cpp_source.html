190 bool MaxMatching::bfSearch() {
191  Node u;
192  std::queue<Node> bfQueue;
193  // Build layers
194  for (auto& it : graph) {
195  if (getMatch(it.first) == NullVertex) {
196  distance[it.first] = 0;
197  bfQueue.push(it.first);
198  } else {
199  distance[it.first] = InfiniteDistance;
200  }
201  }
202 
203  distance[NullVertex] = InfiniteDistance;
204  while (!bfQueue.empty()) {
205  u = bfQueue.front();
206  bfQueue.pop();
207  assert(u != NullVertex);
208  const Edges& children = graph[u];
209  for (auto it : children) {
210  Node mv = getMatch(it);
211  if (getDistance(mv) == InfiniteDistance) {
212  distance[mv] = getDistance(u) + 1;
213  if (mv != NullVertex) {
214  bfQueue.push(mv);
215  }
216  }
217  }
218  }
219  return (getDistance(NullVertex) != InfiniteDistance);
220 }

256 void MinIndexSelection::solve() {
257  // map the keys in the key set to lexicographical order
258  if (searches.empty()) {
259  return;
260  }
261 
262  // discharge multiple inequalities
263  removeExtraInequalities();
264 
265  // map the signatures of each search to a unique index for the matching problem
266  AttributeIndex currentIndex = 1;
267  for (SearchSignature s : searches) {
268  if (s.empty()) {
269  continue;
270  }
271  // map the signature to its unique index in each set
272  signatureToIndexA.insert({s, currentIndex});
273  signatureToIndexB.insert({s, currentIndex + 1});
274  // map each index back to the search signature
275  indexToSignature.insert({currentIndex, s});
276  indexToSignature.insert({currentIndex + 1, s});
277  currentIndex += 2;
278  }
279 
280  // Construct the matching poblem
281  for (auto search : searches) {
282  for (auto itt : searches) {
283  if (search == itt) {
284  continue;
285  }
286 
287  if (search < itt) {
288  matching.addEdge(signatureToIndexA[search], signatureToIndexB[itt]);
289  }
290  }
291  }
292 
293  // Perform the Hopcroft-Karp on the graph and receive matchings (mapped A->B and B->A)
294  // Assume: alg.calculate is not called on an empty graph
295  assert(!searches.empty());
296  const MaxMatching::Matchings& matchings = matching.solve();
297 
298  // Extract the chains given the nodes and matchings
299  ChainOrderMap chains = getChainsFromMatching(matchings, searches);
300 
301  // Should never get no chains back as we never call calculate on an empty graph
302  assert(!chains.empty());
303  for (const auto& chain : chains) {
304  std::vector<uint32_t> ids;
305 
306  SearchSignature initDelta = *(chain.begin());
307  insertIndex(ids, initDelta);
308 
309  // build the lex-order
310  for (auto iit = chain.begin(); next(iit) != chain.end(); ++iit) {
311  SearchSignature delta = SearchSignature::getDelta(*next(iit), *iit);
312  insertIndex(ids, delta);
313  }
314 
315  assert(!ids.empty());
316  orders.push_back(ids);
317  }
318 
319  // Validate the lex-order
320  for (auto chain : chains) {
321  for (auto search : chain) {
322  int idx = map(search);
323  size_t l = card(search);
324 
325  SearchSignature k(search.arity());
326  for (size_t i = 0; i < l; i++) {
327  k[orders[idx][i]] = AttributeConstraint::Equal;
328  }
329  for (size_t i = 0; i < search.arity(); ++i) {
330  if (k[i] == AttributeConstraint::None && search[i] != AttributeConstraint::None) {
331  assert("incorrect lexicographical order");
332  }
333  if (k[i] != AttributeConstraint::None && search[i] == AttributeConstraint::None) {
334  assert("incorrect lexicographical order");
335  }
336  }
337  }
338  }
339 }

341 MinIndexSelection::Chain MinIndexSelection::getChain(
342  const SearchSignature umn, const MaxMatching::Matchings& match) {
343  SearchSignature start = umn; // start at an unmatched node
344  Chain chain;
345  // given an unmapped node from set A we follow it from set B until it cannot be matched from B
346  // if not mateched from B then umn is a chain
347  //
348  // Assume : no circular mappings, i.e. a in A -> b in B -> ........ -> a in A is not allowed.
349  // Given this, the loop will terminate
350  while (true) {
351  auto mit = match.find(signatureToIndexB[start]); // we start from B side
352  // on each iteration we swap sides when collecting the chain so we use the corresponding index map
353  if (std::find(chain.begin(), chain.end(), start) == chain.end()) {
354  chain.push_back(start);
355  }
356 
357  if (mit == match.end()) {
358  std::reverse(chain.begin(), chain.end());
359  return chain;
360  }
361 
362  SearchSignature a = indexToSignature.at(mit->second);
363  if (std::find(chain.begin(), chain.end(), a) == chain.end()) {
364  chain.push_back(a);
365  }
366  start = a;
367  }
368 }

371  const MaxMatching::Matchings& match, const SearchSet& nodes) {
372  assert(!nodes.empty());
373 
374  // Get all unmatched nodes from A
375  const SearchSet& umKeys = getUnmatchedKeys(match, nodes);
376  // Case: if no unmatched nodes then we have an anti-chain
377  if (umKeys.empty()) {
378  for (auto node : nodes) {
379  Chain a;
380  a.push_back(node);
381  chainToOrder.push_back(a);
382  return chainToOrder;
383  }
384  }
385 
386  assert(!umKeys.empty());
387 
388  // A worklist of used nodes
389  SearchSet usedKeys;
390 
391  // Case: nodes < umKeys or if nodes == umKeys then anti chain - this is handled by this loop
392  for (auto umKey : umKeys) {
393  Chain c = getChain(umKey, match);
394  assert(!c.empty());
395  chainToOrder.push_back(c);
396  }
397 
398  assert(!chainToOrder.empty());

437  const SearchSignature& s, const Relation& rel) {
438  // by default we have all attributes w/inequalities discharged
439  AttributeSet allInequalities;
440  for (size_t i = 0; i < s.arity(); ++i) {
441  if (s[i] == AttributeConstraint::Inequal) {
442  allInequalities.insert(i);
443  }
444  }
445 
446  // if we don't have a btree then we don't retain any inequalities
447  if (rel.getRepresentation() != RelationRepresentation::BTREE &&
448  rel.getRepresentation() != RelationRepresentation::DEFAULT) {
449  return allInequalities;
450  }
451 
452  // do not support indexed inequalities with provenance
453  if (Global::config().has("provenance")) {
454  return allInequalities;
455  }
456 
457  // if we are in the interpreter then we only permit signed inequalities
458  AttributeSet inequalitiesNotSigned;
459  for (size_t i = 0; i < s.arity(); ++i) {
460  if (s[i] == AttributeConstraint::Inequal && rel.getAttributeTypes()[i][0] != 'i') {
461  inequalitiesNotSigned.insert(i);
462  }
463  }
464  if (!Global::config().has("compile") && !Global::config().has("dl-program") &&
465  !Global::config().has("generate") && !Global::config().has("swig")) {
466  return inequalitiesNotSigned;
467  }
468 
469  return dischargedMap[s];

472 void IndexAnalysis::run(const TranslationUnit& translationUnit) {
473  // After complete:
474  // 1. All relations should have at least one index (for full-order search).
475  // 2. Two relations involved in a swap operation will have same set of indices.
476  // 3. A 0-arity relation will have only one index where LexOrder is defined as empty. A comparator using
477  // an empty order should regard all elements as equal and therefore only allow one arbitrary tuple to be
478  // inserted.
479  //
480  // TODO:
481  // 0-arity relation in a provenance program still need to be revisited.
482 
483  // visit all nodes to collect searches of each relation
484  visitDepthFirst(translationUnit.getProgram(), [&](const Node& node) {
485  if (const auto* indexSearch = dynamic_cast<const IndexOperation*>(&node)) {
486  MinIndexSelection& indexes = getIndexes(indexSearch->getRelation());
487  indexes.addSearch(getSearchSignature(indexSearch));
488  } else if (const auto* exists = dynamic_cast<const ExistenceCheck*>(&node)) {
489  MinIndexSelection& indexes = getIndexes(exists->getRelation());
490  indexes.addSearch(getSearchSignature(exists));
491  } else if (const auto* provExists = dynamic_cast<const ProvenanceExistenceCheck*>(&node)) {
492  MinIndexSelection& indexes = getIndexes(provExists->getRelation());
493  indexes.addSearch(getSearchSignature(provExists));
494  } else if (const auto* ramRel = dynamic_cast<const Relation*>(&node)) {
495  MinIndexSelection& indexes = getIndexes(*ramRel);
496  indexes.addSearch(getSearchSignature(ramRel));
497  }
498  });
499 
500  // A swap happen between rel A and rel B indicates A should include all indices of B, vice versa.
501  visitDepthFirst(translationUnit.getProgram(), [&](const Swap& swap) {
502  // Note: this naive approach will not work if there exists chain or cyclic swapping.
503  // e.g. swap(relA, relB) swap(relB, relC) swap(relC, relA)
504  // One need to keep merging the search set until a fixed point where no more index is introduced
505  // in any of the relation in a complete iteration.
506  //
507  // Currently RAM does not have such situation.
508  const Relation& relA = swap.getFirstRelation();
509  const Relation& relB = swap.getSecondRelation();
510 
511  MinIndexSelection& indexesA = getIndexes(relA);
512  MinIndexSelection& indexesB = getIndexes(relB);
513  // Add all searchSignature of A into B
514  for (const auto& signature : indexesA.getSearches()) {
515  indexesB.addSearch(signature);
516  }
517 
518  // Add all searchSignature of B into A
519  for (const auto& signature : indexesB.getSearches()) {
520  indexesA.addSearch(signature);
521  }
522  });
523 
524  // find optimal indexes for relations
525  for (auto& cur : minIndexCover) {
526  MinIndexSelection& indexes = cur.second;
527  indexes.solve();
528  }
529 
530  // Only case where indexSet is still empty is when relation has arity == 0
531  for (auto& cur : minIndexCover) {
532  MinIndexSelection& indexes = cur.second;
533  if (indexes.getAllOrders().empty()) {
534  indexes.insertDefaultTotalIndex(0);
535  }
536  }
537 }

603 SearchSignature IndexAnalysis::getSearchSignature(const IndexOperation* search) const {
604  size_t arity = search->getRelation().getArity();
605 
606  auto lower = search->getRangePattern().first;
607  auto upper = search->getRangePattern().second;
608  SearchSignature keys(arity);
609  for (size_t i = 0; i < arity; ++i) {
610  // if both bounds are undefined
611  if (isUndefValue(lower[i]) && isUndefValue(upper[i])) {
612  keys[i] = AttributeConstraint::None;
613  // if bounds are equal we have an equality
614  } else if (*lower[i] == *upper[i]) {
615  keys[i] = AttributeConstraint::Equal;
616  } else {
617  keys[i] = AttributeConstraint::Inequal;
618  }
619  }
620  return keys;
621 }
