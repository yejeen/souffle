123  /**
124  * @brief Generate the tree based on given entry.
125  * Return a NodePtr to the root.
126  */
127  NodePtr generateTree(const ram::Node& root, const ram::Program& program) {
128  this->program = const_cast<ram::Program*>(&program);
129  // Encode all relation, indexPos and viewId.
130  visitDepthFirst(root, [&](const ram::Node& node) {
131  if (isA<ram::Query>(&node)) {
132  newQueryBlock();
133  }
134  if (const auto* indexSearch = dynamic_cast<const ram::IndexOperation*>(&node)) {
135  encodeIndexPos(*indexSearch);
136  encodeView(indexSearch);
137  } else if (const auto* exists = dynamic_cast<const ram::ExistenceCheck*>(&node)) {
138  encodeIndexPos(*exists);
139  encodeView(exists);
140  } else if (const auto* provExists = dynamic_cast<const ram::ProvenanceExistenceCheck*>(&node)) {
141  encodeIndexPos(*provExists);
142  encodeView(provExists);
143  }
144  });
145  // Parse program
146  return visit(root);
147  }

469  NodePtr visitQuery(const ram::Query& query) override {
470  std::shared_ptr<InterpreterViewContext> viewContext = std::make_shared<InterpreterViewContext>();
471  parentQueryViewContext = viewContext;
472  // split terms of conditions of outer-most filter operation
473  // into terms that require a context and terms that
474  // do not require a view
475  const ram::Operation* next = &query.getOperation();
476  std::vector<const ram::Condition*> freeOfView;
477  if (const auto* filter = dynamic_cast<const ram::Filter*>(&query.getOperation())) {
478  next = &filter->getOperation();
479  // Check terms of outer filter operation whether they can be pushed before
480  // the view-generation for speed improvements
481  auto conditions = toConjunctionList(&filter->getCondition());
482  for (auto const& cur : conditions) {
483  bool needView = false;
484  visitDepthFirst(*cur, [&](const ram::Node& node) {
485  if (requireView(&node)) {
486  needView = true;
487  const auto& rel = getRelationRefForView(&node);
488  viewContext->addViewInfoForFilter(
489  encodeRelation(rel), indexTable[&node], encodeView(&node));
490  }
491  });
492 
493  if (needView) {
494  viewContext->addViewOperationForFilter(visit(*cur));
495  } else {
496  viewContext->addViewFreeOperationForFilter(visit(*cur));
497  }
498  }
499  }
500 
501  visitDepthFirst(*next, [&](const ram::Node& node) {
502  if (requireView(&node)) {
503  const auto& rel = getRelationRefForView(&node);
504  viewContext->addViewInfoForNested(encodeRelation(rel), indexTable[&node], encodeView(&node));
505  };
506  });
507 
508  visitDepthFirst(*next, [&](const ram::AbstractParallel&) { viewContext->isParallel = true; });
509 
510  NodePtrVec children;
511  children.push_back(visit(*next));
512 
513  auto res = mk<InterpreterQuery>(I_Query, &query, visit(*next));
514  res->setViewContext(parentQueryViewContext);
515  return res;
516  }

593  size_t encodeIndexPos(RamNode& node) {
594  const ram::analysis::MinIndexSelection& orderSet = isa->getIndexes(node.getRelation());
595  ram::analysis::SearchSignature signature = isa->getSearchSignature(&node);
596  // A zero signature is equivalent as a full order signature.
597  if (signature.empty()) {
598  signature = ram::analysis::SearchSignature::getFullSearchSignature(signature.arity());
599  }

601  indexTable[&node] = i;
602  return i;

668  /**
669  * @brief Convert terms of a conjunction to a list
670  *
671  * Convert a condition of the format C1 /\ C2 /\ ... /\ Cn
672  * to a list {C1, C2, ..., Cn}.
673  */
674  inline std::vector<const ram::Condition*> toConjunctionList(const ram::Condition* condition) {
675  std::vector<const ram::Condition*> conditionList;
676  std::queue<const ram::Condition*> conditionsToProcess;
677  if (condition != nullptr) {
678  conditionsToProcess.push(condition);
679  while (!conditionsToProcess.empty()) {
680  condition = conditionsToProcess.front();
681  conditionsToProcess.pop();
682  if (const auto* ramConj = dynamic_cast<const ram::Conjunction*>(condition)) {
683  conditionsToProcess.push(&ramConj->getLHS());
684  conditionsToProcess.push(&ramConj->getRHS());
685  } else {
686  conditionList.emplace_back(condition);
687  }
688  }
689  }
690  return conditionList;
691  }

717  /**
718  * @brief Encode and return the super-instruction information about a index operation.
719  */
720  InterpreterSuperInstruction getIndexSuperInstInfo(const ram::IndexOperation& ramIndex) {
721  size_t arity = ramIndex.getRelation().getArity();
722  InterpreterSuperInstruction indexOperation(arity);
723  const auto& first = ramIndex.getRangePattern().first;
724  for (size_t i = 0; i < arity; ++i) {
725  auto& low = first[i];
726  // Unbounded
727  if (isUndefValue(low)) {
728  indexOperation.first[i] = MIN_RAM_SIGNED;
729  continue;
730  }
731 
732  // Constant
733  if (isA<ram::Constant>(low)) {
734  indexOperation.first[i] = dynamic_cast<ram::Constant*>(low)->getConstant();
735  continue;
736  }
737 
738  // TupleElement
739  if (isA<ram::TupleElement>(low)) {
740  auto lowTuple = dynamic_cast<ram::TupleElement*>(low);
741  indexOperation.tupleFirst.push_back(
742  {i, (size_t)lowTuple->getTupleId(), lowTuple->getElement()});
743  continue;
744  }
745 
746  // Generic expression
747  indexOperation.exprFirst.push_back(std::pair<size_t, Own<InterpreterNode>>(i, visit(low)));
748  }
749  const auto& second = ramIndex.getRangePattern().second;
750  for (size_t i = 0; i < arity; ++i) {
751  auto& hig = second[i];
752  // Unbounded
753  if (isUndefValue(hig)) {
754  indexOperation.second[i] = MAX_RAM_SIGNED;
755  continue;
756  }
757 
758  // Constant
759  if (isA<ram::Constant>(hig)) {
760  indexOperation.second[i] = dynamic_cast<ram::Constant*>(hig)->getConstant();
761  continue;
762  }
763 
764  // TupleElement
765  if (isA<ram::TupleElement>(hig)) {
766  auto highTuple = dynamic_cast<ram::TupleElement*>(hig);
767  indexOperation.tupleSecond.push_back(
768  {i, (size_t)highTuple->getTupleId(), highTuple->getElement()});
769  continue;
770  }
771 
772  // Generic expression
773  indexOperation.exprSecond.push_back(std::pair<size_t, Own<InterpreterNode>>(i, visit(hig)));
774  }
775  return indexOperation;
776  }
