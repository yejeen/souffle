44 ExpressionPair MakeIndexTransformer::getExpressionPair(
45  const Constraint* binRelOp, size_t& element, int identifier) {
46  if (isLessEqual(binRelOp->getOperator())) {
47  // Tuple[level, element] <= <expr>
48  if (const auto* lhs = dynamic_cast<const TupleElement*>(&binRelOp->getLHS())) {
49  const Expression* rhs = &binRelOp->getRHS();
50  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
51  element = lhs->getElement();
52  return {mk<UndefValue>(), clone(rhs)};
53  }
54  }
55  // <expr> <= Tuple[level, element]
56  if (const auto* rhs = dynamic_cast<const TupleElement*>(&binRelOp->getRHS())) {
57  const Expression* lhs = &binRelOp->getLHS();
58  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
59  element = rhs->getElement();
60  return {clone(lhs), mk<UndefValue>()};
61  }
62  }
63  }
64 
65  if (isGreaterEqual(binRelOp->getOperator())) {
66  // Tuple[level, element] >= <expr>
67  if (const auto* lhs = dynamic_cast<const TupleElement*>(&binRelOp->getLHS())) {
68  const Expression* rhs = &binRelOp->getRHS();
69  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
70  element = lhs->getElement();
71  return {clone(rhs), mk<UndefValue>()};
72  }
73  }
74  // <expr> >= Tuple[level, element]
75  if (const auto* rhs = dynamic_cast<const TupleElement*>(&binRelOp->getRHS())) {
76  const Expression* lhs = &binRelOp->getLHS();
77  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
78  element = rhs->getElement();
79  return {mk<UndefValue>(), clone(lhs)};
80  }
81  }
82  }
83  return {mk<UndefValue>(), mk<UndefValue>()};
84 }

88 ExpressionPair MakeIndexTransformer::getLowerUpperExpression(Condition* c, size_t& element, int identifier) {
89  if (auto* binRelOp = dynamic_cast<Constraint*>(c)) {
90  if (isEqConstraint(binRelOp->getOperator())) {
91  if (const auto* lhs = dynamic_cast<const TupleElement*>(&binRelOp->getLHS())) {
92  const Expression* rhs = &binRelOp->getRHS();
93  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
94  element = lhs->getElement();
95  return {clone(rhs), clone(rhs)};
96  }
97  }
98  if (const auto* rhs = dynamic_cast<const TupleElement*>(&binRelOp->getRHS())) {
99  const Expression* lhs = &binRelOp->getLHS();
100  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
101  element = rhs->getElement();
102  return {clone(lhs), clone(lhs)};
103  }
104  }
105  } else if (isWeakIneqConstraint(binRelOp->getOperator())) {
106  return getExpressionPair(binRelOp, element, identifier);
107  }
108  }
109  return {mk<UndefValue>(), mk<UndefValue>()};

112 Own<Condition> MakeIndexTransformer::constructPattern(const std::vector<std::string>& attributeTypes,
113  RamPattern& queryPattern, bool& indexable, VecOwn<Condition> conditionList, int identifier) {
114  // Remaining conditions which cannot be handled by an index
115  Own<Condition> condition;
116  auto addCondition = [&](Own<Condition> c) {
117  if (condition != nullptr) {
118  condition = mk<Conjunction>(std::move(condition), std::move(c));
119  } else {
120  condition = std::move(c);
121  }
122  };
123 
124  // transform condition list so that every strict inequality becomes a weak inequality + filter
125  // e.g. Tuple[level, element] < <expr> --> Tuple[level, element] <= <expr> and Tuple[level, element] !=
126  // <expr>
127  std::vector<std::unique_ptr<Condition>> toAppend;
128  auto it = conditionList.begin();
129  while (it != conditionList.end()) {
130  auto* binRelOp = dynamic_cast<Constraint*>(it->get());
131  if (binRelOp == nullptr) {
132  ++it;
133  continue;
134  }
135 
136  bool transformable = false;
137 
138  if (isStrictIneqConstraint(binRelOp->getOperator())) {
139  if (const auto* lhs = dynamic_cast<const TupleElement*>(&binRelOp->getLHS())) {
140  const Expression* rhs = &binRelOp->getRHS();
141  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
142  transformable = true;
143  }
144  }
145  if (const auto* rhs = dynamic_cast<const TupleElement*>(&binRelOp->getRHS())) {
146  const Expression* lhs = &binRelOp->getLHS();
147  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
148  transformable = true;
149  }
150  }
151  }
152 
153  if (transformable) {
154  // append the weak version of inequality
155  toAppend.emplace_back(
156  std::make_unique<Constraint>(convertStrictToWeakIneqConstraint(binRelOp->getOperator()),
157  clone(&binRelOp->getLHS()), clone(&binRelOp->getRHS())));
158  // append the != constraint
159  toAppend.emplace_back(
160  std::make_unique<Constraint>(convertStrictToNotEqualConstraint(binRelOp->getOperator()),
161  clone(&binRelOp->getLHS()), clone(&binRelOp->getRHS())));
162 
163  // remove the strict version of inequality
164  it = conditionList.erase(it);
165  } else {
166  ++it;
167  }
168  }
169 
170  std::transform(toAppend.begin(), toAppend.end(), std::back_inserter(conditionList),
171  [](const std::unique_ptr<Condition>& cond) { return std::move(clone(cond)); });
172 
173  // Build query pattern and remaining condition
174  for (auto& cond : conditionList) {
175  size_t element = 0;
176  Own<Expression> lowerExpression;
177  Own<Expression> upperExpression;
178  std::tie(lowerExpression, upperExpression) = getLowerUpperExpression(cond.get(), element, identifier);
179 
180  // we have new bounds if at least one is defined
181  if (!isUndefValue(lowerExpression.get()) || !isUndefValue(upperExpression.get())) {
182  // if no previous bounds are set then just assign them, consider both bounds to be set (but not
183  // necessarily defined) in all remaining cases
184  auto type = attributeTypes[element];
185  indexable = true;
186  if (isUndefValue(queryPattern.first[element].get()) &&
187  isUndefValue(queryPattern.second[element].get())) {
188  queryPattern.first[element] = std::move(lowerExpression);
189  queryPattern.second[element] = std::move(upperExpression);
190  // if lower bound is undefined and we have a new lower bound then assign it
191  } else if (isUndefValue(queryPattern.first[element].get()) &&
192  !isUndefValue(lowerExpression.get()) && isUndefValue(upperExpression.get())) {
193  queryPattern.first[element] = std::move(lowerExpression);
194  // if upper bound is undefined and we have a new upper bound then assign it
195  } else if (isUndefValue(queryPattern.second[element].get()) &&
196  isUndefValue(lowerExpression.get()) && !isUndefValue(upperExpression.get())) {
197  queryPattern.second[element] = std::move(upperExpression);
198  // if both bounds are defined ...
199  // and equal then we have a previous equality constraint i.e. Tuple[level, element] = <expr1>
200  } else if (!isUndefValue(queryPattern.first[element].get()) &&
201  !isUndefValue(queryPattern.second[element].get()) &&
202  (*(queryPattern.first[element]) == *(queryPattern.second[element]))) {
203  // new equality constraint i.e. Tuple[level, element] = <expr2>
204  // simply hoist <expr1> = <expr2> to the outer loop
205  if (!isUndefValue(lowerExpression.get()) && !isUndefValue(upperExpression.get())) {
206  // FIXME: `FEQ` handling; need to know if the expr is a float exp or not
207  addCondition(mk<Constraint>(getEqConstraint(type),
208  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
209  }
210  // new lower bound i.e. Tuple[level, element] >= <expr2>
211  // we need to hoist <expr1> >= <expr2> to the outer loop
212  else if (!isUndefValue(lowerExpression.get()) && isUndefValue(upperExpression.get())) {
213  addCondition(mk<Constraint>(getGreaterEqualConstraint(type),
214  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
215  }
216  // new upper bound i.e. Tuple[level, element] <= <expr2>
217  // we need to hoist <expr1> <= <expr2> to the outer loop
218  else if (isUndefValue(lowerExpression.get()) && !isUndefValue(upperExpression.get())) {
219  addCondition(mk<Constraint>(getLessEqualConstraint(type),
220  souffle::clone(queryPattern.first[element]), std::move(upperExpression)));
221  }
222  // if either bound is defined but they aren't equal we must consider the cases for updating
223  // them note that at this point we know that if we have a lower/upper bound it can't be the
224  // first one
225  } else if (!isUndefValue(queryPattern.first[element].get()) ||
226  !isUndefValue(queryPattern.second[element].get())) {
227  // if we have a new equality constraint and previous inequality constraints
228  if (!isUndefValue(lowerExpression.get()) && !isUndefValue(upperExpression.get()) &&
229  *lowerExpression == *upperExpression) {
230  // if Tuple[level, element] >= <expr1> and we see Tuple[level, element] = <expr2>
231  // need to hoist <expr2> >= <expr1> to the outer loop
232  if (!isUndefValue(queryPattern.first[element].get())) {
233  addCondition(mk<Constraint>(getGreaterEqualConstraint(type),
234  souffle::clone(lowerExpression), std::move(queryPattern.first[element])));
235  }
236  // if Tuple[level, element] <= <expr1> and we see Tuple[level, element] = <expr2>
237  // need to hoist <expr2> <= <expr1> to the outer loop
238  if (!isUndefValue(queryPattern.second[element].get())) {
239  addCondition(mk<Constraint>(getLessEqualConstraint(type),
240  souffle::clone(upperExpression), std::move(queryPattern.second[element])));
241  }
242  // finally replace bounds with equality constraint
243  queryPattern.first[element] = std::move(lowerExpression);
244  queryPattern.second[element] = std::move(upperExpression);
245  // if we have a new lower bound
246  } else if (!isUndefValue(lowerExpression.get())) {
247  // we want the tightest lower bound so we take the max
248  VecOwn<Expression> maxArguments;
249  maxArguments.push_back(std::move(queryPattern.first[element]));
250  maxArguments.push_back(std::move(lowerExpression));
251 
252  queryPattern.first[element] =
253  mk<IntrinsicOperator>(getMaxOp(type), std::move(maxArguments));
254  // if we have a new upper bound
255  } else if (!isUndefValue(upperExpression.get())) {
256  // we want the tightest upper bound so we take the min
257  VecOwn<Expression> minArguments;
258  minArguments.push_back(std::move(queryPattern.second[element]));
259  minArguments.push_back(std::move(upperExpression));
260 
261  queryPattern.second[element] =
262  mk<IntrinsicOperator>(getMinOp(type), std::move(minArguments));
263  }
264  }
265  } else {
266  addCondition(std::move(cond));
267  }
268  }
269 
270  // Avoid null-pointers for condition and query pattern
271  if (condition == nullptr) {
272  condition = mk<True>();
273  }
274  return condition;
275 }

277 Own<Operation> MakeIndexTransformer::rewriteAggregate(const Aggregate* agg) {
278  if (dynamic_cast<const True*>(&agg->getCondition()) == nullptr) {
279  const Relation& rel = agg->getRelation();
280  int identifier = agg->getTupleId();
281  RamPattern queryPattern;
282  for (unsigned int i = 0; i < rel.getArity(); ++i) {
283  queryPattern.first.push_back(mk<UndefValue>());
284  queryPattern.second.push_back(mk<UndefValue>());
285  }
286 
287  bool indexable = false;
288  Own<Condition> condition = constructPattern(rel.getAttributeTypes(), queryPattern, indexable,
289  toConjunctionList(&agg->getCondition()), identifier);
290  if (indexable) {
291  return mk<IndexAggregate>(souffle::clone(&agg->getOperation()), agg->getFunction(),
292  mk<RelationReference>(&rel), souffle::clone(&agg->getExpression()), std::move(condition),
293  std::move(queryPattern), agg->getTupleId());
294  }
295  }
296  return nullptr;
297 }

299 Own<Operation> MakeIndexTransformer::rewriteScan(const Scan* scan) {
300  if (const auto* filter = dynamic_cast<const Filter*>(&scan->getOperation())) {
301  const Relation& rel = scan->getRelation();
302  const int identifier = scan->getTupleId();
303  RamPattern queryPattern;
304  for (unsigned int i = 0; i < rel.getArity(); ++i) {
305  queryPattern.first.push_back(mk<UndefValue>());
306  queryPattern.second.push_back(mk<UndefValue>());
307  }
308 
309  bool indexable = false;
310  Own<Condition> condition = constructPattern(rel.getAttributeTypes(), queryPattern, indexable,
311  toConjunctionList(&filter->getCondition()), identifier);
312  if (indexable) {
313  Own<Operation> op = souffle::clone(&filter->getOperation());
314  if (!isTrue(condition.get())) {
315  op = mk<Filter>(std::move(condition), std::move(op));
316  }
317  return mk<IndexScan>(mk<RelationReference>(&rel), identifier, std::move(queryPattern),
318  std::move(op), scan->getProfileText());
319  }
320  }
321  return nullptr;
322 }

326  const Relation& rel = iscan->getRelation();
327  const int identifier = iscan->getTupleId();

338  if (indexable) {

341  Own<Operation> op = souffle::clone(&filter->getOperation());

346  std::move(op), iscan->getProfileText());

352 bool MakeIndexTransformer::makeIndex(Program& program) {
353  bool changed = false;
354  visitDepthFirst(program, [&](const Query& query) {
355  std::function<Own<Node>(Own<Node>)> scanRewriter = [&](Own<Node> node) -> Own<Node> {
356  if (const Scan* scan = dynamic_cast<Scan*>(node.get())) {
357  if (scan->getRelation().getRepresentation() != RelationRepresentation::INFO) {
358  if (Own<Operation> op = rewriteScan(scan)) {
359  changed = true;
360  node = std::move(op);
361  }
362  }
363  } else if (const IndexScan* iscan = dynamic_cast<IndexScan*>(node.get())) {
364  if (Own<Operation> op = rewriteIndexScan(iscan)) {
365  changed = true;
366  node = std::move(op);
367  }
368  } else if (const Aggregate* agg = dynamic_cast<Aggregate*>(node.get())) {
369  if (Own<Operation> op = rewriteAggregate(agg)) {
370  changed = true;
371  node = std::move(op);
372  }
373  }
374  node->apply(makeLambdaRamMapper(scanRewriter));
375  return node;
376  };
377  const_cast<Query*>(&query)->apply(makeLambdaRamMapper(scanRewriter));
378  });
379  return changed;
380 }
