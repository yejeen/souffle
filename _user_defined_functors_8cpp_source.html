38  Own<Node> operator()(Own<Node> node) const override {
39  node->apply(*this);
40 
41  if (auto* userFunctor = dynamic_cast<UserDefinedFunctor*>(node.get())) {
42  const FunctorDeclaration* functorDeclaration =
43  getIf(program.getFunctorDeclarations(), [&](const FunctorDeclaration* current) {
44  return current->getName() == userFunctor->getName();
45  });
46 
47  // Check if the functor has been declared
48  if (functorDeclaration == nullptr) {
49  report.addError("User-defined functor hasn't been declared", userFunctor->getSrcLoc());
50  return node;
51  }
52 
53  // Check arity correctness.
54  if (functorDeclaration->getArity() != userFunctor->getArguments().size()) {
55  report.addError("Mismatching number of arguments of functor", userFunctor->getSrcLoc());
56  return node;
57  }
58 
59  // Set types of functor instance based on its declaration.
60  userFunctor->setTypes(functorDeclaration->getArgsTypes(), functorDeclaration->getReturnType(),
61  functorDeclaration->isStateful());
62 
63  changed = true;
64  }
65  return node;
66  }
67  };
