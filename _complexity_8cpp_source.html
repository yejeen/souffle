32 int ComplexityAnalysis::getComplexity(const Node* node) const {
33  // visitor
34  class ValueComplexityVisitor : public Visitor<int> {
35  public:
36  // conjunction
37  int visitConjunction(const Conjunction& conj) override {
38  return visit(conj.getLHS()) + visit(conj.getRHS());
39  }
40 
41  // negation
42  int visitNegation(const Negation& neg) override {
43  return visit(neg.getOperand());
44  }
45 
46  // existence check
47  int visitExistenceCheck(const ExistenceCheck&) override {
48  return 2;
49  }
50 
51  // provenance existence check
52  int visitProvenanceExistenceCheck(const ProvenanceExistenceCheck&) override {
53  return 2;
54  }
55 
56  // emptiness check
57  int visitEmptinessCheck(const EmptinessCheck& emptiness) override {
58  // emptiness check for nullary relations is for free; others have weight one
59  return (emptiness.getRelation().getArity() > 0) ? 1 : 0;
60  }
61 
62  // default rule
63  int visitNode(const Node&) override {
64  return 0;
65  }
66  };
67 
68  assert((isA<Expression>(node) || isA<Condition>(node)) && "not an expression/condition/operation");
69  return ValueComplexityVisitor().visit(node);
70 }
