86 template <typename R>
87 struct TypeVisitor {
88  virtual ~TypeVisitor() = default;
89 
90  R operator()(const Type& type) const {
91  return visit(type);
92  }
93 
94 #define FORWARD(TYPE) \
95  if (auto* t = dynamic_cast<const TYPE##Type*>(&type)) return visit##TYPE##Type(*t);
96 
97  virtual R visit(const Type& type) const {
98  FORWARD(Constant);
99  FORWARD(Subset);
100  FORWARD(Union);
101  FORWARD(Record);
102  FORWARD(AlgebraicData);
103 
104  fatal("Unsupported type encountered!");
105  }
106 #undef FORWARD
107 
108 #define VISIT(TYPE) \
109  virtual R visit##TYPE##Type(const TYPE##Type& type) const { \
110  return visitType(type); \
111  }
112 
113  VISIT(Constant)
114  VISIT(Subset)
115  VISIT(Union)
116  VISIT(Record)
117  VISIT(AlgebraicData)
118 
119  virtual R visitType(const Type& /*type*/) const {
120  return R();
121  }
122 
123 #undef VISIT
124 };

126 /**
127  * A visitor for types visiting each type only once (effectively breaking
128  * recursive cycles).
129  */
130 template <typename R>
131 class VisitOnceTypeVisitor : public TypeVisitor<R> {
132 protected:
133  mutable std::map<const Type*, R> seen;
134 
135 public:
136  R visit(const Type& type) const override {
137  auto pos = seen.find(&type);
138  if (pos != seen.end()) {
139  return pos->second;
140  }
141  auto& res = seen[&type]; // mark as seen
142  return res = TypeVisitor<R>::visit(type);
143  }
144 };

146 /**
147  * Determines whether the given type is a sub-type of the given root type.
148  */
149 bool isOfRootType(const Type& type, const Type& root) {
150  struct visitor : public VisitOnceTypeVisitor<bool> {
151  const Type& root;
152 
153  explicit visitor(const Type& root) : root(root) {}
154 
155  bool visitConstantType(const ConstantType& type) const override {
156  return type == root;
157  }
158  bool visitSubsetType(const SubsetType& type) const override {
159  return type == root || isOfRootType(type.getBaseType(), root);
160  }
161 
162  bool visitAlgebraicDataType(const AlgebraicDataType& type) const override {
163  return type == root;
164  }
165 
166  bool visitUnionType(const UnionType& type) const override {
167  return type == root ||
168  all_of(type.getElementTypes(), [&](const Type* cur) { return this->visit(*cur); });
169  }
170 
171  bool visitRecordType(const RecordType& type) const override {
172  return type == root;
173  }
174 
175  bool visitType(const Type& /*unused*/) const override {
176  return false;
177  }
178  };
179 
180  return visitor(root).visit(type);
181 }

198 std::string getTypeQualifier(const Type& type) {
199  struct visitor : public VisitOnceTypeVisitor<std::string> {
200  std::string visitUnionType(const UnionType& type) const override {
201  return tfm::format("%s[%s]", visitType(type),
202  join(type.getElementTypes(), ", ",
203  [&](std::ostream& out, const auto* elementType) { out << visit(*elementType); }));
204  }
205 
206  std::string visitRecordType(const RecordType& type) const override {
207  return tfm::format("%s{%s}", visitType(type),
208  join(type.getFields(), ", ",
209  [&](std::ostream& out, const auto* field) { out << visit(*field); }));
210  }
211 
212  std::string visitType(const Type& type) const override {
213  std::string str;
214 
215  if (isOfKind(type, TypeAttribute::Signed)) {
216  str.append("i");
217  } else if (isOfKind(type, TypeAttribute::Unsigned)) {
218  str.append("u");
219  } else if (isOfKind(type, TypeAttribute::Float)) {
220  str.append("f");
221  } else if (isOfKind(type, TypeAttribute::Symbol)) {
222  str.append("s");
223  } else if (isOfKind(type, TypeAttribute::Record)) {
224  str.append("r");
225  } else if (isOfKind(type, TypeAttribute::ADT)) {
226  str.append("+");
227  } else {
228  fatal("Unsupported kind");
229  }
230 
231  str.append(":");
232  str.append(toString(type.getName()));
233  seen[&type] = str;
234  return str;
235  }
236  };
237 
238  return visitor().visit(type);
239 }

241 bool isSubtypeOf(const Type& a, const Type& b) {
242  assert(&a.getTypeEnvironment() == &b.getTypeEnvironment() &&
243  "Types must be in the same type environment");
244 
245  if (isOfRootType(a, b)) {
246  return true;
247  }
248 
249  if (isA<UnionType>(a)) {
250  return all_of(static_cast<const UnionType&>(a).getElementTypes(),
251  [&b](const Type* type) { return isSubtypeOf(*type, b); });
252  }
253 
254  if (isA<UnionType>(b)) {
255  return any_of(static_cast<const UnionType&>(b).getElementTypes(),
256  [&a](const Type* type) { return isSubtypeOf(a, *type); });
257  }
258 
259  return false;
260 }

269 TypeSet getGreatestCommonSubtypes(const Type& a, const Type& b) {
270  assert(&a.getTypeEnvironment() == &b.getTypeEnvironment() &&
271  "Types must be in the same type environment");
272 
273  if (isSubtypeOf(a, b)) {
274  return TypeSet(a);
275  }
276  if (isSubtypeOf(b, a)) {
277  return TypeSet(b);
278  }
279 
280  TypeSet res;
281  if (isA<UnionType>(a) && isA<UnionType>(b)) {
282  // collect common sub-types of union types
283  struct collector : public TypeVisitor<void> {
284  const Type& b;
285  TypeSet& res;
286  collector(const Type& b, TypeSet& res) : b(b), res(res) {}
287 
288  void visit(const Type& type) const override {
289  if (isSubtypeOf(type, b)) {
290  res.insert(type);
291  } else {
292  TypeVisitor<void>::visit(type);
293  }
294  }
295  void visitUnionType(const UnionType& type) const override {
296  for (const auto& cur : type.getElementTypes()) {
297  visit(*cur);
298  }
299  }
300  };
301 
302  // collect all common sub-types
303  collector(b, res).visit(a);
304  }
305 
306  // otherwise there is no common super type
307  return res;
308 }

310 TypeSet getGreatestCommonSubtypes(const TypeSet& set) {
311  // Edge cases.
312  if (set.empty() || set.isAll()) {
313  return TypeSet();
314  }
315 
316  TypeSet greatestCommonSubtypes;
317  greatestCommonSubtypes.insert(*set.begin());
318 
319  for (auto& type : set) {
320  greatestCommonSubtypes = getGreatestCommonSubtypes(TypeSet(type), greatestCommonSubtypes);
321  }
322 
323  return greatestCommonSubtypes;
324 }

326 TypeSet getGreatestCommonSubtypes(const TypeSet& a, const TypeSet& b) {
327  // special cases
328  if (a.empty()) {
329  return a;
330  }
331  if (b.empty()) {
332  return b;
333  }
334 
335  if (a.isAll()) {
336  return b;
337  }
338  if (b.isAll()) {
339  return a;

342  // compute pairwise greatest common sub types
343  TypeSet res;
344  for (const Type& x : a) {

352 bool haveCommonSupertype(const Type& a, const Type& b) {
353  assert(&a.getTypeEnvironment() == &b.getTypeEnvironment() &&
354  "Types must be in the same type environment");
355 
356  if (a == b) {
357  return true;
358  }
359 
360  if (isSubtypeOf(a, b) || isSubtypeOf(b, a)) {
361  return true;
362  }
363 
364  return any_of(a.getTypeEnvironment().getTypes(),
365  [&](const Type& type) { return isSubtypeOf(a, type) && isSubtypeOf(b, type); });
366 }
