143 size_t getClauseNum(const Program* program, const Clause* clause) {
144  // TODO (azreika): This number might change between the provenance transformer and the AST->RAM
145  // translation. Might need a better way to assign IDs to clauses... (see PR #1288).
146  const Relation* rel = getRelation(*program, clause->getHead()->getQualifiedName());
147  assert(rel != nullptr && "clause relation does not exist");
148 
149  size_t clauseNum = 1;
150  for (const auto* cur : getClauses(*program, *rel)) {
151  bool isFact = cur->getBodyLiterals().empty();
152  if (cur == clause) {
153  return isFact ? 0 : clauseNum;
154  }
155 
156  if (!isFact) {
157  clauseNum++;
158  }
159  }
160 
161  fatal("clause does not exist");
162 }

207 bool isFact(const Clause& clause) {
208  // there must be a head
209  if (clause.getHead() == nullptr) {
210  return false;
211  }
212  // there must not be any body clauses
213  if (!clause.getBodyLiterals().empty()) {
214  return false;
215  }
216 
217  // and there are no aggregates
218  bool hasAggregatesOrMultiResultFunctor = false;
219  visitDepthFirst(*clause.getHead(), [&](const Argument& arg) {
220  if (dynamic_cast<const Aggregator*>(&arg)) {
221  hasAggregatesOrMultiResultFunctor = true;
222  }
223 
224  auto func = dynamic_cast<const IntrinsicFunctor*>(&arg);
225  auto info = func ? func->getFunctionInfo() : nullptr;
226  if (info && info->multipleResults) {
227  hasAggregatesOrMultiResultFunctor = true;
228  }
229  });
230  return !hasAggregatesOrMultiResultFunctor;
231 }

251 Clause* reorderAtoms(const Clause* clause, const std::vector<unsigned int>& newOrder) {
252  // Find all atom positions
253  std::vector<unsigned int> atomPositions;
254  std::vector<Literal*> bodyLiterals = clause->getBodyLiterals();
255  for (unsigned int i = 0; i < bodyLiterals.size(); i++) {
256  if (isA<Atom>(bodyLiterals[i])) {
257  atomPositions.push_back(i);
258  }
259  }
260 
261  // Validate given order
262  assert(newOrder.size() == atomPositions.size());
263  std::vector<unsigned int> nopOrder;
264  for (unsigned int i = 0; i < atomPositions.size(); i++) {
265  nopOrder.push_back(i);
266  }
267  assert(std::is_permutation(nopOrder.begin(), nopOrder.end(), newOrder.begin()));
268 
269  // Create a new clause with the given atom order, leaving the rest unchanged
270  Clause* newClause = cloneHead(clause);
271  unsigned int currentAtom = 0;
272  for (unsigned int currentLiteral = 0; currentLiteral < bodyLiterals.size(); currentLiteral++) {
273  Literal* literalToAdd = bodyLiterals[currentLiteral];
274  if (isA<Atom>(literalToAdd)) {
275  // Atoms should be reordered
276  literalToAdd = bodyLiterals[atomPositions[newOrder[currentAtom++]]];
277  }
278  newClause->addToBody(souffle::clone(literalToAdd));
279  }
280 
281  return newClause;
282 }

294 IntrinsicFunctors validOverloads(const analysis::TypeAnalysis& typing, const ast::IntrinsicFunctor& func) {
295  auto typeAttrs = [&](const Argument* arg) -> std::set<TypeAttribute> {
296  auto&& types = typing.getTypes(arg);
297  if (types.isAll())
298  return {TypeAttribute::Signed, TypeAttribute::Unsigned, TypeAttribute::Float,
299  TypeAttribute::Symbol, TypeAttribute::Record};
300 
301  std::set<TypeAttribute> tyAttrs;
302  for (auto&& ty : types)
303  tyAttrs.insert(getTypeAttribute(ty));
304  return tyAttrs;
305  };
306  auto retTys = typeAttrs(&func);
307  auto argTys = map(func.getArguments(), typeAttrs);
308 
309  auto candidates =
310  filterNot(functorBuiltIn(func.getFunction()), [&](const IntrinsicFunctorInfo& x) -> bool {
311  if (!x.variadic && argTys.size() != x.params.size()) return true; // arity mismatch?
312 
313  for (size_t i = 0; i < argTys.size(); ++i)
314  if (!contains(argTys[i], x.params[x.variadic ? 0 : i])) return true;
315 
316  return !contains(retTys, x.result);
317  });
318 
319  std::sort(candidates.begin(), candidates.end(),
320  [&](const IntrinsicFunctorInfo& a, const IntrinsicFunctorInfo& b) {
321  if (a.result != b.result) return a.result < b.result;
322  if (a.variadic != b.variadic) return a.variadic < b.variadic;
323  return std::lexicographical_compare(
324  a.params.begin(), a.params.end(), b.params.begin(), b.params.end());
325  });
326  return candidates;
327 }

329 bool renameAtoms(Node& node, const std::map<QualifiedName, QualifiedName>& oldToNew) {
330  struct rename_atoms : public NodeMapper {
331  mutable bool changed{false};
332  const std::map<QualifiedName, QualifiedName>& oldToNew;
333  rename_atoms(const std::map<QualifiedName, QualifiedName>& oldToNew) : oldToNew(oldToNew) {}
334  Own<Node> operator()(Own<Node> node) const override {
335  node->apply(*this);
336  if (auto* atom = dynamic_cast<Atom*>(node.get())) {
337  if (contains(oldToNew, atom->getQualifiedName())) {
338  auto renamedAtom = souffle::clone(atom);
339  renamedAtom->setQualifiedName(oldToNew.at(atom->getQualifiedName()));
340  changed = true;
341  return renamedAtom;
342  }
343  }
344  return node;
345  }
346  };
347  rename_atoms update(oldToNew);
348  node.apply(update);
349  return update.changed;
350 }
