30 std::vector<unsigned int> SipsMetric::getReordering(const Clause* clause) const {
31  BindingStore bindingStore(clause);
32  auto atoms = getBodyLiterals<Atom>(*clause);
33  std::vector<unsigned int> newOrder(atoms.size());
34 
35  size_t numAdded = 0;
36  while (numAdded < atoms.size()) {
37  // grab the index of the next atom, based on the SIPS function
38  const auto& costs = evaluateCosts(atoms, bindingStore);
39  auto minIdx = std::distance(costs.begin(), std::min_element(costs.begin(), costs.end()));
40  const auto* nextAtom = atoms[minIdx];
41  assert(nextAtom != nullptr && "nullptr atoms should have maximal cost");
42 
43  // set all arguments that are variables as bound
44  for (const auto* arg : nextAtom->getArguments()) {
45  if (const auto* var = dynamic_cast<const Variable*>(arg)) {
46  bindingStore.bindVariableStrongly(var->getName());
47  }
48  }
49 
50  newOrder[numAdded] = minIdx; // add to the ordering
51  atoms[minIdx] = nullptr; // mark as done
52  numAdded++; // move on
53  }
54 
55  return newOrder;
56 }

88  const std::vector<Atom*> atoms, const BindingStore& bindingStore) const {

111 std::vector<double> MaxBoundSips::evaluateCosts(
112  const std::vector<Atom*> atoms, const BindingStore& bindingStore) const {
113  // Goal: prioritise (1) all-bound, then (2) max number of bound vars, then (3) left-most
114  std::vector<double> cost;
115  for (const auto* atom : atoms) {
116  if (atom == nullptr) {
117  cost.push_back(std::numeric_limits<double>::max());
118  continue;
119  }
120 
121  int arity = atom->getArity();
122  int numBound = bindingStore.numBoundArguments(atom);
123  if (arity == numBound) {
124  // Always better than anything else
125  cost.push_back(0);
126  } else if (numBound == 0) {
127  // Always worse than any number of bound vars
128  cost.push_back(2);
129  } else {
130  // Between 0 and 1, decreasing with more num bound
131  cost.push_back(1 / numBound);
132  }
133  }
134  assert(atoms.size() == cost.size() && "each atom should have exactly one cost");
135  return cost;
136 }

180 std::vector<double> LeastFreeVarsSips::evaluateCosts(
181  const std::vector<Atom*> atoms, const BindingStore& bindingStore) const {
182  // Goal: choose the atom with the least amount of unbound variables
183  std::vector<double> cost;
184  for (const auto* atom : atoms) {
185  if (atom == nullptr) {
186  cost.push_back(std::numeric_limits<double>::max());
187  continue;
188  }
189 
190  // use a set to hold all free variables to avoid double-counting
191  std::set<std::string> freeVars;
192  visitDepthFirst(*atom, [&](const Variable& var) {
193  if (bindingStore.isBound(var.getName())) {
194  freeVars.insert(var.getName());
195  }
196  });
197  cost.push_back(freeVars.size());
198  }
199  return cost;

202 std::vector<double> ProfileUseSips::evaluateCosts(
203  const std::vector<Atom*> atoms, const BindingStore& bindingStore) const {
204  // Goal: reorder based on the given profiling information
205  // Metric: cost(atom_R) = log(|atom_R|) * #free/#args
206  // - exception: propositions are prioritised
207  std::vector<double> cost;
208  for (const auto* atom : atoms) {
209  if (atom == nullptr) {
210  cost.push_back(std::numeric_limits<double>::max());
211  continue;
212  }
213 
214  // prioritise propositions
215  int arity = atom->getArity();
216  if (arity == 0) {
217  cost.push_back(0);
218  continue;
219  }
220 
221  // calculate log(|R|) * #free/#args
222  int numBound = bindingStore.numBoundArguments(atom);
223  int numFree = arity - numBound;
224  double value = log(profileUse.getRelationSize(atom->getQualifiedName()));
225  value *= (numFree * 1.0) / arity;
226  }
227  return cost;
228 }
