36 bool ADTtoRecordsTransformer::transform(TranslationUnit& tu) {
37  struct ADTsFuneral : public NodeMapper {
38  mutable bool changed{false};
39  TranslationUnit& tu;
40  const analysis::SumTypeBranchesAnalysis& sumTypesBranches =
41  *tu.getAnalysis<analysis::SumTypeBranchesAnalysis>();
42 
43  ADTsFuneral(TranslationUnit& tu) : tu(tu){};
44 
45  Own<Node> operator()(Own<Node> node) const override {
46  // Rewrite sub-expressions first
47  node->apply(*this);
48 
49  if (!isA<BranchInit>(node)) {
50  return node;
51  }
52 
53  changed = true;
54 
55  auto& adt = *as<BranchInit>(node);
56 
57  auto& type = sumTypesBranches.unsafeGetType(adt.getConstructor());
58  assert(isA<analysis::AlgebraicDataType>(type));
59 
60  auto& branches = as<analysis::AlgebraicDataType>(type)->getBranches();
61 
62  // Find branch ID.
63  analysis::AlgebraicDataType::Branch searchDummy{adt.getConstructor(), {}};
64  auto iterToBranch = std::lower_bound(branches.begin(), branches.end(), searchDummy,
65  [](const analysis::AlgebraicDataType::Branch& left,
66  const analysis::AlgebraicDataType::Branch& right) {
67  return left.name < right.name;
68  });
69 
70  // Branch id corresponds to the position in lexicographical ordering.
71  auto branchID = std::distance(std::begin(branches), iterToBranch);
72 
73  // Collect branch arguments
74  VecOwn<Argument> branchArguments;
75  for (auto* arg : adt.getArguments()) {
76  branchArguments.emplace_back(arg->clone());
77  }
78 
79  // Branch is stored either as [branch_id, [arguments]]
80  // or [branch_id, argument] in case of a single argument.
81  auto branchArgs = [&]() -> Own<ast::Argument> {
82  if (branchArguments.size() != 1) {
83  return mk<Argument, RecordInit>(std::move(branchArguments));
84  } else {
85  return std::move(branchArguments.at(0));
86  }
87  }();
88 
89  // Arguments for the resulting record [branch_id, branch_args].
90  VecOwn<Argument> finalRecordArgs;
91 
92  finalRecordArgs.push_back(mk<Argument, NumericConstant>(branchID));
93  finalRecordArgs.push_back(std::move(branchArgs));
94 
95  return mk<RecordInit>(std::move(finalRecordArgs), adt.getSrcLoc());
96  }
97  };
98 
99  ADTsFuneral mapper(tu);
100  tu.getProgram().apply(mapper);
101  return mapper.changed;
102 }
