44 std::string MaterializeSingletonAggregationTransformer::findUniqueVariableName(const Clause& clause) {
45  static int counter = 0;
46  std::set<std::string> variableNames;
47  visitDepthFirst(clause, [&](const ast::Variable& variable) { variableNames.insert(variable.getName()); });
48  std::string candidateVariableName = "z"; // completely arbitrary
49  while (variableNames.find(candidateVariableName) != variableNames.end()) {
50  candidateVariableName = "z" + toString(counter++);
51  }
52  return candidateVariableName;
53 }

65 bool MaterializeSingletonAggregationTransformer::transform(TranslationUnit& translationUnit) {
66  Program& program = translationUnit.getProgram();
67  std::set<std::pair<Aggregator*, Clause*>> pairs;
68  // collect references to clause / aggregate pairs
69  visitDepthFirst(program, [&](const Clause& clause) {
70  visitDepthFirst(clause, [&](const Aggregator& agg) {
71  // if the aggregate isn't single valued
72  // (ie it relies on a grounding from the outer scope)
73  // or it's the only atom in the clause, then there's no point materialising it!
74  if (!isSingleValued(agg, clause) || clause.getBodyLiterals().size() == 1) {
75  return;
76  }
77  auto* foundAggregate = const_cast<Aggregator*>(&agg);
78  auto* foundClause = const_cast<Clause*>(&clause);
79  pairs.insert(std::make_pair(foundAggregate, foundClause));
80  });
81  });
82  for (auto pair : pairs) {
83  // Clone the aggregate that we're going to be deleting.
84  auto aggregate = souffle::clone(pair.first);
85  Clause* clause = pair.second;
86  // synthesise an aggregate relation
87  // __agg_rel_0()
88  auto aggRel = mk<Relation>();
89  auto aggHead = mk<Atom>();
90  auto aggClause = mk<Clause>();
91 
92  std::string aggRelName = findUniqueAggregateRelationName(program);
93  aggRel->setQualifiedName(aggRelName);
94  aggHead->setQualifiedName(aggRelName);
95 
96  // create a synthesised variable to replace the aggregate term!
97  std::string variableName = findUniqueVariableName(*clause);
98  auto variable = mk<ast::Variable>(variableName);
99 
100  // __agg_rel_0(z) :- ...
101  aggHead->addArgument(souffle::clone(variable));
102  aggRel->addAttribute(mk<Attribute>(variableName, "number"));
103  aggClause->setHead(souffle::clone(aggHead));
104 
105  // A(x) :- x = sum .., B(x).
106  // -> A(x) :- x = z, B(x), __agg_rel_0(z).
107  auto equalityLiteral = mk<BinaryConstraint>(
108  BinaryConstraintOp::EQ, souffle::clone(variable), souffle::clone(aggregate));
109  // __agg_rel_0(z) :- z = sum ...
110  aggClause->addToBody(std::move(equalityLiteral));
111  program.addRelation(std::move(aggRel));
112  program.addClause(std::move(aggClause));
113 
114  // the only thing left to do is just replace the aggregate terms in the original
115  // clause with the synthesised variable
116  struct replaceAggregate : public NodeMapper {
117  const Aggregator& aggregate;
118  const Own<ast::Variable> variable;
119  replaceAggregate(const Aggregator& aggregate, Own<ast::Variable> variable)
120  : aggregate(aggregate), variable(std::move(variable)) {}
121  Own<Node> operator()(Own<Node> node) const override {
122  assert(node != nullptr);
123  if (auto* current = dynamic_cast<Aggregator*>(node.get())) {
124  if (*current == aggregate) {
125  auto replacement = souffle::clone(variable);
126  assert(replacement != nullptr);
127  return replacement;
128  }
129  }
130  node->apply(*this);
131  assert(node != nullptr);
132  return node;
133  }
134  };
135  replaceAggregate update(*aggregate, std::move(variable));
136  clause->apply(update);
137  clause->addToBody(std::move(aggHead));
138  }
139  return pairs.size() > 0;
140 }

142 bool MaterializeSingletonAggregationTransformer::isSingleValued(const Aggregator& agg, const Clause& clause) {
143  std::map<std::string, int> occurrences;
144  visitDepthFirst(clause, [&](const ast::Variable& v) {
145  if (occurrences.find(v.getName()) == occurrences.end()) {
146  occurrences[v.getName()] = 0;
147  }
148  occurrences[v.getName()] = occurrences[v.getName()] + 1;
149  });
150  std::set<std::string> aggVariables;
151  visitDepthFirst(agg, [&](const ast::Variable& v) {
152  aggVariables.insert(v.getName());
153  occurrences[v.getName()] = occurrences[v.getName()] - 1;
154  });
155  for (std::string variableName : aggVariables) {
156  if (occurrences[variableName] != 0) {
157  return false;
158  }
159  }
160  return true;
161 }
