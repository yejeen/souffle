28 InterpreterRelation::InterpreterRelation(std::size_t arity, std::size_t auxiliaryArity, std::string name,
29  std::vector<std::string> attributeTypes, const MinIndexSelection& orderSet, IndexFactory factory)
30  : relName(std::move(name)), arity(arity), auxiliaryArity(auxiliaryArity),
31  attributeTypes(std::move(attributeTypes)) {
32  for (auto order : orderSet.getAllOrders()) {
33  // Expand the order to a total order
34  AttributeSet set{order.begin(), order.end()};
35 
36  for (std::size_t i = 0; i < arity; ++i) {
37  if (set.find(i) == set.end()) {
38  order.push_back(i);
39  }
40  }
41  indexes.push_back(factory(Order(order)));
42  }
43 
44  // Use the first index as default main index
45  main = indexes[0].get();
46 }

164 bool InterpreterIndirectRelation::insert(const TupleRef& tuple) {
165  if (main->contains(tuple)) {
166  return false;
167  }
168 
169  int blockIndex = numTuples / (BLOCK_SIZE / arity);
170  int tupleIndex = (numTuples % (BLOCK_SIZE / arity)) * arity;
171 
172  if (tupleIndex == 0) {
173  blockList.push_back(std::make_unique<RamDomain[]>(BLOCK_SIZE));
174  }
175 
176  RamDomain* newTuple = &blockList[blockIndex][tupleIndex];
177  for (size_t i = 0; i < arity; ++i) {
178  newTuple[i] = tuple[i];
179  }
180 
181  // update all indexes with new tuple
182  for (auto& cur : indexes) {
183  cur->insert(TupleRef(newTuple, arity));
184  }
185 
186  // increment relation size
187  numTuples++;
188 
189  return true;
190 }
