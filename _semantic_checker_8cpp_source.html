180 SemanticCheckerImpl::SemanticCheckerImpl(TranslationUnit& tu) : tu(tu) {
181  // suppress warnings for given relations
182  if (Global::config().has("suppress-warnings")) {
183  std::vector<std::string> suppressedRelations =
184  splitString(Global::config().get("suppress-warnings"), ',');
185 
186  if (std::find(suppressedRelations.begin(), suppressedRelations.end(), "*") !=
187  suppressedRelations.end()) {
188  // mute all relations
189  for (Relation* rel : program.getRelations()) {
190  rel->addQualifier(RelationQualifier::SUPPRESSED);
191  }
192  } else {
193  // mute only the given relations (if they exist)
194  for (auto& relname : suppressedRelations) {
195  const std::vector<std::string> comps = splitString(relname, '.');
196  if (!comps.empty()) {
197  // generate the relation identifier
198  QualifiedName relid(comps[0]);
199  for (size_t i = 1; i < comps.size(); i++) {
200  relid.append(comps[i]);
201  }
202 
203  // update suppressed qualifier if the relation is found
204  if (Relation* rel = getRelation(program, relid)) {
205  rel->addQualifier(RelationQualifier::SUPPRESSED);
206  }
207  }
208  }
209  }
210  }
211 
212  checkTypesDeclarations();
213 
214  checkBranchInits();
215 
216  // check rules
217  for (auto* rel : program.getRelations()) {
218  checkRelation(*rel);
219  }
220  for (auto* clause : program.getClauses()) {
221  checkClause(*clause);
222  }
223 
224  // Group clauses that stem from a single complex rule
225  // with multiple headers/disjunction etc. The grouping
226  // is performed via their source-location.
227  std::map<SrcLocation, std::set<const Clause*>> multiRuleMap;
228  for (auto* clause : program.getClauses()) {
229  // collect clauses of a multi rule, i.e., they have the same source locator
230  multiRuleMap[clause->getSrcLoc()].insert(clause);
231  }
232 
233  // check complex rule
234  for (const auto& multiRule : multiRuleMap) {
235  checkComplexRule(multiRule.second);
236  }
237 
238  checkNamespaces();
239  checkIO();
240  checkWitnessProblem();
241  checkInlining();
242 
243  // Run grounded terms checker
244  GroundedTermsChecker().verify(tu);
245 
246  // Check types
247  TypeChecker{tu}.run();
248 
249  // - stratification --
250  // check for cyclic dependencies
251  for (Relation* cur : program.getRelations()) {
252  size_t scc = sccGraph.getSCC(cur);
253  if (sccGraph.isRecursive(scc)) {
254  for (const Relation* cyclicRelation : sccGraph.getInternalRelations(scc)) {
255  // Negations and aggregations need to be stratified
256  const Literal* foundLiteral = nullptr;
257  bool hasNegation = hasClauseWithNegatedRelation(cyclicRelation, cur, &program, foundLiteral);
258  if (hasNegation ||
259  hasClauseWithAggregatedRelation(cyclicRelation, cur, &program, foundLiteral)) {
260  auto const& relSet = sccGraph.getInternalRelations(scc);
261  std::set<const Relation*, NameComparison> sortedRelSet(relSet.begin(), relSet.end());
262  // Negations and aggregations need to be stratified
263  std::string relationsListStr = toString(join(sortedRelSet, ",",
264  [](std::ostream& out, const Relation* r) { out << r->getQualifiedName(); }));
265  std::vector<DiagnosticMessage> messages;
266  messages.push_back(DiagnosticMessage(
267  "Relation " + toString(cur->getQualifiedName()), cur->getSrcLoc()));
268  std::string negOrAgg = hasNegation ? "negation" : "aggregation";
269  messages.push_back(
270  DiagnosticMessage("has cyclic " + negOrAgg, foundLiteral->getSrcLoc()));
271  report.addDiagnostic(Diagnostic(Diagnostic::Type::ERROR,
272  DiagnosticMessage("Unable to stratify relation(s) {" + relationsListStr + "}"),
273  messages));
274  break;
275  }
276  }
277  }
278  }

322  * Get unnamed variables except those that appear inside aggregates.
323  */
324 std::set<const UnnamedVariable*> getUnnamedVariables(const Node& node) {
325  std::set<const UnnamedVariable*> unnamedInAggregates;
326  visitDepthFirst(node, [&](const Aggregator& agg) {
327  visitDepthFirst(agg, [&](const UnnamedVariable& var) { unnamedInAggregates.insert(&var); });
328  });
329 
330  std::set<const UnnamedVariable*> unnamed;
331  visitDepthFirst(node, [&](const UnnamedVariable& var) {
332  if (!contains(unnamedInAggregates, &var)) {
333  unnamed.insert(&var);
334  }
335  });
336 
337  return unnamed;

342 void SemanticCheckerImpl::checkLiteral(const Literal& literal) {
343  // check potential nested atom
344  if (const auto* atom = as<Atom>(literal)) {
345  checkAtom(*atom);
346  }
347 
348  if (const auto* neg = as<Negation>(literal)) {
349  checkAtom(*neg->getAtom());
350  }
351 
352  if (const auto* constraint = as<BinaryConstraint>(literal)) {
353  checkArgument(*constraint->getLHS());
354  checkArgument(*constraint->getRHS());
355 
356  std::set<const UnnamedVariable*> unnamedInRecord;
357  visitDepthFirst(*constraint, [&](const RecordInit& record) {
358  for (auto* arg : record.getArguments()) {
359  if (auto* unnamed = as<UnnamedVariable>(arg)) {
360  unnamedInRecord.insert(unnamed);
361  }
362  }
363  });
364 
365  // Check if constraint contains unnamed variables.
366  for (auto* unnamed : getUnnamedVariables(*constraint)) {
367  if (!contains(unnamedInRecord, unnamed)) {
368  report.addError("Underscore in binary relation", unnamed->getSrcLoc());
369  }
370  }
371  }

374 /**
375  * agg1, agg2 are clauses which contain no head, and consist of a single literal
376  * that contains an aggregate.
377  * agg1 is dependent on agg2 if agg1 contains a variable which is grounded by agg2, and not by agg1.
378  */
379 bool SemanticCheckerImpl::isDependent(const Clause& agg1, const Clause& agg2) {
380  auto groundedInAgg1 = getGroundedTerms(tu, agg1);
381  auto groundedInAgg2 = getGroundedTerms(tu, agg2);
382  bool dependent = false;
383  // For each variable X in the first aggregate
384  visitDepthFirst(agg1, [&](const ast::Variable& searchVar) {
385  // Try to find the corresponding variable X in the second aggregate
386  // by string comparison
387  const ast::Variable* matchingVarPtr = nullptr;
388  visitDepthFirst(agg2, [&](const ast::Variable& var) {
389  if (var == searchVar) {
390  matchingVarPtr = &var;
391  return;
392  }
393  });
394  // If the variable occurs in both clauses (a match was found)
395  if (matchingVarPtr != nullptr) {
396  if (!groundedInAgg1[&searchVar] && groundedInAgg2[matchingVarPtr]) {
397  dependent = true;
398  }
399  }
400  });
401  return dependent;
402 }

404 void SemanticCheckerImpl::checkAggregator(const Aggregator& aggregator) {
405  auto& report = tu.getErrorReport();
406  const Program& program = tu.getProgram();
407  Clause dummyClauseAggregator;
408 
409  visitDepthFirst(program, [&](const Literal& parentLiteral) {
410  visitDepthFirst(parentLiteral, [&](const Aggregator& candidateAggregate) {
411  if (candidateAggregate != aggregator) {
412  return;
413  }
414  // Get the literal containing the aggregator and put it into a dummy clause
415  // so we can get information about groundedness
416  dummyClauseAggregator.addToBody(souffle::clone(&parentLiteral));
417  });
418  });
419 
420  visitDepthFirst(program, [&](const Literal& parentLiteral) {
421  visitDepthFirst(parentLiteral, [&](const Aggregator& /* otherAggregate */) {
422  // Create the other aggregate's dummy clause
423  Clause dummyClauseOther;
424  dummyClauseOther.addToBody(souffle::clone(&parentLiteral));
425  // Check dependency between the aggregator and this one
426  if (isDependent(dummyClauseAggregator, dummyClauseOther) &&
427  isDependent(dummyClauseOther, dummyClauseAggregator)) {
428  report.addError("Mutually dependent aggregate", aggregator.getSrcLoc());
429  }
430  });
431  });
432 
433  for (Literal* literal : aggregator.getBodyLiterals()) {
434  checkLiteral(*literal);
435  }
436 }

450 /**
451  * Check if the argument can be statically evaluated
452  * and thus in particular, if it should be allowed to appear as argument in facts.
453  **/
454 bool isConstantArgument(const Argument* arg) {
455  assert(arg != nullptr);
456 
457  if (isA<ast::Variable>(arg) || isA<UnnamedVariable>(arg)) {
458  return false;
459  } else if (isA<UserDefinedFunctor>(arg)) {
460  return false;
461  } else if (isA<Counter>(arg)) {
462  return false;
463  } else if (auto* typeCast = as<ast::TypeCast>(arg)) {
464  return isConstantArgument(typeCast->getValue());
465  } else if (auto* term = as<Term>(arg)) {
466  // Term covers intrinsic functor, records and adts. User-functors are handled earlier.
467  return all_of(term->getArguments(), isConstantArgument);
468  } else if (isA<Constant>(arg)) {
469  return true;
470  } else {
471  fatal("unsupported argument type: %s", typeid(arg).name());
472  }
473 }

499 void SemanticCheckerImpl::checkClause(const Clause& clause) {
500  // check head atom
501  checkAtom(*clause.getHead());
502 
503  // Check for absence of underscores in head
504  for (auto* unnamed : getUnnamedVariables(*clause.getHead())) {
505  report.addError("Underscore in head of rule", unnamed->getSrcLoc());
506  }
507 
508  // check body literals
509  for (Literal* lit : clause.getBodyLiterals()) {
510  checkLiteral(*lit);
511  }
512 
513  // check facts
514  if (isFact(clause)) {
515  checkFact(clause);
516  }
517 
518  // check whether named unnamed variables of the form _<ident>
519  // are only used once in a clause; if not, warnings will be
520  // issued.
521  std::map<std::string, int> var_count;
522  std::map<std::string, const ast::Variable*> var_pos;
523  visitDepthFirst(clause, [&](const ast::Variable& var) {
524  var_count[var.getName()]++;
525  var_pos[var.getName()] = &var;
526  });
527  for (const auto& cur : var_count) {
528  int numAppearances = cur.second;
529  const auto& varName = cur.first;
530  const auto& varLocation = var_pos[varName]->getSrcLoc();
531  if (varName[0] == '_') {
532  assert(varName.size() > 1 && "named variable should not be a single underscore");
533  if (numAppearances > 1) {
534  report.addWarning("Variable " + varName + " marked as singleton but occurs more than once",
535  varLocation);
536  }
537  }
538  }
539 
540  // check execution plan
541  if (clause.getExecutionPlan() != nullptr) {
542  auto numAtoms = getBodyLiterals<Atom>(clause).size();
543  for (const auto& cur : clause.getExecutionPlan()->getOrders()) {
544  bool isComplete = true;
545  auto order = cur.second->getOrder();
546  for (unsigned i = 1; i <= order.size(); i++) {
547  if (!contains(order, i)) {
548  isComplete = false;
549  break;
550  }
551  }
552  if (order.size() != numAtoms || !isComplete) {
553  report.addError("Invalid execution order in plan", cur.second->getSrcLoc());
554  }
555  }
556  }
557 
558  // check auto-increment
559  if (recursiveClauses.recursive(&clause)) {
560  visitDepthFirst(clause, [&](const Counter& ctr) {
561  report.addError("Auto-increment functor in a recursive rule", ctr.getSrcLoc());
562  });
563  }
564 }

566 void SemanticCheckerImpl::checkComplexRule(std::set<const Clause*> multiRule) {
567  std::map<std::string, int> var_count;
568  std::map<std::string, const ast::Variable*> var_pos;
569 
570  // Count the variable occurrence for the body of a
571  // complex rule only once.
572  // TODO (b-scholz): for negation / disjunction this is not quite
573  // right; we would need more semantic information here.
574  for (auto literal : (*multiRule.begin())->getBodyLiterals()) {
575  visitDepthFirst(*literal, [&](const ast::Variable& var) {
576  var_count[var.getName()]++;
577  var_pos[var.getName()] = &var;
578  });
579  }
580 
581  // Count variable occurrence for each head separately
582  for (auto clause : multiRule) {
583  visitDepthFirst(*(clause->getHead()), [&](const ast::Variable& var) {
584  var_count[var.getName()]++;
585  var_pos[var.getName()] = &var;
586  });
587  }
588 
589  // Check that a variables occurs more than once
590  for (const auto& cur : var_count) {
591  int numAppearances = cur.second;
592  const auto& varName = cur.first;
593  const auto& varLocation = var_pos[varName]->getSrcLoc();
594  if (varName[0] != '_' && numAppearances == 1) {
595  report.addWarning("Variable " + varName + " only occurs once", varLocation);
596  }

600 void SemanticCheckerImpl::checkRelationDeclaration(const Relation& relation) {
601  const auto& attributes = relation.getAttributes();
602  assert(attributes.size() == relation.getArity() && "mismatching attribute size and arity");
603 
604  for (size_t i = 0; i < relation.getArity(); i++) {
605  Attribute* attr = attributes[i];
606  auto&& typeName = attr->getTypeName();
607  auto* existingType = getIf(program.getTypes(),
608  [&](const ast::Type* type) { return type->getQualifiedName() == typeName; });
609 
610  /* check whether type exists */
611  if (!typeEnv.isPrimitiveType(typeName) && nullptr == existingType) {
612  report.addError(tfm::format("Undefined type in attribute %s", *attr), attr->getSrcLoc());
613  }
614 
615  /* check whether name occurs more than once */
616  for (size_t j = 0; j < i; j++) {
617  if (attr->getName() == attributes[j]->getName()) {
618  report.addError(tfm::format("Doubly defined attribute name %s", *attr), attr->getSrcLoc());
619  }
620  }

624 void SemanticCheckerImpl::checkRelation(const Relation& relation) {
625  if (relation.getRepresentation() == RelationRepresentation::EQREL) {
626  if (relation.getArity() == 2) {
627  const auto& attributes = relation.getAttributes();
628  assert(attributes.size() == 2 && "mismatching attribute size and arity");
629  if (attributes[0]->getTypeName() != attributes[1]->getTypeName()) {
630  report.addError("Domains of equivalence relation " + toString(relation.getQualifiedName()) +
631  " are different",
632  relation.getSrcLoc());
633  }
634  } else {
635  report.addError(
636  "Equivalence relation " + toString(relation.getQualifiedName()) + " is not binary",
637  relation.getSrcLoc());
638  }
639  }
640 
641  // start with declaration
642  checkRelationDeclaration(relation);

824 static const std::vector<SrcLocation> usesInvalidWitness(TranslationUnit& tu,
825  const std::vector<Literal*>& literals, const std::set<Own<Argument>>& groundedArguments) {
826  // Node-mapper that replaces aggregators with new (unique) variables
827  struct M : public NodeMapper {
828  // Variables introduced to replace aggregators
829  mutable std::set<std::string> aggregatorVariables;
830 
831  const std::set<std::string>& getAggregatorVariables() {
832  return aggregatorVariables;
833  }
834 
835  Own<Node> operator()(Own<Node> node) const override {
836  static int numReplaced = 0;
837  if (isA<Aggregator>(node.get())) {
838  // Replace the aggregator with a variable
839  std::stringstream newVariableName;
840  newVariableName << "+aggr_var_" << numReplaced++;
841 
842  // Keep track of which variables are bound to aggregators
843  aggregatorVariables.insert(newVariableName.str());
844 
845  return mk<ast::Variable>(newVariableName.str());
846  }
847  node->apply(*this);
848  return node;
849  }
850  };
851 
852  std::vector<SrcLocation> result;
853 
854  // Create two versions of the original clause
855 
856  // Clause 1 - will remain equivalent to the original clause in terms of variable groundedness
857  auto originalClause = mk<Clause>();
858  originalClause->setHead(mk<Atom>("*"));
859 
860  // Clause 2 - will have aggregators replaced with intrinsically grounded variables
861  auto aggregatorlessClause = mk<Clause>();
862  aggregatorlessClause->setHead(mk<Atom>("*"));
863 
864  // Construct both clauses in the same manner to match the original clause
865  // Must keep track of the subnode in Clause 1 that each subnode in Clause 2 matches to
866  std::map<const Argument*, const Argument*> identicalSubnodeMap;
867  for (const Literal* lit : literals) {
868  auto firstClone = souffle::clone(lit);
869  auto secondClone = souffle::clone(lit);
870 
871  // Construct the mapping between equivalent literal subnodes
872  std::vector<const Argument*> firstCloneArguments;
873  visitDepthFirst(*firstClone, [&](const Argument& arg) { firstCloneArguments.push_back(&arg); });
874 
875  std::vector<const Argument*> secondCloneArguments;
876  visitDepthFirst(*secondClone, [&](const Argument& arg) { secondCloneArguments.push_back(&arg); });
877 
878  for (size_t i = 0; i < firstCloneArguments.size(); i++) {
879  identicalSubnodeMap[secondCloneArguments[i]] = firstCloneArguments[i];
880  }
881 
882  // Actually add the literal clones to each clause
883  originalClause->addToBody(std::move(firstClone));
884  aggregatorlessClause->addToBody(std::move(secondClone));
885  }
886 
887  // Replace the aggregators in Clause 2 with variables
888  M update;
889  aggregatorlessClause->apply(update);
890 
891  // Create a dummy atom to force certain arguments to be grounded in the aggregatorlessClause
892  auto groundingAtomAggregatorless = mk<Atom>("grounding_atom");
893  auto groundingAtomOriginal = mk<Atom>("grounding_atom");
894 
895  // Force the new aggregator variables to be grounded in the aggregatorless clause
896  const std::set<std::string>& aggregatorVariables = update.getAggregatorVariables();
897  for (const std::string& str : aggregatorVariables) {
898  groundingAtomAggregatorless->addArgument(mk<ast::Variable>(str));
899  }
900 
901  // Force the given grounded arguments to be grounded in both clauses
902  for (const Own<Argument>& arg : groundedArguments) {
903  groundingAtomAggregatorless->addArgument(souffle::clone(arg));
904  groundingAtomOriginal->addArgument(souffle::clone(arg));
905  }
906 
907  aggregatorlessClause->addToBody(std::move(groundingAtomAggregatorless));
908  originalClause->addToBody(std::move(groundingAtomOriginal));
909 
910  // Compare the grounded analysis of both generated clauses
911  // All added arguments in Clause 2 were forced to be grounded, so if an ungrounded argument
912  // appears in Clause 2, it must also appear in Clause 1. Consequently, have two cases:
913  // - The argument is also ungrounded in Clause 1 - handled by another check
914  // - The argument is grounded in Clause 1 => the argument was grounded in the
915  // first clause somewhere along the line by an aggregator-body - not allowed!
916  std::set<Own<Argument>> newlyGroundedArguments;
917  auto originalGrounded = getGroundedTerms(tu, *originalClause);
918  for (auto&& pair : getGroundedTerms(tu, *aggregatorlessClause)) {
919  if (!pair.second && originalGrounded[identicalSubnodeMap[pair.first]]) {
920  result.push_back(pair.first->getSrcLoc());
921  }
922 
923  // Otherwise, it can now be considered grounded
924  newlyGroundedArguments.insert(souffle::clone(pair.first));
925  }
926 
927  // All previously grounded are still grounded
928  for (const Own<Argument>& arg : groundedArguments) {
929  newlyGroundedArguments.insert(souffle::clone(arg));
930  }
931 
932  // Everything on this level is fine, check subaggregators of each literal
933  for (const Literal* lit : literals) {
934  visitDepthFirst(*lit, [&](const Aggregator& aggr) {
935  // Check recursively if an invalid witness is used
936  for (auto&& argloc : usesInvalidWitness(tu, aggr.getBodyLiterals(), newlyGroundedArguments)) {
937  result.push_back(argloc);
938  }
939  });
940  }
941 
942  return result;
943 }

969 /**
970  * Find a cycle consisting entirely of inlined relations.
971  * If no cycle exists, then an empty vector is returned.
972  */
973 std::vector<QualifiedName> findInlineCycle(const PrecedenceGraphAnalysis& precedenceGraph,
974  std::map<const Relation*, const Relation*>& origins, const Relation* current, RelationSet& unvisited,
975  RelationSet& visiting, RelationSet& visited) {
976  std::vector<QualifiedName> result;
977 
978  if (current == nullptr) {
979  // Not looking at any nodes at the moment, so choose any node from the unvisited list
980 
981  if (unvisited.empty()) {
982  // Nothing left to visit - so no cycles exist!
983  return result;
984  }
985 
986  // Choose any element from the unvisited set
987  current = *unvisited.begin();
988  origins[current] = nullptr;
989 
990  // Move it to "currently visiting"
991  unvisited.erase(current);
992  visiting.insert(current);
993 
994  // Check if we can find a cycle beginning from this node
995  std::vector<QualifiedName> subresult =
996  findInlineCycle(precedenceGraph, origins, current, unvisited, visiting, visited);
997 
998  if (subresult.empty()) {
999  // No cycle found, try again from another node
1000  return findInlineCycle(precedenceGraph, origins, nullptr, unvisited, visiting, visited);
1001  } else {
1002  // Cycle found! Return it
1003  return subresult;
1004  }
1005  }
1006 
1007  // Check neighbours
1008  const RelationSet& successors = precedenceGraph.graph().successors(current);
1009  for (const Relation* successor : successors) {
1010  // Only care about inlined neighbours in the graph
1011  if (successor->hasQualifier(RelationQualifier::INLINE)) {
1012  if (visited.find(successor) != visited.end()) {
1013  // The neighbour has already been visited, so move on
1014  continue;
1015  }
1016 
1017  if (visiting.find(successor) != visiting.end()) {
1018  // Found a cycle!!
1019  // Construct the cycle in reverse
1020  while (current != nullptr) {
1021  result.push_back(current->getQualifiedName());
1022  current = origins[current];
1023  }
1024  return result;
1025  }
1026 
1027  // Node has not been visited yet
1028  origins[successor] = current;
1029 
1030  // Move from unvisited to visiting
1031  unvisited.erase(successor);
1032  visiting.insert(successor);
1033 
1034  // Visit recursively and check if a cycle is formed
1035  std::vector<QualifiedName> subgraphCycle =
1036  findInlineCycle(precedenceGraph, origins, successor, unvisited, visiting, visited);
1037 
1038  if (!subgraphCycle.empty()) {
1039  // Found a cycle!
1040  return subgraphCycle;
1041  }
1042  }
1043  }
1044 
1045  // Visited all neighbours with no cycle found, so done visiting this node.
1046  visiting.erase(current);
1047  visited.insert(current);
1048  return result;
1049 }

1051 void SemanticCheckerImpl::checkInlining() {
1052  auto isInline = [&](const Relation* rel) { return rel->hasQualifier(RelationQualifier::INLINE); };
1053 
1054  // Find all inlined relations
1055  RelationSet inlinedRelations;
1056  for (const auto& relation : program.getRelations()) {
1057  if (isInline(relation)) {
1058  inlinedRelations.insert(relation);
1059  if (ioTypes.isIO(relation)) {
1060  report.addError(
1061  "IO relation " + toString(relation->getQualifiedName()) + " cannot be inlined",
1062  relation->getSrcLoc());
1063  }
1064  }
1065  }
1066 
1067  // Check 1:
1068  // Let G' be the subgraph of the precedence graph G containing only those nodes
1069  // which are marked with the inline directive.
1070  // If G' contains a cycle, then inlining cannot be performed.
1071 
1072  RelationSet unvisited; // nodes that have not been visited yet
1073  RelationSet visiting; // nodes that we are currently visiting
1074  RelationSet visited; // nodes that have been completely explored
1075 
1076  // All nodes are initially unvisited
1077  for (const Relation* rel : inlinedRelations) {
1078  unvisited.insert(rel);
1079  }
1080 
1081  // Remember the parent node of each visited node to construct the found cycle
1082  std::map<const Relation*, const Relation*> origins;
1083 
1084  std::vector<QualifiedName> result =
1085  findInlineCycle(precedenceGraph, origins, nullptr, unvisited, visiting, visited);
1086 
1087  // If the result contains anything, then a cycle was found
1088  if (!result.empty()) {
1089  Relation* cycleOrigin = getRelation(program, result[result.size() - 1]);
1090 
1091  // Construct the string representation of the cycle
1092  std::stringstream cycle;
1093  cycle << "{" << cycleOrigin->getQualifiedName();
1094 
1095  // Print it backwards to preserve the initial cycle order
1096  for (int i = result.size() - 2; i >= 0; i--) {
1097  cycle << ", " << result[i];
1098  }
1099 
1100  cycle << "}";
1101 
1102  report.addError(
1103  "Cannot inline cyclically dependent relations " + cycle.str(), cycleOrigin->getSrcLoc());
1104  }
1105 
1106  // Check 2:
1107  // Cannot use the counter argument ('$') in inlined relations
1108 
1109  // Check if an inlined literal ever takes in a $
1110  visitDepthFirst(program, [&](const Atom& atom) {
1111  Relation* associatedRelation = getRelation(program, atom.getQualifiedName());
1112  if (associatedRelation != nullptr && isInline(associatedRelation)) {
1113  visitDepthFirst(atom, [&](const Argument& arg) {
1114  if (isA<Counter>(&arg)) {
1115  report.addError(
1116  "Cannot inline literal containing a counter argument '$'", arg.getSrcLoc());
1117  }
1118  });
1119  }
1120  });
1121 
1122  // Check if an inlined clause ever contains a $
1123  for (const Relation* rel : inlinedRelations) {
1124  for (Clause* clause : getClauses(program, *rel)) {
1125  visitDepthFirst(*clause, [&](const Argument& arg) {
1126  if (isA<Counter>(&arg)) {
1127  report.addError(
1128  "Cannot inline clause containing a counter argument '$'", arg.getSrcLoc());
1129  }
1130  });
1131  }
1132  }
1133 
1134  // Check 3:
1135  // Suppose the relation b is marked with the inline directive, but appears negated
1136  // in a clause. Then, if b introduces a new variable in its body, we cannot inline
1137  // the relation b.
1138 
1139  // Find all relations with the inline declarative that introduce new variables in their bodies
1140  RelationSet nonNegatableRelations;
1141  for (const Relation* rel : inlinedRelations) {
1142  bool foundNonNegatable = false;
1143  for (const Clause* clause : getClauses(program, *rel)) {
1144  // Get the variables in the head
1145  std::set<std::string> headVariables;
1146  visitDepthFirst(*clause->getHead(),
1147  [&](const ast::Variable& var) { headVariables.insert(var.getName()); });
1148 
1149  // Get the variables in the body
1150  std::set<std::string> bodyVariables;
1151  visitDepthFirst(clause->getBodyLiterals(),
1152  [&](const ast::Variable& var) { bodyVariables.insert(var.getName()); });
1153 
1154  // Check if all body variables are in the head
1155  // Do this separately to the above so only one error is printed per variable
1156  for (const std::string& var : bodyVariables) {
1157  if (headVariables.find(var) == headVariables.end()) {
1158  nonNegatableRelations.insert(rel);
1159  foundNonNegatable = true;
1160  break;
1161  }
1162  }
1163 
1164  if (foundNonNegatable) {
1165  break;
1166  }
1167  }
1168  }
1169 
1170  // Check that these relations never appear negated
1171  visitDepthFirst(program, [&](const Negation& neg) {
1172  Relation* associatedRelation = getRelation(program, neg.getAtom()->getQualifiedName());
1173  if (associatedRelation != nullptr &&
1174  nonNegatableRelations.find(associatedRelation) != nonNegatableRelations.end()) {
1175  report.addError(
1176  "Cannot inline negated relation which may introduce new variables", neg.getSrcLoc());
1177  }
1178  });
1179 
1180  // Check 4:
1181  // Don't support inlining atoms within aggregators at this point.
1182 
1183  // Reasoning: Suppose we have an aggregator like `max X: a(X)`, where `a` is inlined to `a1` and `a2`.
1184  // Then, `max X: a(X)` will become `max( max X: a1(X), max X: a2(X) )`. Suppose further that a(X) has
1185  // values X where it is true, while a2(X) does not. Then, the produced argument
1186  // `max( max X: a1(X), max X: a2(X) )` will not return anything (as one of its arguments fails), while
1187  // `max X: a(X)` will.
1188  // Can work around this with emptiness checks (e.g. `!a1(_), ... ; !a2(_), ... ; ...`)
1189 
1190  // This corner case prevents generalising aggregator inlining with the current set up.
1191 
1192  visitDepthFirst(program, [&](const Aggregator& aggr) {
1193  visitDepthFirst(aggr, [&](const Atom& subatom) {
1194  const Relation* rel = getRelation(program, subatom.getQualifiedName());
1195  if (rel != nullptr && isInline(rel)) {
1196  report.addError("Cannot inline relations that appear in aggregator", subatom.getSrcLoc());
1197  }
1198  });
1199  });
1200 
1201  // Check 5:
1202  // Suppose a relation `a` is inlined, appears negated in a clause, and contains a
1203  // (possibly nested) unnamed variable in its arguments. Then, the atom can't be
1204  // inlined, as unnamed variables are named during inlining (since they may appear
1205  // multiple times in an inlined-clause's body) => ungroundedness!
1206 
1207  // Exception: It's fine if the unnamed variable appears in a nested aggregator, as
1208  // the entire aggregator will automatically be grounded.
1209 
1210  // TODO (azreika): special case where all rules defined for `a` use the
1211  // underscored-argument exactly once: can workaround by remapping the variable
1212  // back to an underscore - involves changes to the actual inlining algo, though
1213 
1214  // Returns the pair (isValid, lastSrcLoc) where:
1215  // - isValid is true if and only if the node contains an invalid underscore, and
1216  // - lastSrcLoc is the source location of the last visited node
1217  std::function<std::pair<bool, SrcLocation>(const Node*)> checkInvalidUnderscore = [&](const Node* node) {
1218  if (isA<UnnamedVariable>(node)) {
1219  // Found an invalid underscore
1220  return std::make_pair(true, node->getSrcLoc());
1221  } else if (isA<Aggregator>(node)) {
1222  // Don't care about underscores within aggregators
1223  return std::make_pair(false, node->getSrcLoc());
1224  }
1225 
1226  // Check if any children nodes use invalid underscores
1227  for (const Node* child : node->getChildNodes()) {
1228  std::pair<bool, SrcLocation> childStatus = checkInvalidUnderscore(child);
1229  if (childStatus.first) {
1230  // Found an invalid underscore
1231  return childStatus;
1232  }
1233  }
1234 
1235  return std::make_pair(false, node->getSrcLoc());
1236  };
1237 
1238  // Perform the check
1239  visitDepthFirst(program, [&](const Negation& negation) {
1240  const Atom* associatedAtom = negation.getAtom();
