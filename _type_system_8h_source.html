209 /**
210  * @class AlgebraicDataType
211  * @brief Aggregates types using sums and products.
212  *
213  *
214  * Invariant: branches are in stored in lexicographical order.

216 class AlgebraicDataType : public Type {
217 public:
218  struct Branch {
219  std::string name; // < the name of the branch
220  std::vector<const Type*> types; // < Product type associated with this branch.
221 
222  void print(std::ostream& out) const {
223  out << tfm::format("%s {%s}", this->name,
224  join(types, ", ", [](std::ostream& out, const Type* t) { out << t->getName(); }));
225  }
226  };
227 
228  void print(std::ostream& out) const override {
229  out << tfm::format("%s = %s", name,
230  join(branches, " | ", [](std::ostream& out, const Branch& branch) { branch.print(out); }));

234  branches = std::move(bs);
235  std::sort(branches.begin(), branches.end(),
236  [](const Branch& left, const Branch& right) { return left.name < right.name; });
237  }
238 
239  const std::vector<const Type*>& getBranchTypes(const std::string& constructor) const {
240  for (auto& branch : branches) {
241  if (branch.name == constructor) return branch.types;
242  }
243  // Branch doesn't exist.
244  throw std::out_of_range("Trying to access non-existing branch.");
245  }
246 
247  /** Return the branches as a sorted vector */
248  const std::vector<Branch>& getBranches() const {
249  return branches;
250  }
251 
252 private:

260 /**
261  * A collection to represent sets of types. In addition to ordinary set capabilities
262  * it may also represent the set of all types -- without being capable of iterating over those.
263  *
264  * It is the basic entity to conduct sub- and super-type computations.
265  */
266 struct TypeSet {
267 public:
268  using const_iterator = IterDerefWrapper<typename std::set<const Type*>::const_iterator>;
269 
270  TypeSet(bool all = false) : all(all) {}
271 
272  template <typename... Types>
273  explicit TypeSet(const Types&... types) : all(false) {
274  for (const Type* cur : toVector<const Type*>(&types...)) {
275  this->types.insert(cur);
276  }
277  }
278 
279  TypeSet(const TypeSet& other) = default;
280  TypeSet(TypeSet&& other) = default;
281  TypeSet& operator=(const TypeSet& other) = default;
282  TypeSet& operator=(TypeSet&& other) = default;
283 
284  bool empty() const {
285  return !all && types.empty();
286  }
287 
288  /** Universality check */
289  bool isAll() const {
290  return all;
291  }
292 
293  /** Determines the size of this set unless it is the universal set */
294  std::size_t size() const {
295  assert(!all && "Unable to give size of universe.");
296  return types.size();
297  }
298 
299  /** Determines whether a given type is included or not */
300  bool contains(const Type& type) const {
301  return all || types.find(&type) != types.end();
302  }
303 
304  /** Adds the given type to this set */
305  void insert(const Type& type) {
306  if (!all) {
307  types.insert(&type);
308  }
309  }
310 
311  /** Calculate intersection of two TypeSet */
312  static TypeSet intersection(const TypeSet& left, const TypeSet& right) {
313  TypeSet result;
314 
315  if (left.isAll()) {
316  return right;
317  } else if (right.isAll()) {
318  return left;
319  }
320 
321  for (const auto& element : left) {
322  if (right.contains(element)) {
323  result.insert(element);
324  }
325  }
326 
327  return result;
328  }
329 
330  template <typename F>
331  TypeSet filter(TypeSet whenAll, F&& f) const {
332  if (all) return whenAll;
333 
334  TypeSet cpy;
335  for (auto&& t : *this)
336  if (f(t)) cpy.insert(t);
337  return cpy;
338  }
339 
340  /** Inserts all the types of the given set into this set */
341  void insert(const TypeSet& set) {
342  if (all) {
343  return;
344  }
345 
346  // if the other set is universal => make this one universal
347  if (set.isAll()) {
348  all = true;
349  types.clear();
350  return;
351  }
352 
353  // add types one by one
354  for (const auto& t : set) {
355  insert(t);
356  }
357  }
358 
359  /** Allows to iterate over the types contained in this set (only if not universal) */
360  const_iterator begin() const {
361  assert(!all && "Unable to enumerate universe.");
362  return derefIter(types.begin());
363  }
364 
365  /** Allows to iterate over the types contained in this set (only if not universal) */
366  const_iterator end() const {
367  assert(!all && "Unable to enumerate universe.");
368  return derefIter(types.end());
369  }
370 
371  /** Determines whether this set is a subset of the given set */
372  bool isSubsetOf(const TypeSet& b) const {
373  if (all) {
374  return b.isAll();
375  }
376  return all_of(*this, [&](const Type& cur) { return b.contains(cur); });
377  }
378 
379  /** Determines equality between type sets */
380  bool operator==(const TypeSet& other) const {
381  return all == other.all && types == other.types;
382  }
383 
384  /** Determines inequality between type sets */
385  bool operator!=(const TypeSet& other) const {
386  return !(*this == other);
387  }
388 
389  /** Adds print support for type sets */
390  void print(std::ostream& out) const {
391  if (all) {
392  out << "{ - all types - }";
393  } else {
394  out << "{"
395  << join(types, ",", [](std::ostream& out, const Type* type) { out << type->getName(); })
396  << "}";
397  }
398  }
399 
400  friend std::ostream& operator<<(std::ostream& out, const TypeSet& set) {
401  set.print(out);
402  return out;
403  }
404 
405 private:
406  /** True if it is the all-types set, false otherwise */
407  bool all;
408 
409  /** The enumeration of types in case it is not the all-types set */
410  std::set<const Type*, deref_less<Type>> types;
411 };
