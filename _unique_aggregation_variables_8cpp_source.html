28 bool UniqueAggregationVariablesTransformer::transform(TranslationUnit& translationUnit) {
29  bool changed = false;
30 
31  // make variables in aggregates unique
32  int aggNumber = 0;
33  Program& program = translationUnit.getProgram();
34  visitDepthFirstPostOrder(program, [&](const Aggregator& agg) {
35  // only applicable for aggregates with target expression
36  if (agg.getTargetExpression() == nullptr) {
37  return;
38  }
39 
40  // get all variables in the target expression
41  std::set<std::string> names;
42  visitDepthFirst(
43  *agg.getTargetExpression(), [&](const ast::Variable& var) { names.insert(var.getName()); });
44 
45  // rename them
46  visitDepthFirst(agg, [&](const ast::Variable& var) {
47  auto pos = names.find(var.getName());
48  if (pos == names.end()) {
49  return;
50  }
51  const_cast<ast::Variable&>(var).setName(" " + var.getName() + toString(aggNumber));
52  changed = true;
53  });
54 
55  // increment aggregation number
56  aggNumber++;
57  });
58  return changed;
59 }
