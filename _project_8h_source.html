35 /**
36  * @class Project
37  * @brief Project a result into the target relation.
38  *
39  * For example:
40  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
41  * FOR t0 IN A
42  * ...
43  * PROJECT (t0.a, t0.b, t0.c) INTO @new_X
44  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
45  */
46 class Project : public Operation {
47 public:
48  Project(Own<RelationReference> relRef, VecOwn<Expression> expressions)
49  : relationRef(std::move(relRef)), expressions(std::move(expressions)) {
50  assert(relationRef != nullptr && "Relation reference is a null-pointer");
51  for (auto const& expr : expressions) {
52  assert(expr != nullptr && "Expression is a null-pointer");
53  }
54  }
55 
56  /** @brief Get relation */
57  const Relation& getRelation() const {
58  return *relationRef->get();
59  }
60 
61  /** @brief Get expressions */
62  std::vector<Expression*> getValues() const {
63  return toPtrVector(expressions);
64  }
65 
66  std::vector<const Node*> getChildNodes() const override {
67  std::vector<const Node*> res;
68  res.push_back(relationRef.get());
69  for (const auto& expr : expressions) {
70  res.push_back(expr.get());
71  }
72  return res;
73  }
74 
75  Project* clone() const override {
76  VecOwn<Expression> newValues;
77  for (auto& expr : expressions) {
78  newValues.emplace_back(expr->clone());
79  }
80  return new Project(souffle::clone(relationRef), std::move(newValues));
81  }
82 
83  void apply(const NodeMapper& map) override {
84  relationRef = map(std::move(relationRef));
85  for (auto& expr : expressions) {
86  expr = map(std::move(expr));
87  }
88  }
89 
90 protected:
91  void print(std::ostream& os, int tabpos) const override {
92  os << times(" ", tabpos);
93  os << "PROJECT (" << join(expressions, ", ", print_deref<Own<Expression>>()) << ") INTO "
94  << getRelation().getName() << std::endl;
95  }
96 
97  bool equal(const Node& node) const override {
98  const auto& other = static_cast<const Project&>(node);
99  return equal_ptr(relationRef, other.relationRef) && equal_targets(expressions, other.expressions);
100  }
101 
102  /** Relation that values are projected into */
103  Own<RelationReference> relationRef;
104 
105  /* Values (expressions) for projection */
106  VecOwn<Expression> expressions;
107 };
