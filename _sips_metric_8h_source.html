30 /**
31  * Class for SIPS cost-metric functions
32  * Each subclass represents a different heuristic used for evaluating
33  * the cost of choosing an atom next in the schedule.
34  */
35 class SipsMetric {
36 public:
37  virtual ~SipsMetric() = default;
38 
39  /**
40  * Determines the new ordering of a clause after the SIPS is applied.
41  * @param clause clause to reorder
42  * @return the vector of new positions; v[i] = j iff atom j moves to pos i
43  */
44  std::vector<unsigned int> getReordering(const Clause* clause) const;
45 
46 protected:
47  /**
48  * Evaluates the cost of choosing each atom next in the current schedule
49  * @param atoms atoms to choose from; may be nullptr
50  * @param bindingStore the variables already bound to a value
51  */
52  virtual std::vector<double> evaluateCosts(
53  const std::vector<Atom*> atoms, const BindingStore& bindingStore) const = 0;
54 };
