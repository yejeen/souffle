37  * @class Aggregator
38  * @brief Defines the aggregator class
39  *
40  * Example:

43  * Aggregates over a sub-query using an aggregate operator

48  Aggregator(AggregateOp fun, Own<Argument> expr = nullptr, VecOwn<Literal> body = {}, SrcLocation loc = {})
49  : Argument(std::move(loc)), fun(fun), targetExpression(std::move(expr)), body(std::move(body)) {}

74  }

76  std::vector<const Node*> getChildNodes() const override {
77  auto res = Argument::getChildNodes();
78  if (targetExpression) {
79  res.push_back(targetExpression.get());
80  }
81  for (auto& cur : body) {
82  res.push_back(cur.get());
83  }
84  return res;
85  }

89  }
90 
91  void apply(const NodeMapper& map) override {
92  if (targetExpression) {
93  targetExpression = map(std::move(targetExpression));
94  }
95  for (auto& cur : body) {
96  cur = map(std::move(cur));
97  }
98  }
99 
100 protected:

102  os << fun;
103  if (targetExpression) {
104  os << " " << *targetExpression;
105  }
106  os << " : { " << join(body) << " }";
107  }
108 
109  bool equal(const Node& node) const override {
110  const auto& other = static_cast<const Aggregator&>(node);
111  return fun == other.fun && equal_ptr(targetExpression, other.targetExpression) &&
112  equal_targets(body, other.body);
113  }
