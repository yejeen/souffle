102 /**
103  * The generic base type of all RamVisitors realizing the dispatching of
104  * visitor calls. Each visitor may define a return type R and a list of
105  * extra parameters to be passed along with the visited Nodes to the
106  * corresponding visitor function.
107  *
108  * @tparam R the result type produced by a visit call
109  * @tparam Params extra parameters to be passed to the visit call
110  */

112 struct Visitor : public ram_visitor_tag {
113  /** A virtual destructor */

126  /**
127  * The main entry for a visit process conducting the dispatching of
128  * a visit to the various sub-types of Nodes. Sub-classes may override
129  * this implementation to conduct pre-visit operations.

132  * before their super-classes; otherwise sub-classes cannot be visited.

323 /**
324  * A utility function visiting all nodes within the RAM fragment rooted by the given node
325  * recursively in a depth-first pre-order fashion applying the given visitor to each
326  * encountered node.
327  *
328  * @param root the root of the RAM fragment to be visited
329  * @param visitor the visitor to be applied on each node
330  * @param args a list of extra parameters to be forwarded to the visitor
331  */
332 template <typename R, typename... Ps, typename... Args>
333 void visitDepthFirstPreOrder(const Node& root, Visitor<R, Ps...>& visitor, Args&... args) {
334  visitor(root, args...);
335  for (const Node* cur : root.getChildNodes()) {
336  if (cur != nullptr) {
337  visitDepthFirstPreOrder(*cur, visitor, args...);
338  }
339  }
340 }

358 /**
359  * A specialized visitor wrapping a lambda function -- an auxiliary type required
360  * for visitor convenience functions.
361  */
362 template <typename R, typename N>
363 struct LambdaVisitor : public Visitor<void> {
364  std::function<R(const N&)> lambda;
365  LambdaVisitor(std::function<R(const N&)> lambda) : lambda(std::move(lambda)) {}
366  void visit(const Node& node) override {
367  if (const auto* n = dynamic_cast<const N*>(&node)) {
368  lambda(*n);
369  }
370  }
371 };
372 
373 /**
374  * A factory function for creating LambdaVisitor instances.
375  */
376 template <typename R, typename N>
377 LambdaVisitor<R, N> makeLambdaVisitor(const std::function<R(const N&)>& fun) {
378  return LambdaVisitor<R, N>(fun);
379 }
